22 LogicalResult
23 GPUFuncOpLowering::matchAndRewrite(gpu::GPUFuncOp gpuFuncOp, OpAdaptor adaptor,
24  ConversionPatternRewriter &rewriter) const {
25  Location loc = gpuFuncOp.getLoc();
26 
27  SmallVector<LLVM::GlobalOp, 3> workgroupBuffers;
28  workgroupBuffers.reserve(gpuFuncOp.getNumWorkgroupAttributions());
29  for (const auto &en : llvm::enumerate(gpuFuncOp.getWorkgroupAttributions())) {
30  BlockArgument attribution = en.value();
31 
32  auto type = dyn_cast<MemRefType>(attribution.getType());
33  assert(type && type.hasStaticShape() && "unexpected type in attribution");
34 
35  uint64_t numElements = type.getNumElements();
36 
37  auto elementType =
38  cast<Type>(typeConverter->convertType(type.getElementType()));
39  auto arrayType = LLVM::LLVMArrayType::get(elementType, numElements);
40  std::string name = std::string(
41  llvm::formatv("__wg_{0}_{1}", gpuFuncOp.getName(), en.index()));
42  uint64_t alignment = 0;
43  if (auto alignAttr =
44  dyn_cast_or_null<IntegerAttr>(gpuFuncOp.getWorkgroupAttributionAttr(
45  en.index(), LLVM::LLVMDialect::getAlignAttrName())))
46  alignment = alignAttr.getInt();
47  auto globalOp = rewriter.create<LLVM::GlobalOp>(
48  gpuFuncOp.getLoc(), arrayType, /*isConstant=*/false,
49  LLVM::Linkage::Internal, name, /*value=*/Attribute(), alignment,
50  workgroupAddrSpace);
51  workgroupBuffers.push_back(globalOp);
52  }
53 
54  // Remap proper input types.
55  TypeConverter::SignatureConversion signatureConversion(
56  gpuFuncOp.front().getNumArguments());
57 
58  Type funcType = getTypeConverter()->convertFunctionSignature(
59  gpuFuncOp.getFunctionType(), /*isVariadic=*/false,
60  getTypeConverter()->getOptions().useBarePtrCallConv, signatureConversion);
61  if (!funcType) {
62  return rewriter.notifyMatchFailure(gpuFuncOp, [&](Diagnostic &diag) {
63  diag << "failed to convert function signature type for: "
64  << gpuFuncOp.getFunctionType();
65  });
66  }
67 
68  // Create the new function operation. Only copy those attributes that are
69  // not specific to function modeling.
70  SmallVector<NamedAttribute, 4> attributes;
71  ArrayAttr argAttrs;
72  for (const auto &attr : gpuFuncOp->getAttrs()) {
73  if (attr.getName() == SymbolTable::getSymbolAttrName() ||
74  attr.getName() == gpuFuncOp.getFunctionTypeAttrName() ||
75  attr.getName() ==
76  gpu::GPUFuncOp::getNumWorkgroupAttributionsAttrName() ||
77  attr.getName() == gpuFuncOp.getWorkgroupAttribAttrsAttrName() ||
78  attr.getName() == gpuFuncOp.getPrivateAttribAttrsAttrName())
79  continue;
80  if (attr.getName() == gpuFuncOp.getArgAttrsAttrName()) {
81  argAttrs = gpuFuncOp.getArgAttrsAttr();
82  continue;
83  }
84  attributes.push_back(attr);
85  }
86  // Add a dialect specific kernel attribute in addition to GPU kernel
87  // attribute. The former is necessary for further translation while the
88  // latter is expected by gpu.launch_func.
89  if (gpuFuncOp.isKernel())
90  attributes.emplace_back(kernelAttributeName, rewriter.getUnitAttr());
91  auto llvmFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
92  gpuFuncOp.getLoc(), gpuFuncOp.getName(), funcType,
93  LLVM::Linkage::External, /*dsoLocal=*/false, /*cconv=*/LLVM::CConv::C,
94  /*comdat=*/nullptr, attributes);
95 
96  {
97  // Insert operations that correspond to converted workgroup and private
98  // memory attributions to the body of the function. This must operate on
99  // the original function, before the body region is inlined in the new
100  // function to maintain the relation between block arguments and the
101  // parent operation that assigns their semantics.
102  OpBuilder::InsertionGuard guard(rewriter);
103 
104  // Rewrite workgroup memory attributions to addresses of global buffers.
105  rewriter.setInsertionPointToStart(&gpuFuncOp.front());
106  unsigned numProperArguments = gpuFuncOp.getNumArguments();
107 
108  for (const auto &en : llvm::enumerate(workgroupBuffers)) {
109  LLVM::GlobalOp global = en.value();
110  auto ptrType = LLVM::LLVMPointerType::get(rewriter.getContext(),
111  global.getAddrSpace());
112  Value address = rewriter.create<LLVM::AddressOfOp>(
113  loc, ptrType, global.getSymNameAttr());
114  Value memory =
115  rewriter.create<LLVM::GEPOp>(loc, ptrType, global.getType(), address,
116  ArrayRef<LLVM::GEPArg>{0, 0});
117 
118  // Build a memref descriptor pointing to the buffer to plug with the
119  // existing memref infrastructure. This may use more registers than
120  // otherwise necessary given that memref sizes are fixed, but we can try
121  // and canonicalize that away later.
122  Value attribution = gpuFuncOp.getWorkgroupAttributions()[en.index()];
123  auto type = cast<MemRefType>(attribution.getType());
124  auto descr = MemRefDescriptor::fromStaticShape(
125  rewriter, loc, *getTypeConverter(), type, memory);
126  signatureConversion.remapInput(numProperArguments + en.index(), descr);
127  }
128 
129  // Rewrite private memory attributions to alloca'ed buffers.
130  unsigned numWorkgroupAttributions = gpuFuncOp.getNumWorkgroupAttributions();
131  auto int64Ty = IntegerType::get(rewriter.getContext(), 64);
132  for (const auto &en : llvm::enumerate(gpuFuncOp.getPrivateAttributions())) {
133  Value attribution = en.value();
134  auto type = cast<MemRefType>(attribution.getType());
135  assert(type && type.hasStaticShape() && "unexpected type in attribution");
136 
137  // Explicitly drop memory space when lowering private memory
138  // attributions since NVVM models it as `alloca`s in the default
139  // memory space and does not support `alloca`s with addrspace(5).
140  Type elementType = typeConverter->convertType(type.getElementType());
141  auto ptrType =
142  LLVM::LLVMPointerType::get(rewriter.getContext(), allocaAddrSpace);
143  Value numElements = rewriter.create<LLVM::ConstantOp>(
144  gpuFuncOp.getLoc(), int64Ty, type.getNumElements());
145  uint64_t alignment = 0;
146  if (auto alignAttr =
147  dyn_cast_or_null<IntegerAttr>(gpuFuncOp.getPrivateAttributionAttr(
148  en.index(), LLVM::LLVMDialect::getAlignAttrName())))
149  alignment = alignAttr.getInt();
150  Value allocated = rewriter.create<LLVM::AllocaOp>(
151  gpuFuncOp.getLoc(), ptrType, elementType, numElements, alignment);
152  auto descr = MemRefDescriptor::fromStaticShape(
153  rewriter, loc, *getTypeConverter(), type, allocated);
154  signatureConversion.remapInput(
155  numProperArguments + numWorkgroupAttributions + en.index(), descr);
156  }
157  }
158 
159  // Move the region to the new function, update the entry block signature.
160  rewriter.inlineRegionBefore(gpuFuncOp.getBody(), llvmFuncOp.getBody(),
161  llvmFuncOp.end());
162  if (failed(rewriter.convertRegionTypes(&llvmFuncOp.getBody(), *typeConverter,
163  &signatureConversion)))
164  return failure();
165 
166  // If bare memref pointers are being used, remap them back to memref
167  // descriptors This must be done after signature conversion to get rid of the
168  // unrealized casts.
169  if (getTypeConverter()->getOptions().useBarePtrCallConv) {
170  OpBuilder::InsertionGuard guard(rewriter);
171  rewriter.setInsertionPointToStart(&llvmFuncOp.getBody().front());
172  for (const auto &en : llvm::enumerate(gpuFuncOp.getArgumentTypes())) {
173  auto memrefTy = dyn_cast<MemRefType>(en.value());
174  if (!memrefTy)
175  continue;
176  assert(memrefTy.hasStaticShape() &&
177  "Bare pointer convertion used with dynamically-shaped memrefs");
178  // Use a placeholder when replacing uses of the memref argument to prevent
179  // circular replacements.
180  auto remapping = signatureConversion.getInputMapping(en.index());
181  assert(remapping && remapping->size == 1 &&
182  "Type converter should produce 1-to-1 mapping for bare memrefs");
183  BlockArgument newArg =
184  llvmFuncOp.getBody().getArgument(remapping->inputNo);
185  auto placeholder = rewriter.create<LLVM::UndefOp>(
186  loc, getTypeConverter()->convertType(memrefTy));
187  rewriter.replaceUsesOfBlockArgument(newArg, placeholder);
188  Value desc = MemRefDescriptor::fromStaticShape(
189  rewriter, loc, *getTypeConverter(), memrefTy, newArg);
190  rewriter.replaceOp(placeholder, {desc});
191  }
192  }
193 
194  // Get memref type from function arguments and set the noalias to
195  // pointer arguments.
196  for (const auto &en : llvm::enumerate(gpuFuncOp.getArgumentTypes())) {
197  auto memrefTy = en.value().dyn_cast<MemRefType>();
198  NamedAttrList argAttr = argAttrs
199  ? argAttrs[en.index()].cast<DictionaryAttr>()
200  : NamedAttrList();
201 
202  auto copyPointerAttribute = [&](StringRef attrName) {
203  Attribute attr = argAttr.erase(attrName);
204 
205  // This is a proxy for the bare pointer calling convention.
206  if (!attr)
207  return;
208  auto remapping = signatureConversion.getInputMapping(en.index());
209  if (remapping->size > 1 &&
210  attrName == LLVM::LLVMDialect::getNoAliasAttrName()) {
211  emitWarning(llvmFuncOp.getLoc(),
212  "Cannot copy noalias with non-bare pointers.\n");
213  return;
214  }
215  for (size_t i = 0, e = remapping->size; i < e; ++i) {
216  if (llvmFuncOp.getArgument(remapping->inputNo + i)
217  .getType()
218  .isa<LLVM::LLVMPointerType>()) {
219  llvmFuncOp.setArgAttr(remapping->inputNo + i, attrName, attr);
220  }
221  }
222  };
223 
224  if (argAttr.empty())
225  continue;
226 
227  if (memrefTy) {
228  copyPointerAttribute(LLVM::LLVMDialect::getNoAliasAttrName());
229  copyPointerAttribute(LLVM::LLVMDialect::getReadonlyAttrName());
230  copyPointerAttribute(LLVM::LLVMDialect::getWriteOnlyAttrName());
231  copyPointerAttribute(LLVM::LLVMDialect::getNonNullAttrName());
232  copyPointerAttribute(LLVM::LLVMDialect::getDereferenceableAttrName());
233  copyPointerAttribute(
234  LLVM::LLVMDialect::getDereferenceableOrNullAttrName());
235  }
236  }
237  rewriter.eraseOp(gpuFuncOp);
238  return success();
239 }

241 static SmallString<16> getUniqueFormatGlobalName(gpu::GPUModuleOp moduleOp) {
242  const char formatStringPrefix[] = "printfFormat_";
243  // Get a unique global name.
244  unsigned stringNumber = 0;
245  SmallString<16> stringConstName;
246  do {
247  stringConstName.clear();

249  } while (moduleOp.lookupSymbol(stringConstName));

268 LogicalResult GPUPrintfOpToHIPLowering::matchAndRewrite(
269  gpu::PrintfOp gpuPrintfOp, gpu::PrintfOpAdaptor adaptor,
270  ConversionPatternRewriter &rewriter) const {
271  Location loc = gpuPrintfOp->getLoc();
272 
273  mlir::Type llvmI8 = typeConverter->convertType(rewriter.getI8Type());
274  auto ptrType = LLVM::LLVMPointerType::get(rewriter.getContext());
275  mlir::Type llvmI32 = typeConverter->convertType(rewriter.getI32Type());
276  mlir::Type llvmI64 = typeConverter->convertType(rewriter.getI64Type());
277  // Note: this is the GPUModule op, not the ModuleOp that surrounds it
278  // This ensures that global constants and declarations are placed within
279  // the device code, not the host code
280  auto moduleOp = gpuPrintfOp->getParentOfType<gpu::GPUModuleOp>();
281 
282  auto ocklBegin =
283  getOrDefineFunction(moduleOp, loc, rewriter, "__ockl_printf_begin",
284  LLVM::LLVMFunctionType::get(llvmI64, {llvmI64}));
285  LLVM::LLVMFuncOp ocklAppendArgs;
286  if (!adaptor.getArgs().empty()) {
287  ocklAppendArgs = getOrDefineFunction(
288  moduleOp, loc, rewriter, "__ockl_printf_append_args",
289  LLVM::LLVMFunctionType::get(
290  llvmI64, {llvmI64, /*numArgs*/ llvmI32, llvmI64, llvmI64, llvmI64,
291  llvmI64, llvmI64, llvmI64, llvmI64, /*isLast*/ llvmI32}));
292  }
293  auto ocklAppendStringN = getOrDefineFunction(
294  moduleOp, loc, rewriter, "__ockl_printf_append_string_n",
295  LLVM::LLVMFunctionType::get(
296  llvmI64,
297  {llvmI64, ptrType, /*length (bytes)*/ llvmI64, /*isLast*/ llvmI32}));
298 
299  /// Start the printf hostcall
300  Value zeroI64 = rewriter.create<LLVM::ConstantOp>(loc, llvmI64, 0);
301  auto printfBeginCall = rewriter.create<LLVM::CallOp>(loc, ocklBegin, zeroI64);
302  Value printfDesc = printfBeginCall.getResult();
303 
304  // Get a unique global name for the format.
305  SmallString<16> stringConstName = getUniqueFormatGlobalName(moduleOp);
306 
307  llvm::SmallString<20> formatString(adaptor.getFormat());
308  formatString.push_back('\0'); // Null terminate for C
309  size_t formatStringSize = formatString.size_in_bytes();
310 
311  auto globalType = LLVM::LLVMArrayType::get(llvmI8, formatStringSize);
312  LLVM::GlobalOp global;
313  {
314  ConversionPatternRewriter::InsertionGuard guard(rewriter);
315  rewriter.setInsertionPointToStart(moduleOp.getBody());
316  global = rewriter.create<LLVM::GlobalOp>(
317  loc, globalType,
318  /*isConstant=*/true, LLVM::Linkage::Internal, stringConstName,
319  rewriter.getStringAttr(formatString));
320  }
321 
322  // Get a pointer to the format string's first element and pass it to printf()
323  Value globalPtr = rewriter.create<LLVM::AddressOfOp>(
324  loc,
325  LLVM::LLVMPointerType::get(rewriter.getContext(), global.getAddrSpace()),
326  global.getSymNameAttr());
327  Value stringStart = rewriter.create<LLVM::GEPOp>(
328  loc, ptrType, globalType, globalPtr, ArrayRef<LLVM::GEPArg>{0, 0});
329  Value stringLen =
330  rewriter.create<LLVM::ConstantOp>(loc, llvmI64, formatStringSize);
331 
332  Value oneI32 = rewriter.create<LLVM::ConstantOp>(loc, llvmI32, 1);
333  Value zeroI32 = rewriter.create<LLVM::ConstantOp>(loc, llvmI32, 0);
334 
335  auto appendFormatCall = rewriter.create<LLVM::CallOp>(
336  loc, ocklAppendStringN,
337  ValueRange{printfDesc, stringStart, stringLen,
338  adaptor.getArgs().empty() ? oneI32 : zeroI32});
339  printfDesc = appendFormatCall.getResult();
340 
341  // __ockl_printf_append_args takes 7 values per append call
342  constexpr size_t argsPerAppend = 7;
343  size_t nArgs = adaptor.getArgs().size();
344  for (size_t group = 0; group < nArgs; group += argsPerAppend) {
345  size_t bound = std::min(group + argsPerAppend, nArgs);
346  size_t numArgsThisCall = bound - group;
347 
348  SmallVector<mlir::Value, 2 + argsPerAppend + 1> arguments;
349  arguments.push_back(printfDesc);
350  arguments.push_back(
351  rewriter.create<LLVM::ConstantOp>(loc, llvmI32, numArgsThisCall));
352  for (size_t i = group; i < bound; ++i) {
353  Value arg = adaptor.getArgs()[i];
354  if (auto floatType = dyn_cast<FloatType>(arg.getType())) {
355  if (!floatType.isF64())
356  arg = rewriter.create<LLVM::FPExtOp>(
357  loc, typeConverter->convertType(rewriter.getF64Type()), arg);
358  arg = rewriter.create<LLVM::BitcastOp>(loc, llvmI64, arg);
359  }
360  if (arg.getType().getIntOrFloatBitWidth() != 64)
361  arg = rewriter.create<LLVM::ZExtOp>(loc, llvmI64, arg);
362 
363  arguments.push_back(arg);
364  }
365  // Pad out to 7 arguments since the hostcall always needs 7
366  for (size_t extra = numArgsThisCall; extra < argsPerAppend; ++extra) {
367  arguments.push_back(zeroI64);
368  }
369 
370  auto isLast = (bound == nArgs) ? oneI32 : zeroI32;
371  arguments.push_back(isLast);
372  auto call = rewriter.create<LLVM::CallOp>(loc, ocklAppendArgs, arguments);
373  printfDesc = call.getResult();
374  }
375  rewriter.eraseOp(gpuPrintfOp);
376  return success();
377 }

436 LogicalResult GPUPrintfOpToVPrintfLowering::matchAndRewrite(
437  gpu::PrintfOp gpuPrintfOp, gpu::PrintfOpAdaptor adaptor,
438  ConversionPatternRewriter &rewriter) const {
439  Location loc = gpuPrintfOp->getLoc();
440 
441  mlir::Type llvmI8 = typeConverter->convertType(rewriter.getIntegerType(8));
442  mlir::Type ptrType = LLVM::LLVMPointerType::get(rewriter.getContext());
443 
444  // Note: this is the GPUModule op, not the ModuleOp that surrounds it
445  // This ensures that global constants and declarations are placed within
446  // the device code, not the host code
447  auto moduleOp = gpuPrintfOp->getParentOfType<gpu::GPUModuleOp>();
448 
449  auto vprintfType =
450  LLVM::LLVMFunctionType::get(rewriter.getI32Type(), {ptrType, ptrType});
451  LLVM::LLVMFuncOp vprintfDecl =
452  getOrDefineFunction(moduleOp, loc, rewriter, "vprintf", vprintfType);
453 
454  // Get a unique global name for the format.
455  SmallString<16> stringConstName = getUniqueFormatGlobalName(moduleOp);
456 
457  llvm::SmallString<20> formatString(adaptor.getFormat());
458  formatString.push_back('\0'); // Null terminate for C
459  auto globalType =
460  LLVM::LLVMArrayType::get(llvmI8, formatString.size_in_bytes());
461  LLVM::GlobalOp global;
462  {
463  ConversionPatternRewriter::InsertionGuard guard(rewriter);
464  rewriter.setInsertionPointToStart(moduleOp.getBody());
465  global = rewriter.create<LLVM::GlobalOp>(
466  loc, globalType,
467  /*isConstant=*/true, LLVM::Linkage::Internal, stringConstName,
468  rewriter.getStringAttr(formatString), /*allignment=*/0);
469  }
470 
471  // Get a pointer to the format string's first element
472  Value globalPtr = rewriter.create<LLVM::AddressOfOp>(loc, global);
473  Value stringStart = rewriter.create<LLVM::GEPOp>(
474  loc, ptrType, globalType, globalPtr, ArrayRef<LLVM::GEPArg>{0, 0});
475  SmallVector<Type> types;
476  SmallVector<Value> args;
477  // Promote and pack the arguments into a stack allocation.
478  for (Value arg : adaptor.getArgs()) {
479  Type type = arg.getType();
480  Value promotedArg = arg;
481  assert(type.isIntOrFloat());
482  if (isa<FloatType>(type)) {
483  type = rewriter.getF64Type();
484  promotedArg = rewriter.create<LLVM::FPExtOp>(loc, type, arg);
485  }
486  types.push_back(type);
487  args.push_back(promotedArg);
488  }
489  Type structType =
490  LLVM::LLVMStructType::getLiteral(gpuPrintfOp.getContext(), types);
491  Value one = rewriter.create<LLVM::ConstantOp>(loc, rewriter.getI64Type(),
492  rewriter.getIndexAttr(1));
493  Value tempAlloc =
494  rewriter.create<LLVM::AllocaOp>(loc, ptrType, structType, one,
495  /*alignment=*/0);
496  for (auto [index, arg] : llvm::enumerate(args)) {
497  Value ptr = rewriter.create<LLVM::GEPOp>(
498  loc, ptrType, structType, tempAlloc, ArrayRef<LLVM::GEPArg>{0, index});
499  rewriter.create<LLVM::StoreOp>(loc, arg, ptr);
500  }
501  std::array<Value, 2> printfArgs = {stringStart, tempAlloc};
502 
503  rewriter.create<LLVM::CallOp>(loc, vprintfDecl, printfArgs);
504  rewriter.eraseOp(gpuPrintfOp);
505  return success();
506 }

508 /// Unrolls op if it's operating on vectors.
509 LogicalResult impl::scalarizeVectorOp(Operation *op, ValueRange operands,
510  ConversionPatternRewriter &rewriter,
511  const LLVMTypeConverter &converter) {
512  TypeRange operandTypes(operands);
513  if (llvm::none_of(operandTypes,
514  [](Type type) { return isa<VectorType>(type); })) {
515  return rewriter.notifyMatchFailure(op, "expected vector operand");
516  }
517  if (op->getNumRegions() != 0 || op->getNumSuccessors() != 0)
518  return rewriter.notifyMatchFailure(op, "expected no region/successor");
519  if (op->getNumResults() != 1)
520  return rewriter.notifyMatchFailure(op, "expected single result");
521  VectorType vectorType = dyn_cast<VectorType>(op->getResult(0).getType());
522  if (!vectorType)
523  return rewriter.notifyMatchFailure(op, "expected vector result");
524 
525  Location loc = op->getLoc();
526  Value result = rewriter.create<LLVM::UndefOp>(loc, vectorType);
527  Type indexType = converter.convertType(rewriter.getIndexType());
528  StringAttr name = op->getName().getIdentifier();
529  Type elementType = vectorType.getElementType();
530 
531  for (int64_t i = 0; i < vectorType.getNumElements(); ++i) {
532  Value index = rewriter.create<LLVM::ConstantOp>(loc, indexType, i);
533  auto extractElement = [&](Value operand) -> Value {
534  if (!isa<VectorType>(operand.getType()))
535  return operand;
536  return rewriter.create<LLVM::ExtractElementOp>(loc, operand, index);
537  };
538  auto scalarOperands = llvm::map_to_vector(operands, extractElement);
539  Operation *scalarOp =
540  rewriter.create(loc, name, scalarOperands, elementType, op->getAttrs());
541  result = rewriter.create<LLVM::InsertElementOp>(
542  loc, result, scalarOp->getResult(0), index);
543  }
544 
545  rewriter.replaceOp(op, result);
546  return success();
547 }

553 /// Generates a symbol with 0-sized array type for dynamic shared memory usage,
554 /// or uses existing symbol.
555 LLVM::GlobalOp
556 getDynamicSharedMemorySymbol(ConversionPatternRewriter &rewriter,
557  Operation *moduleOp, gpu::DynamicSharedMemoryOp op,
558  const LLVMTypeConverter *typeConverter,
559  MemRefType memrefType, unsigned alignmentBit) {
560  uint64_t alignmentByte = alignmentBit / memrefType.getElementTypeBitWidth();
561 
562  FailureOr<unsigned> addressSpace =
563  typeConverter->getMemRefAddressSpace(memrefType);
564  if (failed(addressSpace)) {
565  op->emitError() << "conversion of memref memory space "
566  << memrefType.getMemorySpace()
567  << " to integer address space "
568  "failed. Consider adding memory space conversions.";
569  }
570 
571  // Step 1. Collect symbol names of LLVM::GlobalOp Ops. Also if any of
572  // LLVM::GlobalOp is suitable for shared memory, return it.
573  llvm::StringSet<> existingGlobalNames;
574  for (auto globalOp :
575  moduleOp->getRegion(0).front().getOps<LLVM::GlobalOp>()) {
576  existingGlobalNames.insert(globalOp.getSymName());
577  if (auto arrayType = dyn_cast<LLVM::LLVMArrayType>(globalOp.getType())) {
578  if (globalOp.getAddrSpace() == addressSpace.value() &&
579  arrayType.getNumElements() == 0 &&
580  globalOp.getAlignment().value_or(0) == alignmentByte) {
581  return globalOp;
582  }
583  }
584  }
585 
586  // Step 2. Find a unique symbol name
587  unsigned uniquingCounter = 0;
588  SmallString<128> symName = SymbolTable::generateSymbolName<128>(
589  "__dynamic_shmem_",
590  [&](StringRef candidate) {
591  return existingGlobalNames.contains(candidate);
592  },
593  uniquingCounter);
594 
595  // Step 3. Generate a global op
596  OpBuilder::InsertionGuard guard(rewriter);
597  rewriter.setInsertionPoint(&moduleOp->getRegion(0).front().front());
598 
599  auto zeroSizedArrayType = LLVM::LLVMArrayType::get(
600  typeConverter->convertType(memrefType.getElementType()), 0);
601 
602  return rewriter.create<LLVM::GlobalOp>(
603  op->getLoc(), zeroSizedArrayType, /*isConstant=*/false,
604  LLVM::Linkage::Internal, symName, /*value=*/Attribute(), alignmentByte,
605  addressSpace.value());
606 }

608 LogicalResult GPUDynamicSharedMemoryOpLowering::matchAndRewrite(
609  gpu::DynamicSharedMemoryOp op, OpAdaptor adaptor,
610  ConversionPatternRewriter &rewriter) const {
611  Location loc = op.getLoc();
612  MemRefType memrefType = op.getResultMemref().getType();
613  Type elementType = typeConverter->convertType(memrefType.getElementType());
614 
615  // Step 1: Generate a memref<0xi8> type
616  MemRefLayoutAttrInterface layout = {};
617  auto memrefType0sz =
618  MemRefType::get({0}, elementType, layout, memrefType.getMemorySpace());
619 
620  // Step 2: Generate a global symbol or existing for the dynamic shared
621  // memory with memref<0xi8> type
622  LLVM::LLVMFuncOp funcOp = op->getParentOfType<LLVM::LLVMFuncOp>();
623  LLVM::GlobalOp shmemOp = {};
624  Operation *moduleOp = funcOp->getParentWithTrait<OpTrait::SymbolTable>();
625  shmemOp = getDynamicSharedMemorySymbol(
626  rewriter, moduleOp, op, getTypeConverter(), memrefType0sz, alignmentBit);
627 
628  // Step 3. Get address of the global symbol
629  OpBuilder::InsertionGuard guard(rewriter);
630  rewriter.setInsertionPoint(op);
631  auto basePtr = rewriter.create<LLVM::AddressOfOp>(loc, shmemOp);
632  Type baseType = basePtr->getResultTypes().front();
633 
634  // Step 4. Generate GEP using offsets
635  SmallVector<LLVM::GEPArg> gepArgs = {0};
636  Value shmemPtr = rewriter.create<LLVM::GEPOp>(loc, baseType, elementType,
637  basePtr, gepArgs);
638  // Step 5. Create a memref descriptor
639  SmallVector<Value> shape, strides;
640  Value sizeBytes;
641  getMemRefDescriptorSizes(loc, memrefType0sz, {}, rewriter, shape, strides,
642  sizeBytes);
643  auto memRefDescriptor = this->createMemRefDescriptor(
644  loc, memrefType0sz, shmemPtr, shmemPtr, shape, strides, rewriter);
645 
646  // Step 5. Replace the op with memref descriptor
647  rewriter.replaceOp(op, {memRefDescriptor});
648  return success();
649 }
