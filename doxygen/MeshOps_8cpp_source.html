147 LogicalResult ClusterOp::verify() {
148  ArrayRef<int64_t> dimSizes = getDimSizes();
149  uint64_t rank = getRank();
150 
151  if (rank == 0)
152  return emitOpError("rank of cluster is expected to be a positive integer");
153 
154  if (dimSizes.size() > rank)
155  return emitOpError(
156  "rank of dim_sizes is not expected to be larger than rank of cluster");
157 
158  for (int64_t dimSize : dimSizes) {
159  if (dimSize < 0 && !ShapedType::isDynamic(dimSize))
160  return emitOpError("dimension size of a mesh cluster is expected to be "
161  "non-negative or dynamic");
162  }
163 
164  return success();
165 }

178 LogicalResult
179 MeshShardingAttr::verify(function_ref<InFlightDiagnostic()> emitError,
180  SymbolRefAttr, ArrayRef<DenseI32ArrayAttr> splitAxes,
181  ArrayRef<int32_t> partialAxes, Partial) {
182  // TODO: At present cluster symbol ref is not verified. This is due to the
183  // difficulty in fetching the corresponding symbol op based on an attribute.
184 
185  llvm::SmallSet<int32_t, 4> visitedAxes;
186 
187  auto checkMeshAxis = [&](ArrayRef<int32_t> axesArray) -> LogicalResult {
188  for (int32_t axis : axesArray) {
189  if (axis < 0)
190  return emitError() << "mesh axis is expected to be non-negative";
191  if (!visitedAxes.insert(axis).second)
192  return emitError() << "mesh axis duplicated";
193  }
194  return success();
195  };
196 
197  for (DenseI32ArrayAttr subAxes : splitAxes) {
198  ArrayRef<int32_t> subAxesArray = subAxes.asArrayRef();
199  if (failed(checkMeshAxis(subAxesArray)))
200  return failure();
201  }
202  if (failed(checkMeshAxis(partialAxes)))
203  return failure();
204  return success();
205 }

213 template <typename Op>
214 struct EmptyMeshAxesCanonicalizationPattern : OpRewritePattern<Op> {
215  using OpRewritePattern<Op>::OpRewritePattern;
216  LogicalResult matchAndRewrite(Op op,
217  PatternRewriter &rewriter) const override {
218  auto meshAxes = op.getMeshAxes();
219  if (!meshAxes.empty()) {
220  return failure();
221  }
222  if (op.getInput().getType() != op.getResult().getType()) {
223  return failure();
224  }
225 
226  rewriter.replaceAllUsesWith(op.getResult(), op.getInput());
227  rewriter.eraseOp(op.getOperation());
228  return success();
229  }
230 };

246 template <typename It>
247 bool isUnique(It begin, It end) {
248  if (begin == end) {
249  return true;
250  }
251  It next = std::next(begin);
252  if (next == end) {
253  return true;
254  }
255  for (; next != end; ++next, ++begin) {
256  if (*begin == *next) {
257  return false;
258  }
259  }
260  return true;
261 }

263 static LogicalResult verifyMeshAxes(Location loc, ArrayRef<MeshAxis> axes,
264  ClusterOp mesh) {
265  SmallVector<MeshAxis> sorted = llvm::to_vector(axes);
266  llvm::sort(sorted);
267  if (!isUnique(sorted.begin(), sorted.end())) {
268  return emitError(loc) << "Mesh axes contains duplicate elements.";
269  }
270 
271  MeshAxis rank = mesh.getRank();
272  for (auto axis : axes) {
273  if (axis >= rank || axis < 0) {
274  return emitError(loc)
275  << "0-based mesh axis index " << axis
276  << " is out of bounds. The referenced mesh \"" << mesh.getSymName()
277  << "\" is of rank " << rank << ".";
278  }
279  }
280 
281  return success();
282 }

309 static int64_t collectiveDeviceGroupSize(ArrayRef<MeshAxis> meshAxes,
310  ArrayRef<int64_t> meshShape) {
311  int64_t res = 1;
312 
313  for (MeshAxis axis : meshAxes) {
314  if (ShapedType::isDynamic(meshShape[axis])) {
315  return ShapedType::kDynamic;
316  }
317  assert(size_t(axis) < meshShape.size());
318  res *= meshShape[axis];
319  }
320 
321  return res;
322 }

324 static LogicalResult verifyDimensionCompatibility(Location loc,
325  int64_t expectedDimSize,
326  int64_t resultDimSize,
327  int64_t resultAxis) {
328  if (!ShapedType::isDynamic(resultDimSize) &&
329  expectedDimSize != resultDimSize) {
330  return emitError(loc) << "Dimension size mismatch for result axis "
331  << resultAxis << ". Expected "
332  << (ShapedType::isDynamic(expectedDimSize)
333  ? Twine("dynamic")
334  : Twine(expectedDimSize))
335  << ", but got " << resultDimSize << ".";
336  }
337 
338  return success();

341 static LogicalResult verifyAllGatherOperandAndResultShape(
342  Value operand, Value result, int64_t gatherAxis,
343  ArrayRef<MeshAxis> meshAxes, ArrayRef<int64_t> meshShape) {
344  auto resultRank = result.getType().template cast<ShapedType>().getRank();
345  if (gatherAxis < 0 || gatherAxis >= resultRank) {
346  return emitError(result.getLoc())
347  << "Gather axis " << gatherAxis << " is out of bounds [0, "
348  << resultRank << ").";
349  }
350 
351  ShapedType operandType = operand.getType().cast<ShapedType>();
352  ShapedType resultType = result.getType().cast<ShapedType>();
353  auto deviceGroupSize =
354  DimensionSize(collectiveDeviceGroupSize(meshAxes, meshShape));
355  for (int64_t axis = 0; axis < operandType.getRank(); ++axis) {
356  auto operandDimSize = DimensionSize(operandType.getDimSize(axis));
357  auto resultDimSize = DimensionSize(resultType.getDimSize(axis));
358  auto expectedResultDimSize =
359  axis == gatherAxis ? deviceGroupSize * operandDimSize : operandDimSize;
360  if (failed(verifyDimensionCompatibility(
361  result.getLoc(), expectedResultDimSize, resultDimSize, axis))) {
362  return failure();
363  }
364  }
365  return success();
366 }

368 static LogicalResult verifyAllToAllOperandAndResultShape(
369  Value operand, Value result, int64_t splitAxis, int64_t concatAxis,
370  ArrayRef<MeshAxis> meshAxes, ArrayRef<int64_t> meshShape) {
371  ShapedType operandType = operand.getType().cast<ShapedType>();
372  ShapedType resultType = result.getType().cast<ShapedType>();
373  for (int64_t axis = 0; axis < operandType.getRank(); ++axis) {
374  if ((axis != splitAxis && axis != concatAxis) || splitAxis == concatAxis) {
375  if (failed(verifyDimensionCompatibility(
376  result.getLoc(), operandType.getDimSize(axis),
377  resultType.getDimSize(axis), axis))) {
378  return failure();
379  }
380  }
381  }
382 
383  if (splitAxis == concatAxis) {
384  return success();
385  }
386 
387  auto deviceGroupSize =
388  DimensionSize(collectiveDeviceGroupSize(meshAxes, meshShape));
389  auto operandConcatDimSize = DimensionSize(operandType.getDimSize(concatAxis));
390  auto operandSplitDimSize = DimensionSize(operandType.getDimSize(splitAxis));
391  DimensionSize expectedResultConcatDimSize =
392  operandConcatDimSize * deviceGroupSize;
393  DimensionSize expectedResultSplitDimSize =
394  operandSplitDimSize / deviceGroupSize;
395  if (!expectedResultSplitDimSize.isDynamic() &&
396  int64_t(operandSplitDimSize) % int64_t(deviceGroupSize) != 0) {
397  expectedResultSplitDimSize = DimensionSize::dynamic();
398  }
399  if (failed(verifyDimensionCompatibility(
400  result.getLoc(), expectedResultConcatDimSize.value(),
401  resultType.getDimSize(concatAxis), concatAxis))) {
402  return failure();
403  }
404  if (failed(verifyDimensionCompatibility(
405  result.getLoc(), expectedResultSplitDimSize.value(),
406  resultType.getDimSize(splitAxis), splitAxis))) {
407  return failure();
408  }
409 
410  return success();
411 }

413 static LogicalResult verifyReduceScatterOperandAndResultShape(
414  Value operand, Value result, int64_t scatterAxis,
415  ArrayRef<MeshAxis> meshAxes, ArrayRef<int64_t> meshShape) {
416  ShapedType operandType = operand.getType().cast<ShapedType>();
417  ShapedType resultType = result.getType().cast<ShapedType>();
418  for (int64_t axis = 0; axis < operandType.getRank(); ++axis) {
419  if (axis != scatterAxis) {
420  if (failed(verifyDimensionCompatibility(
421  result.getLoc(), operandType.getDimSize(axis),
422  resultType.getDimSize(axis), axis))) {
423  return failure();
424  }
425  }
426  }
427 
428  auto deviceGroupSize =
429  DimensionSize(collectiveDeviceGroupSize(meshAxes, meshShape));
430  auto operandScatterDimSize =
431  DimensionSize(operandType.getDimSize(scatterAxis));
432  if (!operandScatterDimSize.isDynamic() && !deviceGroupSize.isDynamic() &&
433  int64_t(operandScatterDimSize) % int64_t(deviceGroupSize) != 0) {
434  return emitError(result.getLoc())
435  << "Operand dimension size " << int64_t(operandScatterDimSize)
436  << " is not divisible by collective device group size "
437  << int64_t(deviceGroupSize) << " for scatter axis " << scatterAxis
438  << ".";
439  }
440  DimensionSize expectedResultScatterDimSize =
441  operandScatterDimSize / deviceGroupSize;
442  if (failed(verifyDimensionCompatibility(
443  result.getLoc(), expectedResultScatterDimSize.value(),
444  resultType.getDimSize(scatterAxis), scatterAxis))) {
445  return failure();
446  }
447 
448  return success();
449 }
