57 LogicalResult AddOp::verify() {
58  Type lhsType = getLhs().getType();
59  Type rhsType = getRhs().getType();
60 
61  if (lhsType.isa<emitc::PointerType>() && rhsType.isa<emitc::PointerType>())
62  return emitOpError("requires that at most one operand is a pointer");
63 
64  if ((lhsType.isa<emitc::PointerType>() &&
65  !rhsType.isa<IntegerType, emitc::OpaqueType>()) ||
66  (rhsType.isa<emitc::PointerType>() &&
67  !lhsType.isa<IntegerType, emitc::OpaqueType>()))
68  return emitOpError("requires that one operand is an integer or of opaque "
69  "type if the other is a pointer");
70 
71  return success();
72 }

134 LogicalResult emitc::CallOpaqueOp::verify() {
135  // Callee must not be empty.
136  if (getCallee().empty())
137  return emitOpError("callee must not be empty");
138 
139  if (std::optional<ArrayAttr> argsAttr = getArgs()) {
140  for (Attribute arg : *argsAttr) {
141  auto intAttr = llvm::dyn_cast<IntegerAttr>(arg);
142  if (intAttr && llvm::isa<IndexType>(intAttr.getType())) {
143  int64_t index = intAttr.getInt();
144  // Args with elements of type index must be in range
145  // [0..operands.size).
146  if ((index < 0) || (index >= static_cast<int64_t>(getNumOperands())))
147  return emitOpError("index argument is out of range");
148 
149  // Args with elements of type ArrayAttr must have a type.
150  } else if (llvm::isa<ArrayAttr>(
151  arg) /*&& llvm::isa<NoneType>(arg.getType())*/) {
152  // FIXME: Array attributes never have types
153  return emitOpError("array argument has no type");
154  }
155  }
156  }
157 
158  if (std::optional<ArrayAttr> templateArgsAttr = getTemplateArgs()) {
159  for (Attribute tArg : *templateArgsAttr) {
160  if (!llvm::isa<TypeAttr, IntegerAttr, FloatAttr, emitc::OpaqueAttr>(tArg))
161  return emitOpError("template argument has invalid type");
162  }
163  }
164 
165  return success();
166 }

196 void ForOp::build(OpBuilder &builder, OperationState &result, Value lb,
197  Value ub, Value step, BodyBuilderFn bodyBuilder) {
198  result.addOperands({lb, ub, step});
199  Type t = lb.getType();
200  Region *bodyRegion = result.addRegion();
201  bodyRegion->push_back(new Block);
202  Block &bodyBlock = bodyRegion->front();
203  bodyBlock.addArgument(t, result.location);
204 
205  // Create the default terminator if the builder is not provided.
206  if (!bodyBuilder) {
207  ForOp::ensureTerminator(*bodyRegion, builder, result.location);
208  } else {
209  OpBuilder::InsertionGuard guard(builder);
210  builder.setInsertionPointToStart(&bodyBlock);
211  bodyBuilder(builder, result.location, bodyBlock.getArgument(0));
212  }
213 }

217 ParseResult ForOp::parse(OpAsmParser &parser, OperationState &result) {
218  Builder &builder = parser.getBuilder();
219  Type type;
220 
221  OpAsmParser::Argument inductionVariable;
222  OpAsmParser::UnresolvedOperand lb, ub, step;
223 
224  // Parse the induction variable followed by '='.
225  if (parser.parseOperand(inductionVariable.ssaName) || parser.parseEqual() ||
226  // Parse loop bounds.
227  parser.parseOperand(lb) || parser.parseKeyword("to") ||
228  parser.parseOperand(ub) || parser.parseKeyword("step") ||
229  parser.parseOperand(step))
230  return failure();
231 
232  // Parse the optional initial iteration arguments.
233  SmallVector<OpAsmParser::Argument, 4> regionArgs;
234  SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
235  regionArgs.push_back(inductionVariable);
236 
237  // Parse optional type, else assume Index.
238  if (parser.parseOptionalColon())
239  type = builder.getIndexType();
240  else if (parser.parseType(type))
241  return failure();
242 
243  // Resolve input operands.
244  regionArgs.front().type = type;
245  if (parser.resolveOperand(lb, type, result.operands) ||
246  parser.resolveOperand(ub, type, result.operands) ||
247  parser.resolveOperand(step, type, result.operands))
248  return failure();
249 
250  // Parse the body region.
251  Region *body = result.addRegion();
252  if (parser.parseRegion(*body, regionArgs))
253  return failure();
254 
255  ForOp::ensureTerminator(*body, builder, result.location);
256 
257  // Parse the optional attribute list.
258  if (parser.parseOptionalAttrDict(result.attributes))
259  return failure();
260 
261  return success();
262 }

323 void IfOp::build(OpBuilder &builder, OperationState &result, Value cond,
324  function_ref<void(OpBuilder &, Location)> thenBuilder,
325  function_ref<void(OpBuilder &, Location)> elseBuilder) {
326  assert(thenBuilder && "the builder callback for 'then' must be present");

328 
329  // Build then region.

343 ParseResult IfOp::parse(OpAsmParser &parser, OperationState &result) {
344  // Create the regions for 'then'.
345  result.regions.reserve(2);
346  Region *thenRegion = result.addRegion();
347  Region *elseRegion = result.addRegion();
348 
349  Builder &builder = parser.getBuilder();
350  OpAsmParser::UnresolvedOperand cond;
351  Type i1Type = builder.getIntegerType(1);
352  if (parser.parseOperand(cond) ||
353  parser.resolveOperand(cond, i1Type, result.operands))
354  return failure();
355  // Parse the 'then' region.
356  if (parser.parseRegion(*thenRegion, /*arguments=*/{}, /*argTypes=*/{}))
357  return failure();
358  IfOp::ensureTerminator(*thenRegion, parser.getBuilder(), result.location);
359 
360  // If we find an 'else' keyword then parse the 'else' region.
361  if (!parser.parseOptionalKeyword("else")) {
362  if (parser.parseRegion(*elseRegion, /*arguments=*/{}, /*argTypes=*/{}))
363  return failure();
364  IfOp::ensureTerminator(*elseRegion, parser.getBuilder(), result.location);
365  }
366 
367  // Parse the optional attribute list.
368  if (parser.parseOptionalAttrDict(result.attributes))
369  return failure();
370  return success();
371 }

394 /// Given the region at `index`, or the parent operation if `index` is None,
395 /// return the successor regions. These are the regions that may be selected
396 /// during the flow of control. `operands` is a set of optional attributes that
397 /// correspond to a constant value for each operand, or null if that operand is
398 /// not a constant.
399 void IfOp::getSuccessorRegions(RegionBranchPoint point,
400  SmallVectorImpl<RegionSuccessor> &regions) {
401  // The `then` and the `else` region branch back to the parent operation.
402  if (!point.isParent()) {
403  regions.push_back(RegionSuccessor());
404  return;
405  }
406 
407  regions.push_back(RegionSuccessor(&getThenRegion()));
408 
409  // Don't consider the else region if it is empty.
410  Region *elseRegion = &this->getElseRegion();
411  if (elseRegion->empty())
412  regions.push_back(RegionSuccessor());
413  else
414  regions.push_back(RegionSuccessor(elseRegion));
415 }

417 void IfOp::getEntrySuccessorRegions(ArrayRef<Attribute> operands,
418  SmallVectorImpl<RegionSuccessor> &regions) {
419  FoldAdaptor adaptor(operands, *this);
420  auto boolAttr = dyn_cast_or_null<BoolAttr>(adaptor.getCondition());
421  if (!boolAttr || boolAttr.getValue())
422  regions.emplace_back(&getThenRegion());
423 
424  // If the else region is empty, execution continues after the parent op.
425  if (!boolAttr || !boolAttr.getValue()) {
426  if (!getElseRegion().empty())
427  regions.emplace_back(&getElseRegion());
428  else
429  regions.emplace_back();
430  }
431 }

433 void IfOp::getRegionInvocationBounds(
434  ArrayRef<Attribute> operands,
435  SmallVectorImpl<InvocationBounds> &invocationBounds) {
436  if (auto cond = llvm::dyn_cast_or_null<BoolAttr>(operands[0])) {
437  // If the condition is known, then one region is known to be executed once
438  // and the other zero times.
439  invocationBounds.emplace_back(0, cond.getValue() ? 1 : 0);
440  invocationBounds.emplace_back(0, cond.getValue() ? 0 : 1);
441  } else {
442  // Non-constant condition. Each region may be executed 0 or 1 times.
443  invocationBounds.assign(2, {0, 1});
444  }
445 }
