29 /// Adjusts `indices` as follows for a given tile slice and returns them in
30 /// `outIndices`:
31 /// rank 1: (indices[0] + (tileSliceIndex * tileSliceNumElts))
32 /// rank 2: (indices[0] + tileSliceIndex, indices[1])
33 void getMemrefIndices(ValueRange indices, unsigned rank, Value tileSliceIndex,
34  Value tileSliceNumElts,
35  SmallVectorImpl<Value> &outIndices, Location loc,
36  PatternRewriter &rewriter) {
37  assert((rank == 1 || rank == 2) && "memref has unexpected rank!");
38 
39  auto tileSliceOffset = tileSliceIndex;
40  if (rank == 1)
41  tileSliceOffset =
42  rewriter.create<arith::MulIOp>(loc, tileSliceOffset, tileSliceNumElts);
43 
44  auto baseIndexPlusTileSliceOffset =
45  rewriter.create<arith::AddIOp>(loc, indices[0], tileSliceOffset);
46  outIndices.push_back(baseIndexPlusTileSliceOffset);
47 
48  if (rank == 2)
49  outIndices.push_back(indices[1]);
50 }

52 /// Lower `arm_sme.tile_load` to a loop over the tile slices and load each slice
53 /// using `arm_sme.load_tile_slice`.
54 ///
55 /// BEFORE:
56 /// ```mlir
57 /// %tile = arm_sme.tile_load %src[%c0, %c0] :
58 /// memref<?x?xi32>, vector<[4]x[4]xi32>
59 /// ```
60 ///
61 /// AFTER:
62 /// ```mlir
63 /// %ptrue_s = arith.constant dense<true> : vector<[4]xi1>
64 /// %init_tile = arm_sme.get_tile : vector<[4]x[4]xi32>
65 /// %vscale = vector.vscale
66 /// %c0 = arith.constant 0 : index
67 /// %c1 = arith.constant 1 : index
68 /// %min_svl_s = arith.constant 4 : index
69 /// %svl_s = arith.muli %min_svl_s, %vscale : index
70 /// %tile = scf.for %tile_slice_idx = %c0 to %svl_s step %c1
71 /// iter_args(%iter_tile = %init_tile) -> (vector<[4]x[4]xi32>) {
72 /// %tile_update = arm_sme.load_tile_slice %src[%tile_slice_idx],
73 /// %ptrue_s, %iter_tile, %tile_slice_idx
74 /// : memref<?x?xi32>, vector<[4]xi1>, vector<[4]x[4]xi32>
75 /// scf.yield %tile_update : vector<[4]x[4]xi32>
76 /// }
77 /// ```
78 struct TileLoadOpConversion : public OpRewritePattern<arm_sme::TileLoadOp> {
79  using OpRewritePattern<arm_sme::TileLoadOp>::OpRewritePattern;
80 
81  LogicalResult matchAndRewrite(arm_sme::TileLoadOp tileLoadOp,
82  PatternRewriter &rewriter) const override {
83  if (tileLoadOp.getMask())
84  return rewriter.notifyMatchFailure(tileLoadOp,
85  "op has mask, apply masked patterns");
86 
87  OpBuilder::InsertionGuard g(rewriter);
88  auto loc = tileLoadOp.getLoc();
89  auto tileType = tileLoadOp.getVectorType();
90  auto tileElementType = tileType.getElementType();
91 
92  // Allocate a new SME tile.
93  auto initTile = tileLoadOp.createOpAndForwardTileId<arm_sme::GetTileOp>(
94  rewriter, loc, tileType);
95 
96  // Create a loop that loads each ZA tile slice from memory.
97  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
98  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
99  loc, arm_sme::getSMETileSliceMinNumElts(tileElementType));
100  auto vscale =
101  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
102  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
103  // This describes both the number of ZA tile slices and the number of
104  // elements in a vector of SVL bits for a given element type (SVL_B, SVL_H,
105  // ..., SVL_Q).
106  auto numTileSlices =
107  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
108  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, numTileSlices,
109  step, ValueRange{initTile});
110 
111  rewriter.setInsertionPointToStart(forOp.getBody());
112 
113  // Create an 'all true' predicate for the tile slice.
114  auto predicateType =
115  VectorType::get(tileType.getDimSize(1), rewriter.getI1Type(), true);
116  auto allTruePredicate = rewriter.create<arith::ConstantOp>(
117  loc, DenseElementsAttr::get(predicateType, true));
118 
119  // Create 'arm_sme.load_tile_slice' to load tile slice from memory into
120  // tile.
121  SmallVector<Value> memrefIndices;
122  auto tileSliceIndex = forOp.getInductionVar();
123  getMemrefIndices(tileLoadOp.getIndices(),
124  tileLoadOp.getMemRefType().getRank(), tileSliceIndex,
125  numTileSlices, memrefIndices, loc, rewriter);
126  auto currentTile = forOp.getRegionIterArg(0);
127  auto loadSlice =
128  tileLoadOp.createOpAndForwardTileId<arm_sme::LoadTileSliceOp>(
129  rewriter, loc, tileType, tileLoadOp.getBase(), allTruePredicate,
130  currentTile, memrefIndices, tileSliceIndex, tileLoadOp.getLayout());
131  rewriter.create<scf::YieldOp>(loc, loadSlice.getResult());
132 
133  rewriter.setInsertionPointAfter(forOp);
134 
135  // Replace 'arm_sme.tile_load' with the result.
136  rewriter.replaceOp(tileLoadOp, forOp.getResult(0));
137 
138  return success();
139  }
140 };

142 /// Lower `arm_sme.tile_load` with mask and pad of constant zero.
143 ///
144 /// BEFORE:
145 /// ```mlir
146 /// %pad = arith.constant 0 : i32
147 /// %num_rows = arith.constant 2 : index
148 /// %num_cols = arith.constant 4 : index
149 /// %mask = vector.create_mask %num_rows, %num_cols : vector<[4]x[4]xi1>
150 /// %tile = arm_sme.tile_load %src[%c0, %c0], %pad, %mask :
151 /// memref<?x?xi32>, vector<[4]x[4]xi32>
152 /// ```
153 ///
154 /// AFTER:
155 /// ```mlir
156 /// %c0 = arith.constant 0 : index
157 /// %c1 = arith.constant 1 : index
158 /// %init_tile = arm_sme.zero : vector<[4]x[4]xi32>
159 /// %num_rows = arith.constant 2 : index
160 /// %num_cols = vector.create_mask %c4 : vector<[4]xi1>
161 /// %tile = scf.for %tile_slice_idx = %c0 to %num_rows step %c1
162 /// iter_args(%iter_tile = %init_tile) -> (vector<[4]x[4]xi32>) {
163 /// %tile_update = arm_sme.load_tile_slice
164 /// %src[%tile_slice_idx], %num_cols, %iter_tile, %tile_slice_idx :
165 /// memref<?x?xi32>, vector<[1]xi32>, vector<[4]x[4]xi32>
166 /// scf.yield %tile_update : vector<[4]x[4]xi32>
167 /// }
168 /// ```
169 ///
170 /// NOTE: Only mask of 'vector.create_mask' op is currently supported.
171 struct TileLoadOpWithMaskAndPadZeroConversion
172  : public OpRewritePattern<arm_sme::TileLoadOp> {
173  using OpRewritePattern<arm_sme::TileLoadOp>::OpRewritePattern;
174 
175  LogicalResult matchAndRewrite(arm_sme::TileLoadOp tileLoadOp,
176  PatternRewriter &rewriter) const override {
177  OpBuilder::InsertionGuard g(rewriter);
178  auto loc = tileLoadOp.getLoc();
179  auto tileType = tileLoadOp.getVectorType();
180 
181  auto maskOp = tileLoadOp.getMask();
182  if (!maskOp)
183  return rewriter.notifyMatchFailure(
184  tileLoadOp, "op has no mask, needs unmasked pattern");
185 
186  auto padOp = tileLoadOp.getPadding();
187  assert(padOp && "expected padding when masking!");
188 
189  auto createMaskOp = maskOp.getDefiningOp<vector::CreateMaskOp>();
190  if (!createMaskOp)
191  return rewriter.notifyMatchFailure(
192  tileLoadOp, "unsupported mask op, only 'vector.create_mask' is "
193  "currently supported");
194 
195  auto constPadOp = padOp.getDefiningOp<arith::ConstantOp>();
196  if (!constPadOp || constPadOp.getValue() !=
197  rewriter.getZeroAttr(tileType.getElementType()))
198  return rewriter.notifyMatchFailure(
199  tileLoadOp, "op has non-zero pad, needs non-zero pad pattern");
200 
201  auto numRows = createMaskOp.getOperands()[0];
202  auto numCols = createMaskOp.getOperands()[1];
203 
204  auto predicateType =
205  VectorType::get(tileType.getDimSize(1), rewriter.getI1Type(), true);
206  auto numColsOp =
207  rewriter.create<vector::CreateMaskOp>(loc, predicateType, numCols);
208 
209  // Initialize tile with zero to satisfy padding. Inactive cols will be
210  // zeroed anyway since the loads use zeroing predication. For inactive rows
211  // however, no load will occur so these need to be zeroed.
212  auto initTile = tileLoadOp.createOpAndForwardTileId<arm_sme::ZeroOp>(
213  rewriter, loc, tileType);
214 
215  // Create a loop to load the active tile slices from memory.
216  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
217  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
218  auto upperBound = numRows;
219  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, upperBound, step,
220  ValueRange{initTile});
221 
222  rewriter.setInsertionPointToStart(forOp.getBody());
223 
224  // Create 'arm_sme.load_tile_slice' to load tile slice from memory into
225  // tile.
226  SmallVector<Value> memrefIndices;
227  auto tileSliceIndex = forOp.getInductionVar();
228  auto currentTile = forOp.getRegionIterArg(0);
229  getMemrefIndices(tileLoadOp.getIndices(),
230  tileLoadOp.getMemRefType().getRank(), tileSliceIndex,
231  upperBound, memrefIndices, loc, rewriter);
232  auto loadSlice =
233  tileLoadOp.createOpAndForwardTileId<arm_sme::LoadTileSliceOp>(
234  rewriter, loc, tileType, tileLoadOp.getBase(), numColsOp,
235  currentTile, memrefIndices, tileSliceIndex, tileLoadOp.getLayout());
236  rewriter.create<scf::YieldOp>(loc, loadSlice.getResult());
237 
238  rewriter.setInsertionPointAfter(forOp);
239 
240  // Replace 'arm_sme.tile_load' with the result.
241  rewriter.replaceOp(tileLoadOp, forOp.getResult(0));
242 
243  return success();
244  }
245 };

247 /// Lower `arm_sme.tile_load` with mask and non-zero pad.
248 ///
249 /// BEFORE:
250 /// ```mlir
251 /// %pad = arith.constant 1 : i32
252 /// %num_rows = arith.constant 2 : index
253 /// %num_cols = arith.constant 4 : index
254 /// %mask = vector.create_mask %num_rows, %num_cols : vector<[4]x[4]xi1>
255 /// %tile = arm_sme.tile_load %src[%c0, %c0], %pad, %mask :
256 /// memref<?x?xi32>, vector<[4]x[4]xi32>
257 /// ```
258 ///
259 /// AFTER:
260 /// ```mlir
261 /// ...
262 /// %pad_1d = arith.constant dense<1> : vector<[4]xi32>
263 /// %tile = scf.for %tile_slice_idx = %c0 to %svl_s step %c1
264 /// iter_args(%iter_tile = %init_tile) -> (vector<[4]x[4]xi32>) {
265 /// ...
266 /// %mask_1d = vector.create_mask <combined_mask> : vector<[4]xi1>
267 /// %slice = vector.maskedload %base[%tile_slice_idx, %c0], %mask_1d, %pad_1d
268 /// : memref<?x?xi32>, vector<[4]xi1>,
269 /// vector<[4]xi32> into vector<[4]xi32>
270 /// // Insert slice into tile
271 /// %tile_update = arm_sme.move_vector_to_tile_slice
272 /// %slice, %iter_tile, %tile_slice_idx :
273 /// vector<[4]xi32> into vector<[4]x[4]xi32>
274 /// scf.yield %tile_update : vector<[4]x[4]xi32>
275 /// }
276 /// ```
277 struct TileLoadOpWithMaskAndPadNonZeroConversion
278  : public OpRewritePattern<arm_sme::TileLoadOp> {
279  using OpRewritePattern<arm_sme::TileLoadOp>::OpRewritePattern;
280 
281  LogicalResult matchAndRewrite(arm_sme::TileLoadOp tileLoadOp,
282  PatternRewriter &rewriter) const override {
283  OpBuilder::InsertionGuard g(rewriter);
284  auto loc = tileLoadOp.getLoc();
285  auto tileType = tileLoadOp.getVectorType();
286  auto tileElementType = tileType.getElementType();
287 
288  auto maskOp = tileLoadOp.getMask();
289  if (!maskOp)
290  return rewriter.notifyMatchFailure(
291  tileLoadOp, "op has no mask, needs unmasked pattern");
292 
293  auto padOp = tileLoadOp.getPadding();
294  assert(padOp && "expected padding when masking!");
295 
296  auto createMaskOp = maskOp.getDefiningOp<vector::CreateMaskOp>();
297  if (!createMaskOp)
298  return rewriter.notifyMatchFailure(
299  tileLoadOp, "unsupported mask op, only 'vector.create_mask' is "
300  "currently supported");
301 
302  auto constPadOp = padOp.getDefiningOp<arith::ConstantOp>();
303  if (constPadOp &&
304  constPadOp.getValue() == rewriter.getZeroAttr(tileElementType))
305  return rewriter.notifyMatchFailure(
306  tileLoadOp, "op has constant zero pad, needs zero pad pattern");
307 
308  auto numRows = createMaskOp.getOperands()[0];
309  auto numCols = createMaskOp.getOperands()[1];
310 
311  auto numColsI32 = rewriter.create<arith::IndexCastUIOp>(
312  loc, rewriter.getI32Type(), numCols);
313 
314  // Allocate a new SME tile.
315  auto initTile = tileLoadOp.createOpAndForwardTileId<arm_sme::GetTileOp>(
316  rewriter, loc, tileType);
317 
318  // Create a loop that loads each ZA tile slice from memory.
319  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
320  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
321  loc, arm_sme::getSMETileSliceMinNumElts(tileElementType));
322  auto vscale =
323  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
324  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
325  auto numTileSlices =
326  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
327  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, numTileSlices,
328  step, ValueRange{initTile});
329 
330  rewriter.setInsertionPointToStart(forOp.getBody());
331 
332  auto tileSliceIndex = forOp.getInductionVar();
333  auto currentTile = forOp.getRegionIterArg(0);
334 
335  // Combine masks.
336  auto rowIsActive = rewriter.create<arith::CmpIOp>(
337  loc, arith::CmpIPredicate::ult, tileSliceIndex, numRows);
338  auto rowIsActiveI32 = rewriter.create<arith::ExtSIOp>(
339  loc, rewriter.getI32Type(), rowIsActive);
340  auto mask = rewriter.create<arith::AndIOp>(loc, rowIsActiveI32, numColsI32);
341  auto maskIndex =
342  rewriter.create<arith::IndexCastOp>(loc, rewriter.getIndexType(), mask);
343  auto predicateType =
344  VectorType::get(tileType.getDimSize(1), rewriter.getI1Type(), true);
345  auto maskOp1D = rewriter.create<vector::CreateMaskOp>(
346  loc, predicateType, maskIndex.getResult());
347 
348  SmallVector<Value> memrefIndices;
349  getMemrefIndices(tileLoadOp.getIndices(),
350  tileLoadOp.getMemRefType().getRank(), tileSliceIndex,
351  numTileSlices, memrefIndices, loc, rewriter);
352 
353  // Splat pad into 1-D vector matching type of tile slice.
354  VectorType tileSliceType = VectorType::Builder(tileType).dropDim(0);
355  auto pad1DOp = rewriter.create<vector::SplatOp>(loc, tileSliceType, padOp);
356 
357  auto loadSlice = rewriter.create<vector::MaskedLoadOp>(
358  loc, tileSliceType, tileLoadOp.getBase(), memrefIndices, maskOp1D,
359  /*passthru=*/pad1DOp);
360 
361  // Create 'arm_sme.move_vector_to_tile_slice' to move slice into tile.
362  auto moveSlice =
363  tileLoadOp.createOpAndForwardTileId<arm_sme::MoveVectorToTileSliceOp>(
364  rewriter, loc, tileType, loadSlice->getResult(0), currentTile,
365  tileSliceIndex, tileLoadOp.getLayout());
366  rewriter.create<scf::YieldOp>(loc, moveSlice.getResult());
367 
368  rewriter.setInsertionPointAfter(forOp);
369 
370  // Replace 'arm_sme.tile_load' with the result.
371  rewriter.replaceOp(tileLoadOp, forOp.getResult(0));
372 
373  return success();
374  }

377 /// Lower `arm_sme.tile_store` to a loop over the tile slices and store each
378 /// slice using `arm_sme.store_tile_slice`.
379 ///
380 /// BEFORE:
381 /// ```mlir
382 /// arm_sme.tile_store %tile, %dest[%c0, %c0] layout<vertical>
383 /// : memref<?x?xi32>, vector<[4]x[4]xi32
384 /// ```
385 ///
386 /// AFTER:
387 /// ```mlir
388 /// %vscale = vector.vscale
389 /// %c0 = arith.constant 0 : index
390 /// %c1 = arith.constant 1 : index
391 /// %min_svl_s = arith.constant 4 : index
392 /// %svl_s = arith.muli %min_svl_s, %vscale : index
393 /// scf.for %tile_slice_idx = %c0 to %svl_s step %c1 {
394 /// arm_sme.store_tile_slice %tile, %tile_slice_idx, %dest[%tile_slice_idx],
395 /// layout<vertical> : memref<?x?xi32>, vector<[4]x[4]xi32>
396 /// }
397 /// ```
398 struct TileStoreOpConversion : public OpRewritePattern<arm_sme::TileStoreOp> {
399  using OpRewritePattern<arm_sme::TileStoreOp>::OpRewritePattern;
400 
401  LogicalResult matchAndRewrite(arm_sme::TileStoreOp tileStoreOp,
402  PatternRewriter &rewriter) const override {
403  OpBuilder::InsertionGuard g(rewriter);
404  auto loc = tileStoreOp.getLoc();
405  auto tileType = tileStoreOp.getVectorType();
406  auto tileElementType = tileType.getElementType();
407 
408  auto predicateType =
409  VectorType::get(tileType.getDimSize(1), rewriter.getI1Type(), true);
410 
411  Value maskCols;
412  Value upperBound;
413  auto maskOp = tileStoreOp.getMask();
414  if (maskOp) {
415  auto createMaskOp = maskOp.getDefiningOp<vector::CreateMaskOp>();
416  if (!createMaskOp)
417  return rewriter.notifyMatchFailure(
418  tileStoreOp, "unsupported mask op, only 'vector.create_mask' is "
419  "currently supported");
420 
421  auto numRows = createMaskOp.getOperands()[0];
422  auto numCols = createMaskOp.getOperands()[1];
423 
424  upperBound = numRows;
425  maskCols =
426  rewriter.create<vector::CreateMaskOp>(loc, predicateType, numCols);
427  } else {
428  // Store all tile slices if no mask.
429  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
430  loc, arm_sme::getSMETileSliceMinNumElts(tileElementType));
431  auto vscale =
432  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
433  // This describes both the number of ZA tile slices and the number of
434  // elements in a vector of SVL bits for a given element type (SVL_B,
435  // SVL_H,
436  // ..., SVL_Q).
437  auto numTileSlices =
438  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
439 
440  upperBound = numTileSlices;
441  // Create an 'all true' predicate for the tile slice.
442  maskCols = rewriter.create<arith::ConstantOp>(
443  loc, DenseElementsAttr::get(predicateType, true));
444  }
445 
446  // Create a loop that stores each (active) active ZA tile slice from memory.
447  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
448  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
449  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, upperBound, step);
450 
451  rewriter.setInsertionPointToStart(forOp.getBody());
452 
453  SmallVector<Value> memrefIndices;
454  auto tileSliceIndex = forOp.getInductionVar();
455  getMemrefIndices(tileStoreOp.getIndices(),
456  tileStoreOp.getMemRefType().getRank(), tileSliceIndex,
457  upperBound, memrefIndices, loc, rewriter);
458 
459  tileStoreOp.replaceWithAndForwardTileId<arm_sme::StoreTileSliceOp>(
460  rewriter, tileStoreOp.getValueToStore(), tileSliceIndex, maskCols,
461  tileStoreOp.getBase(), memrefIndices, tileStoreOp.getLayout());
462 
463  return success();
464  }
