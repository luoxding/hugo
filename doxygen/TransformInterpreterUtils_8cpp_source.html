36 /// Expands the given list of `paths` to a list of `.mlir` files.
37 ///
38 /// Each entry in `paths` may either be a regular file, in which case it ends up
39 /// in the result list, or a directory, in which case all (regular) `.mlir`
40 /// files in that directory are added. Any other file types lead to a failure.
41 LogicalResult transform::detail::expandPathsToMLIRFiles(
42  ArrayRef<std::string> paths, MLIRContext *context,
43  SmallVectorImpl<std::string> &fileNames) {
44  for (const std::string &path : paths) {
45  auto loc = FileLineColLoc::get(context, path, 0, 0);
46 
47  if (llvm::sys::fs::is_regular_file(path)) {
48  LLVM_DEBUG(DBGS() << "Adding '" << path << "' to list of files\n");
49  fileNames.push_back(path);
50  continue;
51  }
52 
53  if (!llvm::sys::fs::is_directory(path)) {
54  return emitError(loc)
55  << "'" << path << "' is neither a file nor a directory";
56  }
57 
58  LLVM_DEBUG(DBGS() << "Looking for files in '" << path << "':\n");
59 
60  std::error_code ec;
61  for (llvm::sys::fs::directory_iterator it(path, ec), itEnd;
62  it != itEnd && !ec; it.increment(ec)) {
63  const std::string &fileName = it->path();
64 
65  if (it->type() != llvm::sys::fs::file_type::regular_file &&
66  it->type() != llvm::sys::fs::file_type::symlink_file) {
67  LLVM_DEBUG(DBGS() << " Skipping non-regular file '" << fileName
68  << "'\n");
69  continue;
70  }
71 
72  if (!StringRef(fileName).endswith(".mlir")) {
73  LLVM_DEBUG(DBGS() << " Skipping '" << fileName
74  << "' because it does not end with '.mlir'\n");
75  continue;
76  }
77 
78  LLVM_DEBUG(DBGS() << " Adding '" << fileName << "' to list of files\n");
79  fileNames.push_back(fileName);
80  }
81 
82  if (ec)
83  return emitError(loc) << "error while opening files in '" << path
84  << "': " << ec.message();
85  }
86 
87  return success();
88 }

90 LogicalResult transform::detail::parseTransformModuleFromFile(
91  MLIRContext *context, llvm::StringRef transformFileName,
92  OwningOpRef<ModuleOp> &transformModule) {
93  if (transformFileName.empty()) {
94  LLVM_DEBUG(
95  DBGS() << "no transform file name specified, assuming the transform "
96  "module is embedded in the IR next to the top-level\n");
97  return success();
98  }
99  // Parse transformFileName content into a ModuleOp.
100  std::string errorMessage;
101  auto memoryBuffer = mlir::openInputFile(transformFileName, &errorMessage);
102  if (!memoryBuffer) {
103  return emitError(FileLineColLoc::get(
104  StringAttr::get(context, transformFileName), 0, 0))
105  << "failed to open transform file: " << errorMessage;
106  }
107  // Tell sourceMgr about this buffer, the parser will pick it up.
108  llvm::SourceMgr sourceMgr;
109  sourceMgr.AddNewSourceBuffer(std::move(memoryBuffer), llvm::SMLoc());
110  transformModule =
111  OwningOpRef<ModuleOp>(parseSourceFile<ModuleOp>(sourceMgr, context));
112  return mlir::verify(*transformModule);
113 }

120 transform::TransformOpInterface
121 transform::detail::findTransformEntryPoint(Operation *root, ModuleOp module,
122  StringRef entryPoint) {
123  SmallVector<Operation *, 2> l{root};
124  if (module)
125  l.push_back(module);
126  for (Operation *op : l) {
127  transform::TransformOpInterface transform = nullptr;
128  op->walk<WalkOrder::PreOrder>(
129  [&](transform::NamedSequenceOp namedSequenceOp) {
130  if (namedSequenceOp.getSymName() == entryPoint) {
131  transform = cast<transform::TransformOpInterface>(
132  namedSequenceOp.getOperation());
133  return WalkResult::interrupt();
134  }
135  return WalkResult::advance();
136  });
137  if (transform)
138  return transform;
139  }
140  auto diag = root->emitError()
141  << "could not find a nested named sequence with name: "
142  << entryPoint;
143  return nullptr;
144 }

146 LogicalResult transform::detail::assembleTransformLibraryFromPaths(
147  MLIRContext *context, ArrayRef<std::string> transformLibraryPaths,
148  OwningOpRef<ModuleOp> &transformModule) {
149  // Assemble list of library files.
150  SmallVector<std::string> libraryFileNames;
151  if (failed(detail::expandPathsToMLIRFiles(transformLibraryPaths, context,
152  libraryFileNames)))
153  return failure();
154 
155  // Parse modules from library files.
156  SmallVector<OwningOpRef<ModuleOp>> parsedLibraries;
157  for (const std::string &libraryFileName : libraryFileNames) {
158  OwningOpRef<ModuleOp> parsedLibrary;
159  if (failed(transform::detail::parseTransformModuleFromFile(
160  context, libraryFileName, parsedLibrary)))
161  return failure();
162  parsedLibraries.push_back(std::move(parsedLibrary));
163  }
164 
165  // Merge parsed libraries into one module.
166  auto loc = FileLineColLoc::get(context, "<shared-library-module>", 0, 0);
167  OwningOpRef<ModuleOp> mergedParsedLibraries =
168  ModuleOp::create(loc, "__transform");
169  {
170  mergedParsedLibraries.get()->setAttr("transform.with_named_sequence",
171  UnitAttr::get(context));
172  // TODO: extend `mergeSymbolsInto` to support multiple `other` modules.
173  for (OwningOpRef<ModuleOp> &parsedLibrary : parsedLibraries) {
174  if (failed(transform::detail::mergeSymbolsInto(
175  mergedParsedLibraries.get(), std::move(parsedLibrary))))
176  return parsedLibrary->emitError()
177  << "failed to merge symbols into shared library module";
178  }
179  }
180 
181  transformModule = std::move(mergedParsedLibraries);
182  return success();
183 }

185 LogicalResult transform::applyTransformNamedSequence(
186  Operation *payload, Operation *transformRoot, ModuleOp transformModule,
187  const TransformOptions &options) {
188  // `transformModule` may not be modified.
189  if (transformModule && !transformModule->isAncestor(transformRoot)) {
190  OwningOpRef<Operation *> clonedTransformModule(transformModule->clone());
191  if (failed(detail::mergeSymbolsInto(
192  SymbolTable::getNearestSymbolTable(transformRoot),
193  std::move(clonedTransformModule)))) {
194  return payload->emitError() << "failed to merge symbols";
195  }
196  }
197 
198  LLVM_DEBUG(DBGS() << "Apply\n" << *transformRoot << "\n");
199  LLVM_DEBUG(DBGS() << "To\n" << *payload << "\n");
200 
201  // Apply the transform to the IR, do not enforce top-level constraints.
202  RaggedArray<MappedValue> noExtraMappings;
203  return applyTransforms(payload, cast<TransformOpInterface>(transformRoot),
204  noExtraMappings, options,
205  /*enforceToplevelTransformOp=*/false);
206 }
