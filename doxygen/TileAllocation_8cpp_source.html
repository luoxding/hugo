9 // This pass allocates SME tiles at the 'func.func' op level for ArmSME
10 // operations. It does this using a 16-bit tile mask that has a bit for each
11 // 128-bit element tile (ZA0.Q-ZA15.Q), the smallest ZA tile granule.
12 //
13 // The 128-bit tiles overlap with other element tiles as follows (see section
14 // B2.3.2 of SME spec [1]):
15 //
16 // Tile Overlaps
17 // ---------------------------------------------------------------------------
18 // ZA0.B ZA0.Q, ZA1.Q, ZA2.Q, ZA3.Q, ZA4.Q, ZA5.Q, ZA6.Q, ZA7.Q, ZA8.Q,
19 // ZA9.Q, ZA10.Q, ZA11.Q, ZA12.Q, ZA13.Q, ZA14.Q, ZA15.Q
20 // ZA0.H ZA0.Q, ZA2.Q, ZA4.Q, ZA6.Q, ZA8.Q, ZA10.Q, ZA12.Q, ZA14.Q
21 // ZA1.H ZA1.Q, ZA3.Q, ZA5.Q, ZA7.Q, ZA9.Q, ZA11.Q, ZA13.Q, ZA15.Q
22 // ZA0.S ZA0.Q, ZA4.Q, ZA8.Q, ZA12.Q
23 // ZA1.S ZA1.Q, ZA5.Q, ZA9.Q, ZA13.Q
24 // ZA2.S ZA2.Q, ZA6.Q, ZA10.Q, ZA14.Q
25 // ZA3.S ZA3.Q, ZA7.Q, ZA11.Q, ZA15.Q
26 // ZA0.D ZA0.Q, ZA8.Q
27 // ZA1.D ZA1.Q, ZA9.Q
28 // ZA2.D ZA2.Q, ZA10.Q
29 // ZA3.D ZA3.Q, ZA11.Q
30 // ZA4.D ZA4.Q, ZA12.Q
31 // ZA5.D ZA5.Q, ZA13.Q
32 // ZA6.D ZA6.Q, ZA14.Q
33 // ZA7.D ZA7.Q, ZA15.Q
34 //
35 // The tiles in use are tracked via a function attribute 'arm_sme.tiles_in_use'
36 // that is initalized during the first tile allocation within a function and
37 // updated on each subsequent allocation.
38 //
39 // [1] https://developer.arm.com/documentation/ddi0616/aa
40 //

110 /// Returns the set of masks relevant for the given type.
111 static ArrayRef<TileMask> getMasks(ArmSMETileType type) {
112  static constexpr std::array ZA_B_MASKS = {TileMask::kZA0B};
113  static constexpr std::array ZA_H_MASKS = {TileMask::kZA0H, TileMask::kZA1H};
114  static constexpr std::array ZA_S_MASKS = {TileMask::kZA0S, TileMask::kZA1S,
115  TileMask::kZA2S, TileMask::kZA3S};
116  static constexpr std::array ZA_D_MASKS = {
117  TileMask::kZA0D, TileMask::kZA1D, TileMask::kZA2D, TileMask::kZA3D,
118  TileMask::kZA4D, TileMask::kZA5D, TileMask::kZA6D, TileMask::kZA7D};
119  static constexpr std::array ZA_Q_MASKS = {
120  TileMask::kZA0Q, TileMask::kZA1Q, TileMask::kZA2Q, TileMask::kZA3Q,
121  TileMask::kZA4Q, TileMask::kZA5Q, TileMask::kZA6Q, TileMask::kZA7Q,
122  TileMask::kZA8Q, TileMask::kZA9Q, TileMask::kZA10Q, TileMask::kZA11Q,
123  TileMask::kZA12Q, TileMask::kZA13Q, TileMask::kZA14Q, TileMask::kZA15Q};
124  switch (type) {
125  case ArmSMETileType::ZAB:
126  return ZA_B_MASKS;
127  case ArmSMETileType::ZAH:
128  return ZA_H_MASKS;
129  case ArmSMETileType::ZAS:
130  return ZA_S_MASKS;
131  case ArmSMETileType::ZAD:
132  return ZA_D_MASKS;
133  case ArmSMETileType::ZAQ:
134  return ZA_Q_MASKS;
135  }
136 }

138 /// Allocates and returns a tile ID. Returns an error if there are no tiles
139 /// left.
140 static FailureOr<unsigned> allocateTileId(ArmSMETileType tileType,
141  TileMask &tilesInUse) {
142  auto masks = getMasks(tileType);
143  for (auto [tileId, tileMask] : llvm::enumerate(masks)) {
144  if ((tilesInUse & tileMask) == TileMask::kNone) {
145  tilesInUse |= tileMask;
146  return tileId;
147  }
148  }
149  return failure();
150 }

152 /// Collects transitive uses of a root value through control flow. This can
153 /// handle basic SCF constructs, along with control flow (br and cond_br).
154 /// Simple loops work at the SCF level, while more complex control flow can be
155 /// dealt with after lowering to CF. This is used to implement basic tile
156 /// allocation.
157 static void findDependantOps(Value rootValue,
158  SetVector<Operation *> &dependantOps) {
159  auto traverseCorrespondingValues = [&](auto inputValues, auto exitValues) {
160  for (auto [idx, value] : llvm::enumerate(inputValues)) {
161  if (value == rootValue)
162  findDependantOps(exitValues[idx], dependantOps);
163  }
164  };
165  for (Operation *user : rootValue.getUsers()) {
166  if (dependantOps.contains(user))
167  continue;
168  dependantOps.insert(user);
169  TypeSwitch<Operation *>(user)
170  .Case<cf::BranchOp>([&](auto branchOp) {
171  // (CF) Follow branch.
172  traverseCorrespondingValues(branchOp.getDestOperands(),
173  branchOp.getDest()->getArguments());
174  })
175  .Case<cf::CondBranchOp>([&](auto condBranchOp) {
176  // (CF) Follow true branch.
177  traverseCorrespondingValues(
178  condBranchOp.getTrueOperands(),
179  condBranchOp.getTrueDest()->getArguments());
180  // (CF) Follow false branch.
181  traverseCorrespondingValues(
182  condBranchOp.getFalseOperands(),
183  condBranchOp.getFalseDest()->getArguments());
184  })
185  .Case<LoopLikeOpInterface>([&](auto loopOp) {
186  // (SCF) Follow iter_args of (basic) loops (e.g. for loops).
187  traverseCorrespondingValues(loopOp.getInits(),
188  loopOp.getRegionIterArgs());
189  })
190  .Case<scf::YieldOp>([&](auto yieldOp) {
191  // (SCF) Follow yields of (basic) control flow (e.g. for loops).
192  auto parent = user->getParentOp();
193  traverseCorrespondingValues(user->getOperands(),
194  parent->getResults());
195  })
196  .Default([&](auto) {
197  // Otherwise, assume users of _any_ result are dependant.
198  for (Value result : user->getResults())
199  findDependantOps(result, dependantOps);
200  });
201  }
202 }

204 struct AssignTileIDsPattern
205  : public OpInterfaceRewritePattern<ArmSMETileOpInterface> {
206  using OpInterfaceRewritePattern::OpInterfaceRewritePattern;
207  LogicalResult matchAndRewrite(ArmSMETileOpInterface tileOp,
208  PatternRewriter &rewriter) const override {
209  if (tileOp.getTileId())
210  return failure();
211 
212  std::optional<ArmSMETileType> tileType = tileOp.getAllocatedTileType();
213  if (!tileType)
214  return rewriter.notifyMatchFailure(tileOp, "op does not allocate a tile");
215 
216  auto func = tileOp->getParentOfType<FunctionOpInterface>();
217  TileMask tilesInUse = TileMask::kNone;
218  if (auto tilesInUseAttr = llvm::dyn_cast_or_null<IntegerAttr>(
219  func->getDiscardableAttr(kTilesInUseAttr)))
220  tilesInUse = static_cast<TileMask>(tilesInUseAttr.getInt());
221 
222  auto tileId = allocateTileId(*tileType, tilesInUse);
223  if (failed(tileId))
224  return tileOp.emitError("ran out of SME virtual tiles!");
225 
226  func->setDiscardableAttr(kTilesInUseAttr,
227  rewriter.getI32IntegerAttr((unsigned)tilesInUse));
228 
229  // Find all the ops that (transitively) depend on this tile.
230  SetVector<Operation *> dependantOps;
231  findDependantOps(tileOp->getResult(0), dependantOps);
232 
233  // Set all operations dependent on `tileOp` to use the same tile ID.
234  // This is a naive tile allocation scheme, but works for common cases. For
235  // example, as this only allocates tile IDs to existing ops, it can't solve
236  // cases like this (%tileA and %tileB come from different root operations):
237  //
238  // %tile = scf.if %some_cond -> vector<[4]x[4]xi32> {
239  // scf.yield %tileA {tile_id = 0} : vector<[4]x[4]xi32>
240  // } else {
241  // scf.yield %tileB {tile_id = 1} : vector<[4]x[4]xi32>
242  // }
243  //
244  // This case would require allocating a new tile for the result of the
245  // scf.if, and moving the contents of %tileA or %tileB to result tile (based
246  // on the %some_cond).
247  auto tileIDAttr = rewriter.getI32IntegerAttr(*tileId);
248  tileOp.setTileId(tileIDAttr);
249  for (auto *op : dependantOps) {
250  if (auto tileOp = llvm::dyn_cast<ArmSMETileOpInterface>(op)) {
251  auto currentTileId = tileOp.getTileId();
252  if (currentTileId && unsigned(currentTileId.getInt()) != tileId)
253  return tileOp.emitOpError(
254  "already assigned different SME virtual tile!");
255  tileOp.setTileId(tileIDAttr);
256  }
257  }
258 
259  return success();
260  }
261 };

263 struct TileAllocationPass
264  : public arm_sme::impl::TileAllocationBase<TileAllocationPass> {
265  void runOnOperation() override {
266  RewritePatternSet patterns(&getContext());
267  patterns.add<AssignTileIDsPattern>(patterns.getContext());
268  GreedyRewriteConfig config;
269  // Setting useTopDownTraversal ensures tiles are allocated in program
270  // order.
271  config.useTopDownTraversal = true;
272  if (mlir::failed(mlir::applyPatternsAndFoldGreedily(
273  getOperation(), std::move(patterns), config))) {
274  signalPassFailure();
275  }
276  }
277 };
