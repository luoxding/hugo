32 static LogicalResult
33 checkOperandAffineExprRecursively(AffineExpr expr,
34  SmallVectorImpl<bool> &seenIds) {
35  switch (expr.getKind()) {
36  case AffineExprKind::Add: {
37  auto binOpExpr = cast<AffineBinaryOpExpr>(expr);
38  AffineExpr lhs = binOpExpr.getLHS();
39  AffineExpr rhs = binOpExpr.getRHS();
40  if (failed(checkOperandAffineExprRecursively(lhs, seenIds)))
41  return failure();
42  if (failed(checkOperandAffineExprRecursively(rhs, seenIds)))
43  return failure();
44  return success();
45  }
46  case AffineExprKind::Mul: {
47  auto binOpExpr = cast<AffineBinaryOpExpr>(expr);
48  AffineExpr lhs = binOpExpr.getLHS();
49  AffineExpr rhs = binOpExpr.getRHS();
50  AffineExpr dimExpr;
51  if (lhs.getKind() == AffineExprKind::DimId &&
52  rhs.getKind() == AffineExprKind::Constant) {
53  dimExpr = lhs;
54  } else if (rhs.getKind() == AffineExprKind::DimId &&
55  lhs.getKind() == AffineExprKind::Constant) {
56  dimExpr = rhs;
57  } else
58  return failure();
59  unsigned position = cast<AffineDimExpr>(dimExpr).getPosition();
60  if ((size_t)position >= seenIds.size() || seenIds[position])
61  return failure();
62  seenIds[position] = true;
63  return success();
64  }
65  case AffineExprKind::DimId: {
66  unsigned position = cast<AffineDimExpr>(expr).getPosition();
67  if ((size_t)position >= seenIds.size() || seenIds[position])
68  return failure();
69  seenIds[position] = true;
70  return success();
71  }
72  default:
73  return failure();
74  }
75 }

77 static FailureOr<llvm::SmallSet<unsigned, 2>>
78 checkOperandAffineExpr(AffineExpr expr, unsigned numDims) {
79  SmallVector<bool> seenIds(numDims, false);
80  if (failed(checkOperandAffineExprRecursively(expr, seenIds)))
81  return failure();
82 
83  llvm::SmallSet<unsigned, 2> positions;
84  for (auto it : llvm::enumerate(seenIds)) {
85  if (it.value())
86  positions.insert((unsigned)it.index());
87  }
88  return positions;
89 }

95 FailureOr<std::pair<bool, MeshShardingAttr>>
96 mesh::getMeshShardingAttr(OpResult result) {
97  Value val = result.cast<Value>();
98  bool anyShardedForDef = llvm::any_of(val.getUsers(), [](Operation *user) {
99  auto shardOp = llvm::dyn_cast<mesh::ShardOp>(user);
100  if (!shardOp)
101  return false;
102  return !shardOp.getAnnotateForUsers();
103  });
104 
105  if (anyShardedForDef) {
106  // expected to have exact one use if it has a use of `mesh.shard` without
107  // unit attr annotate_for_users
108  if (!val.hasOneUse())
109  return failure();
110  auto shardOp = llvm::cast<mesh::ShardOp>(*val.getUsers().begin());
111  return std::make_pair(false, shardOp.getShard());
112  }
113 
114  bool anyShardedForUsers = llvm::any_of(val.getUsers(), [](Operation *user) {
115  auto shardOp = llvm::dyn_cast<mesh::ShardOp>(user);
116  if (!shardOp)
117  return false;
118  return shardOp.getAnnotateForUsers();
119  });
120  if (anyShardedForUsers) {
121  SmallVector<ShardOp> shardOps;
122  for (Operation *user : val.getUsers()) {
123  ShardOp shardOp = llvm::dyn_cast<ShardOp>(user);
124  if (shardOp)
125  shardOps.push_back(shardOp);
126  }
127  MeshShardingAttr shardForDef = shardOps[0].getShard();
128  for (size_t i = 1; i < shardOps.size(); ++i) {
129  // TODO: Deduce a reasonable mesh sharding attr for def when they are
130  // different
131  assert(shardOps[i].getShard() == shardForDef &&
132  "only support all shard ops have the same mesh sharding attr");
133  }
134  return std::make_pair(true, shardForDef);
135  }
136  return failure();
137 }

152 LogicalResult mesh::ShardingInterface::verifyShardingInterfaceImpl() {
153  Operation *op = getOperation();
154 
155  // check operands and results type
156  for (Type type : op->getOperandTypes())
157  if (!llvm::isa<RankedTensorType>(type))
158  return failure();
159  for (Type type : op->getResultTypes())
160  if (!llvm::isa<RankedTensorType>(type))
161  return failure();
162 
163  // check loop types
164  SmallVector<IteratorType> loopTypes = getLoopIteratorTypes();
165  if (loopTypes.size() == 0)
166  return failure();
167 
168  // check maps
169  SmallVector<AffineMap> maps = getIndexingMaps();
170  if (maps.size() == 0)
171  return failure();
172  unsigned numOperands = op->getNumOperands();
173  unsigned numResults = op->getNumResults();
174  if (numOperands + numResults != maps.size())
175  return failure();
176 
177  for (OpResult result : op->getResults()) {
178  auto resultType = result.getType().dyn_cast<RankedTensorType>();
179  if (!resultType)
180  return failure();
181  AffineMap map = maps[numOperands + result.getResultNumber()];
182  if (!map.isProjectedPermutation()) {
183  return failure();
184  }
185  }
186 
187  return success();
188 }

215 // Update the given `shardingOption` according to `meshAxes` and `loopIdx`
216 static LogicalResult fillShardingOption(Operation *op,
217  ShardingOption &shardingOption,
218  SymbolRefAttr cluster,
219  ArrayRef<int32_t> meshAxes,
220  unsigned loopIdx) {
221  if ((shardingOption.cluster && cluster &&
222  shardingOption.cluster != cluster) ||
223  (!shardingOption.shardingArray[loopIdx].empty() &&
224  shardingOption.shardingArray[loopIdx] != meshAxes)) {
225  LLVM_DEBUG(DBGS() << "sharding option conflicts on loop iterator "
226  << loopIdx << "\n");
227  return failure();
228  }
229  for (size_t i = 0; i < shardingOption.shardingArray.size(); ++i) {
230  if (i == loopIdx)
231  continue;
232 
233  for (int32_t axis : meshAxes) {
234  if (std::find(shardingOption.shardingArray[i].begin(),
235  shardingOption.shardingArray[i].end(),
236  axis) != shardingOption.shardingArray[i].end()) {
237  LLVM_DEBUG(DBGS() << "sharding option conflicts because mesh axes "
238  << axis << " duplicate");
239  return failure();
240  }
241  }
242  }
243  if (cluster)
244  shardingOption.cluster = cluster;
245  if (shardingOption.shardingArray[loopIdx].empty())
246  shardingOption.shardingArray[loopIdx].append(meshAxes.begin(),
247  meshAxes.end());
248  return success();
249 }

253 FailureOr<ShardingOption> mesh::detail::defaultGetShardingOption(
254  Operation *op, ArrayRef<MeshShardingAttr> operandShardings,
255  ArrayRef<MeshShardingAttr> resultShardings) {
256  ShardingInterface shardingOp = llvm::cast<ShardingInterface>(op);
257  ShardingOption shardingOption;
258 
259  if (failed(shardingOp.verifyShardingInterfaceImpl()))
260  return op->emitOpError() << "invalid sharding interface implementation";
261  SmallVector<IteratorType> loopTypes = shardingOp.getLoopIteratorTypes();
262  SmallVector<AffineMap> maps = shardingOp.getIndexingMaps();
263  unsigned numOperands = op->getNumOperands();
264  shardingOption.shardingArray.resize(loopTypes.size());
265  llvm::SmallVector<int32_t> partialMeshAxes;
266  Partial partialType;
267  llvm::SmallSet<unsigned, 4> visitedLoopIndices;
268  bool anyShardingInResultsOrOperands = false;
269 
270  // 1. Fill sharding option based on op results
271  for (auto shardingIt : llvm::enumerate(resultShardings)) {
272  MeshShardingAttr shardAttr = shardingIt.value();
273  if (!shardAttr)
274  continue;
275  AffineMap map = maps[numOperands + shardingIt.index()];
276  anyShardingInResultsOrOperands = true;
277  // Handle the split axes: calculate the corresponding loop index for each
278  // split axes sub-array, and then store the sub-array to
279  // shardingOption[index]
280  for (auto it : llvm::zip(map.getResults(), shardAttr.getSplitAxes())) {
281  AffineExpr expr = std::get<0>(it);
282  ArrayRef<int32_t> axes = std::get<1>(it).asArrayRef();
283  auto dim = cast<AffineDimExpr>(expr);
284  unsigned index = dim.getPosition();
285  visitedLoopIndices.insert(index);
286  if (failed(fillShardingOption(op, shardingOption, shardAttr.getCluster(),
287  axes, index)))
288  return failure();
289  }
290 
291  // Handle the partial axes: at this stage, the exact loop index/indices
292  // cannot be decided because there could be multiple reduction loops.
293  ArrayRef<int32_t> partialAxes = shardAttr.getPartialAxes();
294  if (!partialAxes.empty()) {
295  if (!partialMeshAxes.empty())
296  return op->emitOpError() << "at most one result with partial axes is "
297  "supported at present";
298  partialType = shardAttr.getPartialType();
299  partialMeshAxes.append(partialAxes.begin(), partialAxes.end());
300  // Add all the reduction loop indices to `visitedLoopIndices` if
301  // `partialAxes` is not empty
302  for (size_t loopIdx = 0; loopIdx < loopTypes.size(); ++loopIdx) {
303  if (isReductionLoop(loopTypes[loopIdx]))
304  visitedLoopIndices.insert(loopIdx);
305  }
306  }
307  }
308 
309  // 2. Fill sharding option based on operands
310  for (auto shardingIt : llvm::enumerate(operandShardings)) {
311  MeshShardingAttr shardAttr = shardingIt.value();
312  if (!shardAttr)
313  continue;
314 
315  anyShardingInResultsOrOperands = true;
316  AffineMap map = maps[shardingIt.index()];
317  unsigned numDims = map.getNumDims();
318 
319  // Handle the split axes. Partial axes don't need to be handled because they
320  // only affect the defining op of the operand.
321  //
322  // TODO: Change to process the operands with single loop index first and
323  // then the operands with multiple loop indices.
324  for (auto it : llvm::zip(map.getResults(), shardAttr.getSplitAxes())) {
325  AffineExpr expr = std::get<0>(it);
326  ArrayRef<int32_t> axes = std::get<1>(it).asArrayRef();
327  FailureOr<llvm::SmallSet<unsigned, 2>> loopIndices =
328  checkOperandAffineExpr(expr, numDims);
329  if (failed(loopIndices))
330  return op->emitOpError()
331  << "operand's affine expression is restricted to const_i * "
332  "dim_i + const_j + dim_j + ...";
333  if (loopIndices->empty())
334  continue;
335  if (loopIndices->size() == 1) {
336  unsigned loopIdx = *loopIndices->begin();
337  visitedLoopIndices.insert(loopIdx);
338  if (failed(fillShardingOption(op, shardingOption,
339  shardAttr.getCluster(), axes, loopIdx)))
340  return failure();
341  }
342  // If multiple loop indices correspond to a dimension of an operand, it is
343  // difficult to infer which loop indices are responsible for sharding.
344  // Therefore, the exact loop index must be specified by others.
345  if (loopIndices->size() > 1) {
346  bool seenLoopIndices = false;
347  for (unsigned loopIdx : *loopIndices) {
348  if (visitedLoopIndices.contains(loopIdx)) {
349  seenLoopIndices = true;
350  break;
351  }
352  }
353  if (!seenLoopIndices)
354  return op->emitOpError()
355  << "the operand " << shardingIt.index()
356  << " has multiple loop indices in a dimension, but none of "
357  "them could be found in the exactly specified annotation "
358  "of op results or operands.";
359  }
360  }
361  }
362 
363  // 3. Finalize sharding option
364  if (!partialMeshAxes.empty()) {
365  bool anyNonEmptyReductionLoop = llvm::any_of(
366  llvm::enumerate(shardingOption.shardingArray), [&](auto it) {
367  SmallVector<int32_t> &subArray = it.value();
368  int64_t idx = it.index();
369  return isReductionLoop(loopTypes[idx]) && !subArray.empty();
370  });
371  if (!anyNonEmptyReductionLoop) {
372  bool filled = false;
373  for (size_t idx = 0; idx < loopTypes.size(); ++idx) {
374  if (isReductionLoop(loopTypes[idx]) &&
375  areReductionAndPartialMatch(loopTypes[idx], partialType)) {
376  std::ignore = fillShardingOption(op, shardingOption, nullptr,
377  partialMeshAxes, idx);
378  filled = true;
379  break;
380  }
381  }
382  if (!filled)
383  return op->emitOpError() << "no matched reduction loop found for the "
384  "result's partial type";
385  }
386  }
387  removeTrailingEmptySubArray(shardingOption.shardingArray);
388  if (!anyShardingInResultsOrOperands)
389  shardingOption.empty = true;
390  return shardingOption;
391 }

399 // To add a `mesh.shard` op for the given result, based on the details provided
400 // in `shardingOption`, `map`, and `loopTypes`.
401 static LogicalResult addShardOp(OpBuilder &b, OpResult result,
402  const ShardingOption &shardingOption,
403  AffineMap map,
404  ArrayRef<IteratorType> loopTypes) {
405  FailureOr<std::pair<bool, MeshShardingAttr>> maybeSharding =
406  getMeshShardingAttr(result);
407  if (succeeded(maybeSharding) && !maybeSharding->first)
408  return success();
409 
410  auto resultType = result.getType().cast<RankedTensorType>();
411  SmallVector<SmallVector<int32_t>> splitAxes(resultType.getRank());
412  SmallVector<int32_t> partialAxes;
413 
414  // process the split axes
415  for (auto it : llvm::enumerate(map.getResults())) {
416  AffineExpr expr = it.value();
417  // `expr` must be an `AffineDimExpr` because `map` is verified by
418  // isProjectedPermutation
419  auto dim = cast<AffineDimExpr>(expr);
420  unsigned loopIdx = dim.getPosition();
421  if (loopIdx < shardingOption.shardingArray.size())
422  splitAxes[it.index()].append(shardingOption.shardingArray[loopIdx]);
423  }
424 
425  // process the partial axes
426  // partialType will be ignored if partialAxes is empty
427  Partial partialType = Partial::Sum;
428  for (auto it : llvm::zip(loopTypes, shardingOption.shardingArray)) {
429  IteratorType iType = std::get<0>(it);
430  if (isReductionLoop(iType)) {
431  Partial curPartialType = getPartialTypeFromReduction(iType);
432  if (!partialAxes.empty())
433  assert(partialType == curPartialType &&
434  "Only one reduction type is supported");
435  partialType = curPartialType;
436  const SmallVector<int32_t> &axis = std::get<1>(it);
437  partialAxes.append(axis);
438  }
439  }
440 
441  removeTrailingEmptySubArray(splitAxes);
442  MeshShardingAttr shardAttr =
443  MeshShardingAttr::get(b.getContext(), shardingOption.cluster, splitAxes,
444  partialAxes, partialType);
445  OpBuilder::InsertionGuard guard(b);
446  b.setInsertionPointAfterValue(result);
447  auto shardOp = b.create<ShardOp>(result.getLoc(), resultType, result,
448  shardAttr, /*annotate_for_users*/ false);
449  result.replaceAllUsesExcept(shardOp, shardOp);
450  return success();
451 }

453 // To add a `mesh.shard` op for the given operand, based on the details provided
454 // in `shardingOption`, `map`, and `loopTypes`.
455 static LogicalResult addShardOp(OpBuilder &b, OpOperand &opOperand,
456  const ShardingOption &shardingOption,
457  AffineMap map,
458  ArrayRef<IteratorType> loopTypes) {
459  auto maybeShardingAttr = getMeshShardingAttr(opOperand);
460  if (succeeded(maybeShardingAttr) && maybeShardingAttr->first)
461  return success();
462  Value operand = opOperand.get();
463  auto operandType = operand.getType().cast<RankedTensorType>();
464  SmallVector<SmallVector<int32_t>> splitAxes(operandType.getRank());
465  unsigned numDims = map.getNumDims();
466  for (auto it : llvm::enumerate(map.getResults())) {
467  int64_t idx = it.index();
468  AffineExpr expr = it.value();
469  FailureOr<llvm::SmallSet<unsigned, 2>> loopIndices =
470  checkOperandAffineExpr(expr, numDims);
471  if (failed(loopIndices))
472  return failure();
473  SmallVector<unsigned> shardedLoopIndices;
474  for (unsigned loopIdx : *loopIndices) {
475  if ((size_t)loopIdx < shardingOption.shardingArray.size() &&
476  !shardingOption.shardingArray[loopIdx].empty())
477  shardedLoopIndices.push_back(loopIdx);
478  }
479  // mostly one sharded loop index is accepted
480  if (shardedLoopIndices.size() > 1)
481  return failure();
482  if (shardedLoopIndices.size() == 1) {
483  splitAxes[idx].append(
484  shardingOption.shardingArray[shardedLoopIndices[0]]);
485  }
486  }
487 
488  removeTrailingEmptySubArray(splitAxes);
489  MeshShardingAttr shardAttr =
490  MeshShardingAttr::get(b.getContext(), shardingOption.cluster, splitAxes);
491  OpBuilder::InsertionGuard guard(b);
492  b.setInsertionPoint(opOperand.getOwner());
493  auto shardOp = b.create<ShardOp>(operand.getLoc(), operandType, operand,
494  shardAttr, true);
495  opOperand.set(shardOp);
496 
497  return success();
498 }

502 LogicalResult mesh::detail::defaultAddShardingAnnotations(
503  Operation *op, OpBuilder &b, const ShardingOption &shardingOption) {
504  ShardingInterface shardingOp = llvm::cast<ShardingInterface>(op);
505  SmallVector<IteratorType> loopTypes = shardingOp.getLoopIteratorTypes();
506  SmallVector<AffineMap> maps = shardingOp.getIndexingMaps();
507  unsigned numOperands = op->getNumOperands();
508 
509  // 1. add mesh.shard ops for all op results
510  for (OpResult result : op->getResults()) {
511  if (failed(addShardOp(b, result, shardingOption,
512  maps[numOperands + result.getResultNumber()],
513  loopTypes)))
514  return failure();
515  }
516 
517  // 2. add mesh.shard ops for all operands
518  for (OpOperand &opOperand : op->getOpOperands()) {
519  if (failed(addShardOp(b, opOperand, shardingOption,
520  maps[opOperand.getOperandNumber()], loopTypes)))
521  return failure();
522  }
523 
524  return success();
525 }
