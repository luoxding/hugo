133 bool LLVM::LoadOp::canUsesBeRemoved(
134  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
135  SmallVectorImpl<OpOperand *> &newBlockingUses) {
136  if (blockingUses.size() != 1)
137  return false;
138  Value blockingUse = (*blockingUses.begin())->get();
139  // If the blocking use is the slot ptr itself, there will be enough
140  // context to reconstruct the result of the load at removal time, so it can

332 bool LLVM::GEPOp::canRewire(const DestructurableMemorySlot &slot,
333  SmallPtrSetImpl<Attribute> &usedIndices,
334  SmallVectorImpl<MemorySlot> &mustBeSafelyUsed) {
335  auto basePtrType = llvm::dyn_cast<LLVM::LLVMPointerType>(getBase().getType());
336  if (!basePtrType)
337  return false;
338 
339  if (getBase() != slot.ptr || slot.elemType != getElemType())
340  return false;
341  if (!isFirstIndexZero(*this))
342  return false;
343  Type reachedType = getResultPtrElementType();
344  if (!reachedType || getIndices().size() < 2)
345  return false;
346  auto firstLevelIndex = dyn_cast<IntegerAttr>(getIndices()[1]);
347  if (!firstLevelIndex)
348  return false;
349  assert(slot.elementPtrs.contains(firstLevelIndex));
350  if (!llvm::isa<LLVM::LLVMPointerType>(slot.elementPtrs.at(firstLevelIndex)))
351  return false;
352  mustBeSafelyUsed.emplace_back<MemorySlot>({getResult(), reachedType});
353  usedIndices.insert(firstLevelIndex);
354  return true;
355 }

357 DeletionKind LLVM::GEPOp::rewire(const DestructurableMemorySlot &slot,
358  DenseMap<Attribute, MemorySlot> &subslots,
359  RewriterBase &rewriter) {
360  IntegerAttr firstLevelIndex =
361  llvm::dyn_cast_if_present<IntegerAttr>(getIndices()[1]);
362  const MemorySlot &newSlot = subslots.at(firstLevelIndex);
363 
364  ArrayRef<int32_t> remainingIndices = getRawConstantIndices().slice(2);
365 
366  // If the GEP would become trivial after this transformation, eliminate it.
367  // A GEP should only be eliminated if it has no indices (except the first
368  // pointer index), as simplifying GEPs with all-zero indices would eliminate
369  // structure information useful for further destruction.
370  if (remainingIndices.empty()) {
371  rewriter.replaceAllUsesWith(getResult(), newSlot.ptr);
372  return DeletionKind::Delete;
373  }
374 
375  rewriter.updateRootInPlace(*this, [&]() {
376  // Rewire the indices by popping off the second index.
377  // Start with a single zero, then add the indices beyond the second.
378  SmallVector<int32_t> newIndices(1);
379  newIndices.append(remainingIndices.begin(), remainingIndices.end());
380  setRawConstantIndices(newIndices);
381 
382  // Rewire the pointed type.
383  setElemType(newSlot.elemType);
384 
385  // Rewire the pointer.
386  getBaseMutable().assign(newSlot.ptr);
387  });
388 
389  return DeletionKind::Keep;
390 }

398 /// Returns the length of the given memory intrinsic in bytes if it can be known
399 /// at compile-time on a best-effort basis, nothing otherwise.
400 template <class MemIntr>
401 std::optional<uint64_t> getStaticMemIntrLen(MemIntr op) {
402  APInt memIntrLen;
403  if (!matchPattern(op.getLen(), m_ConstantInt(&memIntrLen)))
404  return {};
405  if (memIntrLen.getBitWidth() > 64)
406  return {};
407  return memIntrLen.getZExtValue();
408 }

424 /// Returns whether one can be sure the memory intrinsic does not write outside
425 /// of the bounds of the given slot, on a best-effort basis.
426 template <class MemIntr>
427 static bool definitelyWritesOnlyWithinSlot(MemIntr op, const MemorySlot &slot,
428  DataLayout &dataLayout) {
429  if (!isa<LLVM::LLVMPointerType>(slot.ptr.getType()) ||
430  op.getDst() != slot.ptr)
431  return false;
432 
433  std::optional<uint64_t> memIntrLen = getStaticMemIntrLen(op);
434  return memIntrLen && *memIntrLen <= dataLayout.getTypeSize(slot.elemType);
435 }

458 Value LLVM::MemsetOp::getStored(const MemorySlot &slot,
459  RewriterBase &rewriter) {
460  // TODO: Support non-integer types.
461  return TypeSwitch<Type, Value>(slot.elemType)
462  .Case([&](IntegerType intType) -> Value {
463  if (intType.getWidth() == 8)
464  return getVal();
465 
466  assert(intType.getWidth() % 8 == 0);
467 
468  // Build the memset integer by repeatedly shifting the value and
469  // or-ing it with the previous value.
470  uint64_t coveredBits = 8;
471  Value currentValue =
472  rewriter.create<LLVM::ZExtOp>(getLoc(), intType, getVal());
473  while (coveredBits < intType.getWidth()) {
474  Value shiftBy =
475  rewriter.create<LLVM::ConstantOp>(getLoc(), intType, coveredBits);
476  Value shifted =
477  rewriter.create<LLVM::ShlOp>(getLoc(), currentValue, shiftBy);
478  currentValue =
479  rewriter.create<LLVM::OrOp>(getLoc(), currentValue, shifted);
480  coveredBits *= 2;
481  }
482 
483  return currentValue;
484  })
485  .Default([](Type) -> Value {
486  llvm_unreachable(
487  "getStored should not be called on memset to unsupported type");
488  });
489 }

543 DeletionKind LLVM::MemsetOp::rewire(const DestructurableMemorySlot &slot,
544  DenseMap<Attribute, MemorySlot> &subslots,
545  RewriterBase &rewriter) {
546  std::optional<DenseMap<Attribute, Type>> types =
547  slot.elemType.cast<DestructurableTypeInterface>().getSubelementIndexMap();
548 
549  IntegerAttr memsetLenAttr;
550  bool successfulMatch =
551  matchPattern(getLen(), m_Constant<IntegerAttr>(&memsetLenAttr));
552  (void)successfulMatch;
553  assert(successfulMatch);
554 
555  bool packed = false;
556  if (auto structType = dyn_cast<LLVM::LLVMStructType>(slot.elemType))
557  packed = structType.isPacked();
558 
559  Type i32 = IntegerType::get(getContext(), 32);
560  DataLayout dataLayout = DataLayout::closest(*this);
561  uint64_t memsetLen = memsetLenAttr.getValue().getZExtValue();
562  uint64_t covered = 0;
563  for (size_t i = 0; i < types->size(); i++) {
564  // Create indices on the fly to get elements in the right order.
565  Attribute index = IntegerAttr::get(i32, i);
566  Type elemType = types->at(index);
567  uint64_t typeSize = dataLayout.getTypeSize(elemType);
568 
569  if (!packed)
570  covered =
571  llvm::alignTo(covered, dataLayout.getTypeABIAlignment(elemType));
572 
573  if (covered >= memsetLen)
574  break;
575 
576  // If this subslot is used, apply a new memset to it.
577  // Otherwise, only compute its offset within the original memset.
578  if (subslots.contains(index)) {
579  uint64_t newMemsetSize = std::min(memsetLen - covered, typeSize);
580 
581  Value newMemsetSizeValue =
582  rewriter
583  .create<LLVM::ConstantOp>(
584  getLen().getLoc(),
585  IntegerAttr::get(memsetLenAttr.getType(), newMemsetSize))
586  .getResult();
587 
588  rewriter.create<LLVM::MemsetOp>(getLoc(), subslots.at(index).ptr,
589  getVal(), newMemsetSizeValue,
590  getIsVolatile());
591  }
592 
593  covered += typeSize;
594  }
595 
596  return DeletionKind::Delete;
597 }

659 template <class MemcpyLike>
660 static bool memcpyCanRewire(MemcpyLike op, const DestructurableMemorySlot &slot,
661  SmallPtrSetImpl<Attribute> &usedIndices,
662  SmallVectorImpl<MemorySlot> &mustBeSafelyUsed) {
663  if (op.getIsVolatile())
664  return false;
665 
666  if (!slot.elemType.cast<DestructurableTypeInterface>()
667  .getSubelementIndexMap())
668  return false;
669 
670  if (!areAllIndicesI32(slot))
671  return false;
672 
673  // Only full copies are supported.
674  DataLayout dataLayout = DataLayout::closest(op);
675  if (getStaticMemIntrLen(op) != dataLayout.getTypeSize(slot.elemType))
676  return false;
677 
678  if (op.getSrc() == slot.ptr)
679  for (Attribute index : llvm::make_first_range(slot.elementPtrs))
680  usedIndices.insert(index);
681 
682  return true;
683 }

711 /// Rewires a memcpy-like operation. Only copies to or from the full slot are
712 /// supported.
713 template <class MemcpyLike>
714 static DeletionKind memcpyRewire(MemcpyLike op,
715  const DestructurableMemorySlot &slot,
716  DenseMap<Attribute, MemorySlot> &subslots,
717  RewriterBase &rewriter) {
718  if (subslots.empty())
719  return DeletionKind::Delete;
720 
721  DataLayout layout = DataLayout::closest(op);
722 
723  assert((slot.ptr == op.getDst()) != (slot.ptr == op.getSrc()));
724  bool isDst = slot.ptr == op.getDst();
725 
726 #ifndef NDEBUG
727  size_t slotsTreated = 0;
728 #endif
729 
730  // It was previously checked that index types are consistent, so this type can
731  // be fetched now.
732  Type indexType = cast<IntegerAttr>(subslots.begin()->first).getType();
733  for (size_t i = 0, e = slot.elementPtrs.size(); i != e; i++) {
734  Attribute index = IntegerAttr::get(indexType, i);
735  if (!subslots.contains(index))
736  continue;
737  const MemorySlot &subslot = subslots.at(index);
738 
739 #ifndef NDEBUG
740  slotsTreated++;
741 #endif
742 
743  // First get a pointer to the equivalent of this subslot from the source
744  // pointer.
745  SmallVector<LLVM::GEPArg> gepIndices{
746  0, static_cast<int32_t>(
747  cast<IntegerAttr>(index).getValue().getZExtValue())};
748  Value subslotPtrInOther = rewriter.create<LLVM::GEPOp>(
749  op.getLoc(), LLVM::LLVMPointerType::get(op.getContext()), slot.elemType,
750  isDst ? op.getSrc() : op.getDst(), gepIndices);
751 
752  // Then create a new memcpy out of this source pointer.
753  createMemcpyLikeToReplace(rewriter, layout, op,
754  isDst ? subslot.ptr : subslotPtrInOther,
755  isDst ? subslotPtrInOther : subslot.ptr,
756  subslot.elemType, op.getIsVolatile());
757  }
758 
759  assert(subslots.size() == slotsTreated);
760 
761  return DeletionKind::Delete;
762 }
