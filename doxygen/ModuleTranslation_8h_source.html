232  /// Concrete CRTP base class for ModuleTranslation stack frames. When
233  /// translating operations with regions, users of ModuleTranslation can store
234  /// state on ModuleTranslation stack before entering the region and inspect
235  /// it when converting operations nested within that region. Users are
236  /// expected to derive this class and put any relevant information into fields
237  /// of the derived class. The usual isa/dyn_cast functionality is available
238  /// for instances of derived classes.
239  template <typename Derived>
240  class StackFrameBase : public StackFrame {
241  public:
242  explicit StackFrameBase() : StackFrame(TypeID::get<Derived>()) {}
243  };
244 
245  /// Creates a stack frame of type `T` on ModuleTranslation stack. `T` must
246  /// be derived from `StackFrameBase<T>` and constructible from the provided
247  /// arguments. Doing this before entering the region of the op being
248  /// translated makes the frame available when translating ops within that
249  /// region.
250  template <typename T, typename... Args>
251  void stackPush(Args &&...args) {
252  static_assert(
253  std::is_base_of<StackFrame, T>::value,
254  "can only push instances of StackFrame on ModuleTranslation stack");
255  stack.push_back(std::make_unique<T>(std::forward<Args>(args)...));
256  }
257 
258  /// Pops the last element from the ModuleTranslation stack.
259  void stackPop() { stack.pop_back(); }
260 
261  /// Calls `callback` for every ModuleTranslation stack frame of type `T`
262  /// starting from the top of the stack.
263  template <typename T>
264  WalkResult
265  stackWalk(llvm::function_ref<WalkResult(const T &)> callback) const {
266  static_assert(std::is_base_of<StackFrame, T>::value,
267  "expected T derived from StackFrame");
268  if (!callback)
269  return WalkResult::skip();
270  for (const std::unique_ptr<StackFrame> &frame : llvm::reverse(stack)) {
271  if (T *ptr = dyn_cast_or_null<T>(frame.get())) {
272  WalkResult result = callback(*ptr);
273  if (result.wasInterrupted())
274  return result;
275  }
276  }
277  return WalkResult::advance();
278  }
