63 void StorageLayout::foreachField(
64  llvm::function_ref<bool(FieldIndex, SparseTensorFieldKind, Level,
65  LevelType)>
66  callback) const {
67  const auto lvlTypes = enc.getLvlTypes();
68  const Level lvlRank = enc.getLvlRank();
69  const Level cooStart = SparseTensorType(enc).getCOOStart();
70  const Level end = cooStart == lvlRank ? cooStart : cooStart + 1;
71  FieldIndex fieldIdx = kDataFieldStartingIdx;
72  // Per-level storage.
73  for (Level l = 0; l < end; l++) {
74  const auto lt = lvlTypes[l];
75  if (isWithPosLT(lt)) {
76  if (!(callback(fieldIdx++, SparseTensorFieldKind::PosMemRef, l, lt)))
77  return;
78  }
79  if (isWithCrdLT(lt)) {
80  if (!(callback(fieldIdx++, SparseTensorFieldKind::CrdMemRef, l, lt)))
81  return;
82  }
83  }
84  // The values array.
85  if (!(callback(fieldIdx++, SparseTensorFieldKind::ValMemRef, kInvalidLevel,
86  LevelType::Undef)))
87  return;
88  // Put metadata at the end.
89  if (!(callback(fieldIdx++, SparseTensorFieldKind::StorageSpec, kInvalidLevel,
90  LevelType::Undef)))
91  return;
92 }

94 void sparse_tensor::foreachFieldAndTypeInSparseTensor(
95  SparseTensorType stt,
96  llvm::function_ref<bool(Type, FieldIndex, SparseTensorFieldKind, Level,
97  LevelType)>
98  callback) {
99  assert(stt.hasEncoding());
100  // Construct the basic types.
101  const Type crdType = stt.getCrdType();
102  const Type posType = stt.getPosType();
103  const Type eltType = stt.getElementType();
104 
105  const Type specType = StorageSpecifierType::get(stt.getEncoding());
106  // memref<? x pos> positions
107  const Type posMemType = MemRefType::get({ShapedType::kDynamic}, posType);
108  // memref<? x crd> coordinates
109  const Type crdMemType = MemRefType::get({ShapedType::kDynamic}, crdType);
110  // memref<? x eltType> values
111  const Type valMemType = MemRefType::get({ShapedType::kDynamic}, eltType);
112 
113  StorageLayout(stt).foreachField([specType, posMemType, crdMemType, valMemType,
114  callback](FieldIndex fieldIdx,
115  SparseTensorFieldKind fieldKind,
116  Level lvl, LevelType lt) -> bool {
117  switch (fieldKind) {
118  case SparseTensorFieldKind::StorageSpec:
119  return callback(specType, fieldIdx, fieldKind, lvl, lt);
120  case SparseTensorFieldKind::PosMemRef:
121  return callback(posMemType, fieldIdx, fieldKind, lvl, lt);
122  case SparseTensorFieldKind::CrdMemRef:
123  return callback(crdMemType, fieldIdx, fieldKind, lvl, lt);
124  case SparseTensorFieldKind::ValMemRef:
125  return callback(valMemType, fieldIdx, fieldKind, lvl, lt);
126  };
127  llvm_unreachable("unrecognized field kind");
128  });
129 }

154 std::pair<FieldIndex, unsigned>
155 StorageLayout::getFieldIndexAndStride(SparseTensorFieldKind kind,
156  std::optional<Level> lvl) const {
157  FieldIndex fieldIdx = kInvalidFieldIndex;
158  unsigned stride = 1;
159  if (kind == SparseTensorFieldKind::CrdMemRef) {
160  assert(lvl.has_value());
161  const Level cooStart = SparseTensorType(enc).getCOOStart();
162  const Level lvlRank = enc.getLvlRank();
163  if (lvl.value() >= cooStart && lvl.value() < lvlRank) {
164  lvl = cooStart;
165  stride = lvlRank - cooStart;
166  }
167  }
168  foreachField([lvl, kind, &fieldIdx](FieldIndex fIdx,
169  SparseTensorFieldKind fKind, Level fLvl,
170  LevelType lt) -> bool {
171  if ((lvl && fLvl == lvl.value() && kind == fKind) ||
172  (kind == fKind && fKind == SparseTensorFieldKind::ValMemRef)) {
173  fieldIdx = fIdx;
174  // Returns false to break the iteration.
175  return false;
176  }
177  return true;
178  });
179  assert(fieldIdx != kInvalidFieldIndex);
180  return std::pair<FieldIndex, unsigned>(fieldIdx, stride);
181 }

382 SmallVector<int64_t>
383 SparseTensorEncodingAttr::tranlateShape(ArrayRef<int64_t> srcShape,
384  CrdTransDirectionKind dir) const {
385  if (isIdentity())
386  return SmallVector<int64_t>(srcShape);
387 
388  SmallVector<int64_t> ret;
389  unsigned rank =
390  dir == CrdTransDirectionKind::dim2lvl ? getLvlRank() : getDimRank();
391  ret.reserve(rank);
392 
393  if (isPermutation()) {
394  for (unsigned r = 0; r < rank; r++) {
395  unsigned trans = dir == CrdTransDirectionKind::dim2lvl ? toDim(*this, r)
396  : toLvl(*this, r);
397  ret.push_back(srcShape[trans]);
398  }
399  return ret;
400  }
401 
402  // Handle non-permutation maps.
403  AffineMap transMap =
404  dir == CrdTransDirectionKind::dim2lvl ? getDimToLvl() : getLvlToDim();
405 
406  SmallVector<AffineExpr> dimRep;
407  dimRep.reserve(srcShape.size());
408  for (int64_t sz : srcShape) {
409  if (!ShapedType::isDynamic(sz)) {
410  // Push back the max coordinate for the given dimension/level size.
411  dimRep.push_back(getAffineConstantExpr(sz - 1, getContext()));
412  } else {
413  // A dynamic size, use a AffineDimExpr to symbolize the value.
414  dimRep.push_back(getAffineDimExpr(dimRep.size(), getContext()));
415  }
416  };
417 
418  for (AffineExpr exp : transMap.getResults()) {
419  // Do constant propagation on the affine map.
420  AffineExpr evalExp =
421  simplifyAffineExpr(exp.replaceDims(dimRep), srcShape.size(), 0);
422  // use llvm namespace here to avoid ambiguity
423  if (auto c = llvm::dyn_cast<AffineConstantExpr>(evalExp)) {
424  ret.push_back(c.getValue() + 1);
425  } else {
426  if (auto mod = llvm::dyn_cast<AffineBinaryOpExpr>(evalExp);
427  mod && mod.getKind() == AffineExprKind::Mod) {
428  // We can still infer a static bound for expressions in form
429  // "d % constant" since d % constant \in [0, constant).
430  if (auto bound = llvm::dyn_cast<AffineConstantExpr>(mod.getRHS())) {
431  ret.push_back(bound.getValue());
432  continue;
433  }
434  }
435  ret.push_back(ShapedType::kDynamic);
436  }
437  }
438  assert(ret.size() == rank);
439  return ret;
440 }

456 Attribute SparseTensorEncodingAttr::parse(AsmParser &parser, Type type) {
457  // Open "<{" part.
458  if (failed(parser.parseLess()))
459  return {};
460  if (failed(parser.parseLBrace()))
461  return {};
462 
463  // Process the data from the parsed dictionary value into struct-like data.
464  SmallVector<LevelType> lvlTypes;
465  SmallVector<SparseTensorDimSliceAttr> dimSlices;
466  AffineMap dimToLvl = {};
467  AffineMap lvlToDim = {};
468  unsigned posWidth = 0;
469  unsigned crdWidth = 0;
470  StringRef attrName;
471  SmallVector<StringRef, 3> keys = {"map", "posWidth", "crdWidth"};
472  while (succeeded(parser.parseOptionalKeyword(&attrName))) {
473  // Detect admissible keyword.
474  auto *it = find(keys, attrName);
475  if (it == keys.end()) {
476  parser.emitError(parser.getNameLoc(), "unexpected key: ") << attrName;
477  return {};
478  }
479  unsigned keyWordIndex = it - keys.begin();
480  // Consume the `=` after keys
481  if (failed(parser.parseEqual()))
482  return {};
483  // Dispatch on keyword.
484  switch (keyWordIndex) {
485  case 0: { // map
486  ir_detail::DimLvlMapParser cParser(parser);
487  auto res = cParser.parseDimLvlMap();
488  if (failed(res))
489  return {};
490  const auto &dlm = *res;
491 
492  const Level lvlRank = dlm.getLvlRank();
493  for (Level lvl = 0; lvl < lvlRank; lvl++)
494  lvlTypes.push_back(dlm.getLvlType(lvl));
495 
496  const Dimension dimRank = dlm.getDimRank();
497  for (Dimension dim = 0; dim < dimRank; dim++)
498  dimSlices.push_back(dlm.getDimSlice(dim));
499  // NOTE: the old syntax requires an all-or-nothing approach to
500  // `dimSlices`; therefore, if any slice actually exists then we need
501  // to convert null-DSA into default/nop DSA.
502  const auto isDefined = [](SparseTensorDimSliceAttr slice) {
503  return static_cast<bool>(slice.getImpl());
504  };
505  if (llvm::any_of(dimSlices, isDefined)) {
506  const auto defaultSlice =
507  SparseTensorDimSliceAttr::get(parser.getContext());
508  for (Dimension dim = 0; dim < dimRank; dim++)
509  if (!isDefined(dimSlices[dim]))
510  dimSlices[dim] = defaultSlice;
511  } else {
512  dimSlices.clear();
513  }
514 
515  dimToLvl = dlm.getDimToLvlMap(parser.getContext());
516  lvlToDim = dlm.getLvlToDimMap(parser.getContext());
517  break;
518  }
519  case 1: { // posWidth
520  Attribute attr;
521  if (failed(parser.parseAttribute(attr)))
522  return {};
523  auto intAttr = llvm::dyn_cast<IntegerAttr>(attr);
524  if (!intAttr) {
525  parser.emitError(parser.getNameLoc(),
526  "expected an integral position bitwidth");
527  return {};
528  }
529  posWidth = intAttr.getInt();
530  break;
531  }
532  case 2: { // crdWidth
533  Attribute attr;
534  if (failed(parser.parseAttribute(attr)))
535  return {};
536  auto intAttr = llvm::dyn_cast<IntegerAttr>(attr);
537  if (!intAttr) {
538  parser.emitError(parser.getNameLoc(),
539  "expected an integral index bitwidth");
540  return {};
541  }
542  crdWidth = intAttr.getInt();
543  break;
544  }
545  } // switch
546  // Only last item can omit the comma.
547  if (parser.parseOptionalComma().failed())
548  break;
549  }
550 
551  // Close "}>" part.
552  if (failed(parser.parseRBrace()))
553  return {};
554  if (failed(parser.parseGreater()))
555  return {};
556 
557  // Construct struct-like storage for attribute.
558  if (!lvlToDim || lvlToDim.isEmpty()) {
559  lvlToDim = inferLvlToDim(dimToLvl, parser.getContext());
560  }
561  return parser.getChecked<SparseTensorEncodingAttr>(
562  parser.getContext(), lvlTypes, dimToLvl, lvlToDim, posWidth, crdWidth,
563  dimSlices);
564 }

630  function_ref<InFlightDiagnostic()> emitError, ArrayRef<LevelType> lvlTypes,
631  AffineMap dimToLvl, AffineMap lvlToDim, unsigned posWidth,
632  unsigned crdWidth, ArrayRef<SparseTensorDimSliceAttr> dimSlices) {
633  if (!acceptBitWidth(posWidth))
634  return emitError() << "unexpected position bitwidth: " << posWidth;
635  if (!acceptBitWidth(crdWidth))
636  return emitError() << "unexpected coordinate bitwidth: " << crdWidth;
637  if (auto it = std::find_if(lvlTypes.begin(), lvlTypes.end(), isSingletonLT);
638  it != std::end(lvlTypes)) {
639  if (it == lvlTypes.begin() ||
640  (!isCompressedLT(*(it - 1)) && !isLooseCompressedLT(*(it - 1))))
641  return emitError() << "expected compressed or loose_compressed level "
642  "before singleton level";
643  if (!std::all_of(it, lvlTypes.end(),
644  [](LevelType i) { return isSingletonLT(i); }))
645  return emitError() << "expected all singleton lvlTypes "
646  "following a singleton level";
647  }
648  // Before we can check that the level-rank is consistent/coherent
649  // across all fields, we need to define it. The source-of-truth for
650  // the `getLvlRank` method is the length of the level-types array,
651  // since it must always be provided and have full rank; therefore we
652  // use that same source-of-truth here.
653  const Level lvlRank = lvlTypes.size();
654  if (lvlRank == 0)
655  return emitError() << "expected a non-empty array for lvlTypes";
656  // We save `dimRank` here because we'll also need it to verify `dimSlices`.
657  const Dimension dimRank = dimToLvl ? dimToLvl.getNumDims() : lvlRank;
658  if (dimToLvl) {
659  if (dimToLvl.getNumResults() != lvlRank)
660  return emitError()
661  << "level-rank mismatch between dimToLvl and lvlTypes: "
662  << dimToLvl.getNumResults() << " != " << lvlRank;
663  auto inferRes = inferLvlToDim(dimToLvl, dimToLvl.getContext());
664  // Symbols can't be inferred but are acceptable.
665  if (!inferRes && dimToLvl.getNumSymbols() == 0)
666  return emitError() << "failed to infer lvlToDim from dimToLvl";
667  if (lvlToDim && (inferRes != lvlToDim))
668  return emitError() << "expected lvlToDim to be an inverse of dimToLvl";
669  if (dimRank > lvlRank)
670  return emitError() << "unexpected dimToLvl mapping from " << dimRank
671  << " to " << lvlRank;
672  }
673  if (!dimSlices.empty()) {
674  if (dimSlices.size() != dimRank)
675  return emitError()
676  << "dimension-rank mismatch between dimSlices and dimToLvl: "
677  << dimSlices.size() << " != " << dimRank;
678  // Compiler support for `dimSlices` currently requires that the two
679  // ranks agree. (However, it does allow `dimToLvl` to be a permutation.)
680  if (dimRank != lvlRank)
681  return emitError()
682  << "dimSlices expected dimension-rank to match level-rank: "
683  << dimRank << " != " << lvlRank;
684  }
685  return success();
686 }

689  ArrayRef<Size> dimShape, Type elementType,
690  function_ref<InFlightDiagnostic()> emitError) const {
691  // Check structural integrity. In particular, this ensures that the
692  // level-rank is coherent across all the fields.
693  if (failed(verify(emitError, getLvlTypes(), getDimToLvl(), getLvlToDim(),

706  return success();

785 AffineMap mlir::sparse_tensor::inverseBlockSparsity(AffineMap dimToLvl,
786  MLIRContext *context) {
787  SmallVector<AffineExpr> lvlExprs;
788  auto numLvls = dimToLvl.getNumResults();
789  lvlExprs.reserve(numLvls);
790  // lvlExprComponents stores information of the floordiv and mod operations
791  // applied to the same dimension, so as to build the lvlToDim map.
792  std::map<unsigned, SmallVector<AffineExpr, 3>> lvlExprComponents;
793  for (unsigned i = 0, n = numLvls; i < n; i++) {
794  auto result = dimToLvl.getResult(i);
795  if (auto binOp = dyn_cast<AffineBinaryOpExpr>(result)) {
796  if (result.getKind() == AffineExprKind::FloorDiv) {
797  // Position of the dimension in dimToLvl.
798  auto pos = dyn_cast<AffineDimExpr>(binOp.getLHS()).getPosition();
799  assert(lvlExprComponents.find(pos) == lvlExprComponents.end() &&
800  "expected only one floordiv for each dimension");
801  SmallVector<AffineExpr, 3> components;
802  // Level variable for floordiv.
803  components.push_back(getAffineDimExpr(i, context));
804  // Multiplier.
805  components.push_back(binOp.getRHS());
806  // Map key is the position of the dimension.
807  lvlExprComponents[pos] = components;
808  } else if (result.getKind() == AffineExprKind::Mod) {
809  auto pos = dyn_cast<AffineDimExpr>(binOp.getLHS()).getPosition();
810  assert(lvlExprComponents.find(pos) != lvlExprComponents.end() &&
811  "expected floordiv before mod");
812  // Add level variable for mod to the same vector
813  // of the corresponding floordiv.
814  lvlExprComponents[pos].push_back(getAffineDimExpr(i, context));
815  } else {
816  assert(false && "expected floordiv or mod");
817  }
818  } else {
819  lvlExprs.push_back(getAffineDimExpr(i, context));
820  }
821  }
822  // Build lvlExprs from lvlExprComponents.
823  // For example, for il = i floordiv 2 and ii = i mod 2, the components
824  // would be [il, 2, ii]. It could be used to build the AffineExpr
825  // i = il * 2 + ii in lvlToDim.
826  for (auto &components : lvlExprComponents) {
827  assert(components.second.size() == 3 &&
828  "expected 3 components to build lvlExprs");
829  auto mulOp = getAffineBinaryOpExpr(
830  AffineExprKind::Mul, components.second[0], components.second[1]);
831  auto addOp =
832  getAffineBinaryOpExpr(AffineExprKind::Add, mulOp, components.second[2]);
833  lvlExprs.push_back(addOp);
834  }
835  return dimToLvl.get(dimToLvl.getNumResults(), 0, lvlExprs, context);
836 }

855 bool mlir::sparse_tensor::isBlockSparsity(AffineMap dimToLvl) {
856  if (!dimToLvl)
857  return false;
858  std::map<unsigned, int64_t> coeffientMap;
859  for (auto result : dimToLvl.getResults()) {
860  if (auto binOp = dyn_cast<AffineBinaryOpExpr>(result)) {
861  // Check for "dim op const".
862  auto dimOp = dyn_cast<AffineDimExpr>(binOp.getLHS());
863  auto conOp = dyn_cast<AffineConstantExpr>(binOp.getRHS());
864  if (!dimOp || !conOp)
865  return false;
866  // Inspect "dim / const" or "dim % const".
867  auto pos = dimOp.getPosition();
868  if (binOp.getKind() == AffineExprKind::FloorDiv) {
869  // Expect only one floordiv for each dimension.
870  if (coeffientMap.find(pos) != coeffientMap.end())
871  return false;
872  // Record coefficient of the floordiv.
873  coeffientMap[pos] = conOp.getValue();
874  } else if (binOp.getKind() == AffineExprKind::Mod) {
875  // Expect floordiv before mod.
876  if (coeffientMap.find(pos) == coeffientMap.end())
877  return false;
878  // Expect mod to have the same coefficient as floordiv.
879  if (conOp.getValue() != coeffientMap[pos])
880  return false;
881  } else {
882  return false;
883  }
884  }
885  }
886  return !coeffientMap.empty();
887 }

1001 static LogicalResult verifyPackUnPack(Operation *op, bool requiresStaticShape,
1002  SparseTensorType stt,
1003  RankedTensorType valTp,
1004  TypeRange lvlTps) {
1005  if (requiresStaticShape && !stt.hasStaticDimShape())
1006  return op->emitError("the sparse-tensor must have static shape");
1007  if (!stt.hasEncoding())
1008  return op->emitError("the sparse-tensor must have an encoding attribute");
1009  if (!stt.isIdentity())
1010  return op->emitError("the sparse-tensor must have the identity mapping");
1011 
1012  // Verifies the trailing COO.
1013  Level cooStartLvl = stt.getCOOStart();
1014  if (cooStartLvl < stt.getLvlRank()) {
1015  // We only supports trailing COO for now, must be the last input.
1016  auto cooTp = llvm::cast<ShapedType>(lvlTps.back());
1017  // The coordinates should be in shape of <? x rank>
1018  unsigned expCOORank = stt.getLvlRank() - cooStartLvl;
1019  if (cooTp.getRank() != 2 || expCOORank != cooTp.getShape().back()) {
1020  op->emitError("input/output trailing COO level-ranks don't match");
1021  }
1022  }
1023 
1024  // Verifies that all types match.
1025  StorageLayout layout(stt.getEncoding());
1026  if (layout.getNumDataFields() != lvlTps.size() + 1) // plus one value memref
1027  return op->emitError("inconsistent number of fields between input/output");
1028 
1029  unsigned idx = 0;
1030  bool misMatch = false;
1031  layout.foreachField([&idx, &misMatch, stt, valTp,
1032  lvlTps](FieldIndex fid, SparseTensorFieldKind fKind,
1033  Level lvl, LevelType lt) -> bool {
1034  if (fKind == SparseTensorFieldKind::StorageSpec)
1035  return true;
1036 
1037  Type inputTp = nullptr;
1038  if (fKind == SparseTensorFieldKind::ValMemRef) {
1039  inputTp = valTp;
1040  } else {
1041  assert(fid == idx && stt.getLvlType(lvl) == lt);
1042  inputTp = lvlTps[idx++];
1043  }
1044  // The input element type and expected element type should match.
1045  Type inpElemTp = llvm::cast<TensorType>(inputTp).getElementType();
1046  Type expElemTp = getFieldElemType(stt, fKind);
1047  if (inpElemTp != expElemTp) {
1048  misMatch = true;
1049  return false; // to terminate the iteration
1050  }
1051  return true;
1052  });
1053 
1054  if (misMatch)
1055  return op->emitError("input/output element-types don't match");
1056  return success();
1057 }

1495  const Dimension concatDim = getDimension();

1497

1507  const auto i = it.index();

1538  for (const auto src : getInputs()) {
