49 /// This function may be called to register all MLIR dialect extensions with the
50 /// provided registry.
51 /// If you're building a compiler, you generally shouldn't use this: you would
52 /// individually register the specific extensions that are useful for the
53 /// pipelines and transformations you are using.
54 inline void registerAllExtensions(DialectRegistry &registry) {
55  // Register all conversions to LLVM extensions.
56  arith::registerConvertArithToLLVMInterface(registry);
57  registerConvertComplexToLLVMInterface(registry);
58  cf::registerConvertControlFlowToLLVMInterface(registry);
59  func::registerAllExtensions(registry);
60  registerConvertFuncToLLVMInterface(registry);
61  index::registerConvertIndexToLLVMInterface(registry);
62  registerConvertMathToLLVMInterface(registry);
63  registerConvertMemRefToLLVMInterface(registry);
64  registerConvertNVVMToLLVMInterface(registry);
65  ub::registerConvertUBToLLVMInterface(registry);
66 
67  // Register all transform dialect extensions.
68  affine::registerTransformDialectExtension(registry);
69  bufferization::registerTransformDialectExtension(registry);
70  func::registerTransformDialectExtension(registry);
71  gpu::registerTransformDialectExtension(registry);
72  linalg::registerTransformDialectExtension(registry);
73  memref::registerTransformDialectExtension(registry);
74  nvgpu::registerTransformDialectExtension(registry);
75  scf::registerTransformDialectExtension(registry);
76  sparse_tensor::registerTransformDialectExtension(registry);
77  tensor::registerTransformDialectExtension(registry);
78  transform::registerLoopExtension(registry);
79  transform::registerPDLExtension(registry);
80  vector::registerTransformDialectExtension(registry);
81 
82  // Translation extensions need to be registered by calling
83  // `registerAllToLLVMIRTranslations` (see All.h).
84 }
