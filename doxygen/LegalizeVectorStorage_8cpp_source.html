24 // A tag to mark unrealized_conversions produced by this pass. This is used to
25 // detect IR this pass failed to completely legalize, and report an error.
26 // If everything was successfully legalized, no tagged ops will remain after
27 // this pass.
28 constexpr StringLiteral kSVELegalizerTag("__arm_sve_legalize_vector_storage__");
29 
30 /// Definitions:
31 ///
32 /// [1] svbool = vector<...x[16]xi1>, which maps to some multiple of full SVE
33 /// predicate registers. A full predicate is the smallest quantity that can be
34 /// loaded/stored.
35 ///
36 /// [2] SVE mask = hardware-sized SVE predicate mask, i.e. its trailing
37 /// dimension matches the size of a legal SVE vector size (such as
38 /// vector<[4]xi1>), but is too small to be stored to memory (i.e smaller than
39 /// a svbool).

43 /// Checks if a vector type is a SVE mask [2].
44 bool isSVEMaskType(VectorType type) {
45  return type.getRank() > 0 && type.getElementType().isInteger(1) &&
46  type.getScalableDims().back() && type.getShape().back() < 16 &&
47  llvm::isPowerOf2_32(type.getShape().back()) &&
48  !llvm::is_contained(type.getScalableDims().drop_back(), true);
49 }
50 
51 VectorType widenScalableMaskTypeToSvbool(VectorType type) {
52  assert(isSVEMaskType(type));
53  return VectorType::Builder(type).setDim(type.getRank() - 1, 16);
54 }

56 /// A helper for cloning an op and replacing it will a new version, updated by a
57 /// callback.
58 template <typename TOp, typename TLegalizerCallback>
59 void replaceOpWithLegalizedOp(PatternRewriter &rewriter, TOp op,
60  TLegalizerCallback callback) {
61  // Clone the previous op to preserve any properties/attributes.
62  auto newOp = op.clone();
63  rewriter.insert(newOp);
64  rewriter.replaceOp(op, callback(newOp));
65 }

67 /// A helper for cloning an op and replacing it with a new version, updated by a
68 /// callback, and an unrealized conversion back to the type of the replaced op.
69 template <typename TOp, typename TLegalizerCallback>
70 void replaceOpWithUnrealizedConversion(PatternRewriter &rewriter, TOp op,
71  TLegalizerCallback callback) {
72  replaceOpWithLegalizedOp(rewriter, op, [&](TOp newOp) {
73  // Mark our `unrealized_conversion_casts` with a pass label.
74  return rewriter.create<UnrealizedConversionCastOp>(
75  op.getLoc(), TypeRange{op.getResult().getType()},
76  ValueRange{callback(newOp)},
77  NamedAttribute(rewriter.getStringAttr(kSVELegalizerTag),
78  rewriter.getUnitAttr()));
79  });
80 }

82 /// Extracts the widened SVE memref value (that's legal to store/load) from the
83 /// `unrealized_conversion_cast`s added by this pass.
84 static FailureOr<Value> getSVELegalizedMemref(Value illegalMemref) {
85  Operation *definingOp = illegalMemref.getDefiningOp();
86  if (!definingOp || !definingOp->hasAttr(kSVELegalizerTag))
87  return failure();
88  auto unrealizedConversion =
89  llvm::cast<UnrealizedConversionCastOp>(definingOp);
90  return unrealizedConversion.getOperand(0);
91 }

93 /// The default alignment of an alloca in LLVM may request overaligned sizes for
94 /// SVE types, which will fail during stack frame allocation. This rewrite
95 /// explicitly adds a reasonable alignment to allocas of scalable types.
96 struct RelaxScalableVectorAllocaAlignment
97  : public OpRewritePattern<memref::AllocaOp> {
98  using OpRewritePattern::OpRewritePattern;
99 
100  LogicalResult matchAndRewrite(memref::AllocaOp allocaOp,
101  PatternRewriter &rewriter) const override {
102  auto memrefElementType = allocaOp.getType().getElementType();
103  auto vectorType = llvm::dyn_cast<VectorType>(memrefElementType);
104  if (!vectorType || !vectorType.isScalable() || allocaOp.getAlignment())
105  return failure();
106 
107  // Set alignment based on the defaults for SVE vectors and predicates.
108  unsigned aligment = vectorType.getElementType().isInteger(1) ? 2 : 16;
109  allocaOp.setAlignment(aligment);
110 
111  return success();
112  }
113 };

115 /// Replaces allocations of SVE predicates smaller than an svbool [1] (_illegal_
116 /// to load/store) with a wider allocation of svbool (_legal_ to load/store)
117 /// followed by a tagged unrealized conversion to the original type.
118 ///
119 /// Example
120 /// ```
121 /// %alloca = memref.alloca() : memref<vector<[4]xi1>>
122 /// ```
123 /// is rewritten into:
124 /// ```
125 /// %widened = memref.alloca() {alignment = 1 : i64} : memref<vector<[16]xi1>>
126 /// %alloca = builtin.unrealized_conversion_cast %widened
127 /// : memref<vector<[16]xi1>> to memref<vector<[4]xi1>>
128 /// {__arm_sve_legalize_vector_storage__}
129 /// ```
130 template <typename AllocLikeOp>
131 struct LegalizeSVEMaskAllocation : public OpRewritePattern<AllocLikeOp> {
132  using OpRewritePattern<AllocLikeOp>::OpRewritePattern;
133 
134  LogicalResult matchAndRewrite(AllocLikeOp allocLikeOp,
135  PatternRewriter &rewriter) const override {
136  auto vectorType =
137  llvm::dyn_cast<VectorType>(allocLikeOp.getType().getElementType());
138 
139  if (!vectorType || !isSVEMaskType(vectorType))
140  return failure();
141 
142  // Replace this alloc-like op of an SVE mask [2] with one of a (storable)
143  // svbool mask [1]. A temporary unrealized_conversion_cast is added to the
144  // old type to allow local rewrites.
145  replaceOpWithUnrealizedConversion(
146  rewriter, allocLikeOp, [&](AllocLikeOp newAllocLikeOp) {
147  newAllocLikeOp.getResult().setType(
148  llvm::cast<MemRefType>(newAllocLikeOp.getType().cloneWith(
149  {}, widenScalableMaskTypeToSvbool(vectorType))));
150  return newAllocLikeOp;
151  });
152 
153  return success();
154  }
155 };

157 /// Replaces vector.type_casts of unrealized conversions to SVE predicate memref
158 /// types that are _illegal_ to load/store from (!= svbool [1]), with type casts
159 /// of memref types that are _legal_ to load/store, followed by unrealized
160 /// conversions.
161 ///
162 /// Example:
163 /// ```
164 /// %alloca = builtin.unrealized_conversion_cast %widened
165 /// : memref<vector<[16]xi1>> to memref<vector<[8]xi1>>
166 /// {__arm_sve_legalize_vector_storage__}
167 /// %cast = vector.type_cast %alloca
168 /// : memref<vector<3x[8]xi1>> to memref<3xvector<[8]xi1>>
169 /// ```
170 /// is rewritten into:
171 /// ```
172 /// %widened_cast = vector.type_cast %widened
173 /// : memref<vector<3x[16]xi1>> to memref<3xvector<[16]xi1>>
174 /// %cast = builtin.unrealized_conversion_cast %widened_cast
175 /// : memref<3xvector<[16]xi1>> to memref<3xvector<[8]xi1>>
176 /// {__arm_sve_legalize_vector_storage__}
177 /// ```
178 struct LegalizeSVEMaskTypeCastConversion
179  : public OpRewritePattern<vector::TypeCastOp> {
180  using OpRewritePattern::OpRewritePattern;
181 
182  LogicalResult matchAndRewrite(vector::TypeCastOp typeCastOp,
183  PatternRewriter &rewriter) const override {
184  auto resultType = typeCastOp.getResultMemRefType();
185  auto vectorType = llvm::dyn_cast<VectorType>(resultType.getElementType());
186 
187  if (!vectorType || !isSVEMaskType(vectorType))
188  return failure();
189 
190  auto legalMemref = getSVELegalizedMemref(typeCastOp.getMemref());
191  if (failed(legalMemref))
192  return failure();
193 
194  // Replace this vector.type_cast with one of a (storable) svbool mask [1].
195  replaceOpWithUnrealizedConversion(
196  rewriter, typeCastOp, [&](vector::TypeCastOp newTypeCast) {
197  newTypeCast.setOperand(*legalMemref);
198  newTypeCast.getResult().setType(
199  llvm::cast<MemRefType>(newTypeCast.getType().cloneWith(
200  {}, widenScalableMaskTypeToSvbool(vectorType))));
201  return newTypeCast;
202  });
203 
204  return success();
205  }
206 };

208 /// Replaces stores to unrealized conversions to SVE predicate memref types that
209 /// are _illegal_ to load/store from (!= svbool [1]), with
210 /// `arm_sve.convert_to_svbool`s followed by (legal) wider stores.
211 ///
212 /// Example:
213 /// ```
214 /// memref.store %mask, %alloca[] : memref<vector<[8]xi1>>
215 /// ```
216 /// is rewritten into:
217 /// ```
218 /// %svbool = arm_sve.convert_to_svbool %mask : vector<[8]xi1>
219 /// memref.store %svbool, %widened[] : memref<vector<[16]xi1>>
220 /// ```
221 struct LegalizeSVEMaskStoreConversion
222  : public OpRewritePattern<memref::StoreOp> {
223  using OpRewritePattern::OpRewritePattern;
224 
225  LogicalResult matchAndRewrite(memref::StoreOp storeOp,
226  PatternRewriter &rewriter) const override {
227  auto loc = storeOp.getLoc();
228 
229  Value valueToStore = storeOp.getValueToStore();
230  auto vectorType = llvm::dyn_cast<VectorType>(valueToStore.getType());
231 
232  if (!vectorType || !isSVEMaskType(vectorType))
233  return failure();
234 
235  auto legalMemref = getSVELegalizedMemref(storeOp.getMemref());
236  if (failed(legalMemref))
237  return failure();
238 
239  auto legalMaskType = widenScalableMaskTypeToSvbool(
240  llvm::cast<VectorType>(valueToStore.getType()));
241  auto convertToSvbool = rewriter.create<arm_sve::ConvertToSvboolOp>(
242  loc, legalMaskType, valueToStore);
243  // Replace this store with a conversion to a storable svbool mask [1],
244  // followed by a wider store.
245  replaceOpWithLegalizedOp(rewriter, storeOp,
246  [&](memref::StoreOp newStoreOp) {
247  newStoreOp.setOperand(0, convertToSvbool);
248  newStoreOp.setOperand(1, *legalMemref);
249  return newStoreOp;
250  });
251 
252  return success();
253  }
254 };

256 /// Replaces loads from unrealized conversions to SVE predicate memref types
257 /// that are _illegal_ to load/store from (!= svbool [1]), types with (legal)
258 /// wider loads, followed by `arm_sve.convert_from_svbool`s.
259 ///
260 /// Example:
261 /// ```
262 /// %reload = memref.load %alloca[] : memref<vector<[4]xi1>>
263 /// ```
264 /// is rewritten into:
265 /// ```
266 /// %svbool = memref.load %widened[] : memref<vector<[16]xi1>>
267 /// %reload = arm_sve.convert_from_svbool %reload : vector<[4]xi1>
268 /// ```
269 struct LegalizeSVEMaskLoadConversion : public OpRewritePattern<memref::LoadOp> {
270  using OpRewritePattern::OpRewritePattern;
271 
272  LogicalResult matchAndRewrite(memref::LoadOp loadOp,
273  PatternRewriter &rewriter) const override {
274  auto loc = loadOp.getLoc();
275 
276  Value loadedMask = loadOp.getResult();
277  auto vectorType = llvm::dyn_cast<VectorType>(loadedMask.getType());
278 
279  if (!vectorType || !isSVEMaskType(vectorType))
280  return failure();
281 
282  auto legalMemref = getSVELegalizedMemref(loadOp.getMemref());
283  if (failed(legalMemref))
284  return failure();
285 
286  auto legalMaskType = widenScalableMaskTypeToSvbool(vectorType);
287  // Replace this load with a legal load of an svbool type, followed by a
288  // conversion back to the original type.
289  replaceOpWithLegalizedOp(rewriter, loadOp, [&](memref::LoadOp newLoadOp) {
290  newLoadOp.setMemRef(*legalMemref);
291  newLoadOp.getResult().setType(legalMaskType);

293  loc, loadedMask.getType(), newLoadOp);
294  });
295 
296  return success();
297  }

312 namespace {
313 struct LegalizeVectorStorage
314  : public arm_sve::impl::LegalizeVectorStorageBase<LegalizeVectorStorage> {
315 
316  void runOnOperation() override {
317  RewritePatternSet patterns(&getContext());
318  populateLegalizeVectorStoragePatterns(patterns);
319  if (failed(applyPatternsAndFoldGreedily(getOperation(),
320  std::move(patterns)))) {
321  signalPassFailure();
322  }
323  ConversionTarget target(getContext());
324  target.addDynamicallyLegalOp<UnrealizedConversionCastOp>(
325  [](UnrealizedConversionCastOp unrealizedConversion) {
326  return !unrealizedConversion->hasAttr(kSVELegalizerTag);
327  });
328  // This detects if we failed to completely legalize the IR.
329  if (failed(applyPartialConversion(getOperation(), target, {})))
330  signalPassFailure();
331  }
332 };
