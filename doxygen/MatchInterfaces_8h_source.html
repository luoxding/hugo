23 /// Dispatch `matchOperation` based on Operation* or std::optional<Operation*>
24 /// first operand.
25 template <typename OpTy>
26 DiagnosedSilenceableFailure matchOptionalOperation(OpTy op,
27  TransformResults &results,
28  TransformState &state) {
29  if constexpr (std::is_same_v<
30  typename llvm::function_traits<
31  decltype(&OpTy::matchOperation)>::template arg_t<0>,
32  Operation *>) {
33  return op.matchOperation(nullptr, results, state);
34  } else {
35  return op.matchOperation(std::nullopt, results, state);
36  }
37 }
38 } // namespace detail

40 template <typename OpTy>
41 class AtMostOneOpMatcherOpTrait
42  : public OpTrait::TraitBase<OpTy, AtMostOneOpMatcherOpTrait> {
43  template <typename T>
44  using has_get_operand_handle =
45  decltype(std::declval<T &>().getOperandHandle());
46  template <typename T>
47  using has_match_operation_ptr = decltype(std::declval<T &>().matchOperation(
48  std::declval<Operation *>(), std::declval<TransformResults &>(),
49  std::declval<TransformState &>()));
50  template <typename T>
51  using has_match_operation_optional =
52  decltype(std::declval<T &>().matchOperation(
53  std::declval<std::optional<Operation *>>(),
54  std::declval<TransformResults &>(),
55  std::declval<TransformState &>()));
56 
57 public:
58  static LogicalResult verifyTrait(Operation *op) {
59  static_assert(llvm::is_detected<has_get_operand_handle, OpTy>::value,
60  "AtMostOneOpMatcherOpTrait/SingleOpMatcherOpTrait expects "
61  "operation type to have the getOperandHandle() method");
62  static_assert(
63  llvm::is_detected<has_match_operation_ptr, OpTy>::value ||
64  llvm::is_detected<has_match_operation_optional, OpTy>::value,
65  "AtMostOneOpMatcherOpTrait/SingleOpMatcherOpTrait expected operation "
66  "type to have either the matchOperation(Operation *, TransformResults "
67  "&, TransformState &) or the matchOperation(std::optional<Operation*>, "
68  "TransformResults &, TransformState &) method");
69 
70  // This must be a dynamic assert because interface registration is dynamic.
71  assert(
72  isa<MatchOpInterface>(op) &&
73  "AtMostOneOpMatcherOpTrait/SingleOpMatchOpTrait is only available on "
74  "operations with MatchOpInterface");
75  Value operandHandle = cast<OpTy>(op).getOperandHandle();
76  if (!isa<TransformHandleTypeInterface>(operandHandle.getType())) {
77  return op->emitError() << "AtMostOneOpMatcherOpTrait/"
78  "SingleOpMatchOpTrait requires the op handle "
79  "to be of TransformHandleTypeInterface";
80  }
81 
82  return success();
83  }
84 
85  DiagnosedSilenceableFailure apply(TransformRewriter &rewriter,
86  TransformResults &results,
87  TransformState &state) {
88  Value operandHandle = cast<OpTy>(this->getOperation()).getOperandHandle();
89  auto payload = state.getPayloadOps(operandHandle);
90  if (!llvm::hasNItemsOrLess(payload, 1)) {
91  return emitDefiniteFailure(this->getOperation()->getLoc())
92  << "AtMostOneOpMatcherOpTrait requires the operand handle to "
93  "point to at most one payload op";
94  }
95  if (payload.empty()) {
96  return detail::matchOptionalOperation(cast<OpTy>(this->getOperation()),
97  results, state);
98  }
99  return cast<OpTy>(this->getOperation())
100  .matchOperation(*payload.begin(), results, state);
101  }
102 
103  void getEffects(SmallVectorImpl<MemoryEffects::EffectInstance> &effects) {
104  onlyReadsHandle(this->getOperation()->getOperands(), effects);
105  producesHandle(this->getOperation()->getResults(), effects);
106  onlyReadsPayload(effects);
107  }
108 };

110 template <typename OpTy>
111 class SingleOpMatcherOpTrait : public AtMostOneOpMatcherOpTrait<OpTy> {
112 
113 public:
114  DiagnosedSilenceableFailure apply(TransformRewriter &rewriter,
115  TransformResults &results,
116  TransformState &state) {
117  Value operandHandle = cast<OpTy>(this->getOperation()).getOperandHandle();
118  auto payload = state.getPayloadOps(operandHandle);
119  if (!llvm::hasSingleElement(payload)) {
120  return emitDefiniteFailure(this->getOperation()->getLoc())
121  << "SingleOpMatchOpTrait requires the operand handle to point to "
122  "a single payload op";
123  }
124  return static_cast<AtMostOneOpMatcherOpTrait<OpTy> *>(this)->apply(
125  rewriter, results, state);
126  }
