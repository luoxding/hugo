19 /// Returns true if 'val' is a splat of zero, false otherwise.
20 static bool isSplatZero(Type elemType, DenseElementsAttr val) {
21  if (llvm::isa<FloatType>(elemType))
22  return val && val.isSplat() && val.getSplatValue<APFloat>().isZero();
23  if (llvm::isa<IntegerType>(elemType))
24  return val && val.isSplat() && val.getSplatValue<APInt>().isZero();
25  return false;
26 }

28 /// Generates a for loop over ZA tile slices where the induction variable is
29 /// the tile slice index and each iteration yields a new tile. Loop body is
30 /// built via the callback, which returns the next tile value.
31 template <typename LoopBodyCallback>
32 static scf::ForOp createLoopOverTileSlices(PatternRewriter &rewriter,
33  Location loc, Value initTile,
34  LoopBodyCallback callback) {
35  OpBuilder::InsertionGuard g(rewriter);
36  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
37  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
38  loc, llvm::cast<VectorType>(initTile.getType()).getDimSize(0));
39  auto vscale =
40  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
41  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
42  auto numTileSlices =
43  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
44  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, numTileSlices, step,
45  ValueRange{initTile});
46  rewriter.setInsertionPointToStart(forOp.getBody());
47  auto nextTile = callback(forOp);
48  rewriter.create<scf::YieldOp>(loc, nextTile.getResult());
49  return forOp;
50 }

54 /// Conversion pattern for vector.transfer_read.
55 ///
56 /// ---
57 ///
58 /// Example 1: op with identity permutation map to horizontal
59 /// arm_sme.tile_load:
60 ///
61 /// vector.transfer_read ... permutation_map: (d0, d1) -> (d0, d1)
62 ///
63 /// is converted to:
64 ///
65 /// arm_sme.tile_load ...
66 ///
67 /// ---
68 ///
69 /// Example 2: op with transpose permutation map to vertical arm_sme.tile_load
70 /// (in-flight transpose):
71 ///
72 /// vector.transfer_read ... permutation_map: (d0, d1) -> (d1, d0)
73 ///
74 /// is converted to:
75 ///
76 /// arm_sme.tile_load ... layout<vertical>
77 struct TransferReadToArmSMELowering
78  : public OpRewritePattern<vector::TransferReadOp> {
79  using OpRewritePattern<vector::TransferReadOp>::OpRewritePattern;
80 
81  LogicalResult matchAndRewrite(vector::TransferReadOp transferReadOp,
82  PatternRewriter &rewriter) const final {
83  // The permutation map must have two results.
84  if (transferReadOp.getTransferRank() != 2)
85  return rewriter.notifyMatchFailure(transferReadOp,
86  "not a 2 result permutation map");
87 
88  auto vectorType = transferReadOp.getVectorType();
89  if (!arm_sme::isValidSMETileVectorType(vectorType))
90  return rewriter.notifyMatchFailure(transferReadOp,
91  "not a valid vector type for SME");
92 
93  if (!llvm::isa<MemRefType>(transferReadOp.getSource().getType()))
94  return rewriter.notifyMatchFailure(transferReadOp, "not a memref source");
95 
96  // Out-of-bounds dims are not supported.
97  if (transferReadOp.hasOutOfBoundsDim())
98  return rewriter.notifyMatchFailure(transferReadOp,
99  "not inbounds transfer read");
100 
101  arm_sme::TileSliceLayout layout;
102 
103  AffineExpr d0, d1;
104  bindDims(transferReadOp.getContext(), d0, d1);
105  AffineMap map = transferReadOp.getPermutationMap();
106  if (map.isIdentity())
107  layout = arm_sme::TileSliceLayout::Horizontal;
108  else if (map == AffineMap::get(map.getNumDims(), 0, {d1, d0},
109  transferReadOp.getContext()))
110  layout = arm_sme::TileSliceLayout::Vertical;
111  else
112  return rewriter.notifyMatchFailure(transferReadOp,
113  "unsupported permutation map");
114 
115  // Padding isn't optional for transfer_read, but is only used in the case
116  // of out-of-bounds accesses (not supported here) and/or masking. Mask is
117  // optional, if it's not present don't pass padding.
118  auto mask = transferReadOp.getMask();
119  auto padding = mask ? transferReadOp.getPadding() : nullptr;
120  rewriter.replaceOpWithNewOp<arm_sme::TileLoadOp>(
121  transferReadOp, vectorType, transferReadOp.getSource(),
122  transferReadOp.getIndices(), padding, mask, layout);
123 
124  return success();
125  }
126 };

128 /// Conversion pattern for vector.transfer_write.
129 ///
130 /// ---
131 ///
132 /// Example 1: op with identity permutation map to horizontal
133 /// arm_sme.tile_store:
134 ///
135 /// vector.transfer_write %vector, %source[%c0, %c0]
136 /// {in_bounds = [true, true]} : vector<[16]x[16]xi8>, memref<?x?xi8>
137 ///
138 /// is converted to:
139 ///
140 /// arm_sme.tile_store %vector, %source[%c0, %c0] : memref<?x?xi8>,
141 /// vector<[16]x[16]xi8>
142 /// ---
143 ///
144 /// Example 2: op with transpose permutation map to vertical arm_sme.tile_store
145 /// (in-flight transpose):
146 ///
147 /// vector.transfer_write %vector, %source[%c0, %c0]
148 /// {permutation_map = affine_map<(d0, d1) -> (d1, d0)>,
149 /// in_bounds = [true, true]} : vector<[16]x[16]xi8>, memref<?x?xi8>
150 ///
151 /// is converted to:
152 ///
153 /// arm_sme.tile_store %vector, %source[%c0, %c0] layout<vertical>
154 /// : memref<?x?xi8>, vector<[16]x[16]xi8>
155 struct TransferWriteToArmSMELowering
156  : public OpRewritePattern<vector::TransferWriteOp> {
157  using OpRewritePattern<vector::TransferWriteOp>::OpRewritePattern;
158 
159  LogicalResult matchAndRewrite(vector::TransferWriteOp writeOp,
160  PatternRewriter &rewriter) const final {
161  auto vType = writeOp.getVectorType();
162  if (!arm_sme::isValidSMETileVectorType(vType))
163  return failure();
164 
165  if (!llvm::isa<MemRefType>(writeOp.getSource().getType()))
166  return failure();
167 
168  // Out-of-bounds dims are not supported.
169  if (writeOp.hasOutOfBoundsDim())
170  return rewriter.notifyMatchFailure(writeOp,
171  "not inbounds transfer write");
172 
173  AffineExpr d0, d1;
174  bindDims(writeOp.getContext(), d0, d1);
175  AffineMap map = writeOp.getPermutationMap();
176  bool isTranspose = (map == AffineMap::get(map.getNumDims(), 0, {d1, d0},
177  writeOp.getContext()));
178 
179  if (!map.isIdentity() && !isTranspose)
180  return rewriter.notifyMatchFailure(writeOp,
181  "unsupported permutation map");
182 
183  arm_sme::TileSliceLayout layout =
184  isTranspose ? arm_sme::TileSliceLayout::Vertical
185  : arm_sme::TileSliceLayout::Horizontal;
186 
187  rewriter.replaceOpWithNewOp<arm_sme::TileStoreOp>(
188  writeOp, writeOp.getVector(), writeOp.getSource(), writeOp.getIndices(),
189  writeOp.getMask(), layout);
190  return success();
191  }
192 };

226 /// Conversion pattern for dense arith.constant.
227 struct ConstantOpToArmSMELowering : public OpRewritePattern<arith::ConstantOp> {
228  using OpRewritePattern<arith::ConstantOp>::OpRewritePattern;
229 
230  LogicalResult matchAndRewrite(arith::ConstantOp constantOp,
231  PatternRewriter &rewriter) const final {
232  auto tileType = dyn_cast<VectorType>(constantOp.getType());
233  if (!tileType || !arm_sme::isValidSMETileVectorType(tileType))
234  return failure();
235 
236  auto denseAttr = dyn_cast<DenseElementsAttr>(constantOp.getValueAttr());
237  if (!denseAttr || !denseAttr.isSplat())
238  return failure();
239 
240  auto tileElementType = tileType.getElementType();
241 
242  // Lower 'arith.constant dense<0>' to 'arm_sme.zero' op.
243  if (isSplatZero(tileElementType, denseAttr)) {
244  rewriter.replaceOpWithNewOp<arm_sme::ZeroOp>(constantOp, tileType);
245  return success();
246  }
247 
248  // Lower non-zero constants to a loop of 'arm_sme.move_vector_to_tile_slice'
249  // ops that broadcast the constant to each tile slice.
250  auto loc = constantOp.getLoc();
251 
252  // To fill a tile with a constant, we create a 1-D splat of the constant,
253  // then move that into each tile slice (the largest unit we can set at once,
254  // outside of operations like the outerproduct).
255  VectorType tileSliceType = VectorType::Builder(tileType).dropDim(0);
256  auto denseAttr1D = DenseElementsAttr::get(
257  tileSliceType, denseAttr.getSplatValue<Attribute>());
258  auto constantOp1D = rewriter.create<arith::ConstantOp>(loc, denseAttr1D);
259 
260  auto initTile = rewriter.create<arm_sme::GetTileOp>(loc, tileType);
261  auto forOp =
262  createLoopOverTileSlices(rewriter, loc, initTile, [&](auto forOp) {
263  auto tileSliceIndex = forOp.getInductionVar();
264  auto currentTile = forOp.getRegionIterArg(0);
265  // Create 'arm_sme.move_vector_to_tile_slice' to write vector to tile
266  // slice.
267  return rewriter.create<arm_sme::MoveVectorToTileSliceOp>(
268  loc, tileType, constantOp1D, currentTile, tileSliceIndex);
269  });
270  rewriter.replaceOp(constantOp, forOp.getResult(0));
271 
272  return success();
273  }
274 };

276 /// Conversion pattern for vector.broadcast.
277 ///
278 /// Example:
279 ///
280 /// %broadcast_to_tile = vector.broadcast %src : i32 to vector<[4]x[4]xi32>
281 ///
282 /// is converted to:
283 ///
284 /// %broadcast_to_1d = vector.broadcast %src : i32 to vector<[4]xi32>
285 /// %broadcast_to_tile = scf.for %tile_slice_index = %c0 to %num_tile_slices
286 /// step %c1 iter_args(%iter_tile = %init_tile) -> (vector<[4]x[4]xi32>)
287 /// {
288 /// %tile_update = arm_sme.move_vector_to_tile_slice
289 /// %broadcast_to_1d, %iter_tile, %tile_slice_index :
290 /// vector<[4]xi32> into vector<[4]x[4]xi32>
291 /// scf.yield %tile_update : vector<[4]x[4]xi32>
292 /// }
293 ///
294 /// Supports scalar, 0-d vector, and 1-d vector broadcasts.
295 struct BroadcastOpToArmSMELowering
296  : public OpRewritePattern<vector::BroadcastOp> {
297  using OpRewritePattern<vector::BroadcastOp>::OpRewritePattern;
298 
299  LogicalResult matchAndRewrite(vector::BroadcastOp broadcastOp,
300  PatternRewriter &rewriter) const final {
301  auto tileType = broadcastOp.getResultVectorType();
302  if (!tileType || !arm_sme::isValidSMETileVectorType(tileType))
303  return failure();
304 
305  auto loc = broadcastOp.getLoc();
306 
307  auto srcType = broadcastOp.getSourceType();
308  auto srcVectorType = dyn_cast<VectorType>(srcType);
309 
310  Value broadcastOp1D;
311  if (srcType.isIntOrFloat() ||
312  (srcVectorType && (srcVectorType.getRank() == 0))) {
313  // Broadcast scalar or 0-d vector to 1-d vector.
314  VectorType tileSliceType = VectorType::Builder(tileType).dropDim(0);
315  broadcastOp1D = rewriter.create<vector::BroadcastOp>(
316  loc, tileSliceType, broadcastOp.getSource());
317  } else if (srcVectorType && (srcVectorType.getRank() == 1))
318  // Value to broadcast is already a 1-d vector, nothing to do.
319  broadcastOp1D = broadcastOp.getSource();
320  else
321  return failure();
322 
323  auto initTile = rewriter.create<arm_sme::GetTileOp>(loc, tileType);
324 
325  // Create a loop over ZA tile slices.
326  auto forOp =
327  createLoopOverTileSlices(rewriter, loc, initTile, [&](auto forOp) {
328  auto tileSliceIndex = forOp.getInductionVar();
329  auto currentTile = forOp.getRegionIterArg(0);
330  // Create 'arm_sme.move_vector_to_tile_slice' to broadcast the value
331  // to each tile slice.
332  return rewriter.create<arm_sme::MoveVectorToTileSliceOp>(
333  loc, tileType, broadcastOp1D, currentTile, tileSliceIndex);
334  });
335 
336  rewriter.replaceOp(broadcastOp, forOp.getResult(0));
337 
338  return success();
339  }
340 };

342 /// Conversion pattern for vector.splat.
343 ///
344 /// Example:
345 ///
346 /// %splat_to_tile = vector.splat %src : i32 to vector<[4]x[4]xi32>
347 ///
348 /// is converted to:
349 ///
350 /// %broadcast_to_1d = vector.broadcast %src : i32 to vector<[4]xi32>
351 /// %broadcast_to_tile = scf.for %tile_slice_index = %c0 to %num_tile_slices
352 /// step %c1 iter_args(%iter_tile = %init_tile) -> (vector<[4]x[4]xi32>)
353 /// {
354 /// %tile_update = arm_sme.move_vector_to_tile_slice
355 /// %broadcast_to_1d, %iter_tile, %tile_slice_index :
356 /// vector<[4]xi32> into vector<[4]x[4]xi32>
357 /// scf.yield %tile_update : vector<[4]x[4]xi32>
358 /// }
359 ///
360 /// This is identical to vector.broadcast of a scalar.
361 struct SplatOpToArmSMELowering : public OpRewritePattern<vector::SplatOp> {
362  using OpRewritePattern<vector::SplatOp>::OpRewritePattern;
363 
364  LogicalResult matchAndRewrite(vector::SplatOp splatOp,
365  PatternRewriter &rewriter) const final {
366  auto tileType = splatOp.getResult().getType();
367  if (!tileType || !arm_sme::isValidSMETileVectorType(tileType))
368  return failure();
369 
370  auto loc = splatOp.getLoc();
371  auto srcType = splatOp.getOperand().getType();
372 
373  assert(srcType.isIntOrFloat() && "Invalid source type for vector.splat");
374  // Avoid unused-variable warning when building without assertions.
375  (void)srcType;
376 
377  // First, broadcast the scalar to a 1-d vector.
378  VectorType tileSliceType = VectorType::Builder(tileType).dropDim(0);
379  Value broadcastOp1D = rewriter.create<vector::BroadcastOp>(
380  loc, tileSliceType, splatOp.getInput());
381 
382  auto initTile = rewriter.create<arm_sme::GetTileOp>(loc, tileType);
383 
384  // Next, create a loop over ZA tile slices and "move" the generated 1-d
385  // vector to each slice.
386  auto forOp =
387  createLoopOverTileSlices(rewriter, loc, initTile, [&](auto forOp) {
388  auto tileSliceIndex = forOp.getInductionVar();
389  auto currentTile = forOp.getRegionIterArg(0);
390  return rewriter.create<arm_sme::MoveVectorToTileSliceOp>(
391  loc, tileType, broadcastOp1D, currentTile, tileSliceIndex);
392  });
393 
394  rewriter.replaceOp(splatOp, forOp.getResult(0));
395 
396  return success();
397  }
398 };

400 /// Conversion pattern for vector.transpose.
401 ///
402 /// Stores the input tile to memory and reloads vertically.
403 ///
404 /// Example:
405 ///
406 /// %transposed_src = vector.transpose %src, [1, 0]
407 /// : vector<[4]x[4]xi32> to vector<[4]x[4]xi32>
408 ///
409 /// is converted to:
410 ///
411 /// %alloca = memref.alloca(%svl_s, %svl_s) : memref<?x?xi32>
412 /// %arm_sme.tile_store %src, <hor>, %alloca[%c0, %c0]
413 /// : memref<?x?xi32>, vector<[4]x[4]xi32>
414 /// %transposed_src = arm_sme.tile_load %alloca[%c0, %c0]
415 /// layout<vertical> : memref<?x?xi32>, vector<[4]x[4]xi32>
416 ///
417 /// NOTE: Tranposing via memory is obviously expensive, the current intention
418 /// is to avoid the transpose if possible, this is therefore intended as a
419 /// fallback and to provide base support for Vector ops. If it turns out
420 /// transposes can't be avoided then this should be replaced with a more optimal
421 /// implementation, perhaps with tile <-> vector (MOVA) ops.
422 struct TransposeOpToArmSMELowering
423  : public OpRewritePattern<vector::TransposeOp> {
424  using OpRewritePattern<vector::TransposeOp>::OpRewritePattern;
425 
426  LogicalResult matchAndRewrite(vector::TransposeOp transposeOp,
427  PatternRewriter &rewriter) const final {
428  auto tileType = transposeOp.getResultVectorType();
429  if (!tileType || !arm_sme::isValidSMETileVectorType(tileType))
430  return failure();
431 
432  // Bail unless this is a true 2-D matrix transpose.
433  ArrayRef<int64_t> permutation = transposeOp.getPermutation();
434  if (permutation[0] != 1 || permutation[1] != 0)
435  return failure();
436 
437  auto loc = transposeOp.getLoc();
438 
439  // Allocate buffer to store input tile to.
440  Value vscale =
441  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
442  Value minTileSlices = rewriter.create<arith::ConstantOp>(
443  loc, rewriter.getIndexAttr(tileType.getDimSize(0)));
444  Value c0 =
445  rewriter.create<arith::ConstantOp>(loc, rewriter.getIndexAttr(0));
446  Value numTileSlices =
447  rewriter.create<arith::MulIOp>(loc, vscale, minTileSlices);
448  auto bufferType =
449  MemRefType::get({ShapedType::kDynamic, ShapedType::kDynamic},
450  tileType.getElementType());
451  auto buffer = rewriter.create<memref::AllocaOp>(
452  loc, bufferType, ValueRange{numTileSlices, numTileSlices});
453 
454  Value input = transposeOp.getVector();
455 
456  // Store input tile.
457  auto tileStoreOp = rewriter.create<arm_sme::TileStoreOp>(
458  loc, input, buffer, ValueRange{c0, c0});
459 
460  // Reload input tile vertically.
461  rewriter.replaceOpWithNewOp<arm_sme::TileLoadOp>(
462  transposeOp, tileType, tileStoreOp.getBase(), tileStoreOp.getIndices(),
463  arm_sme::TileSliceLayout::Vertical);
464 
465  return success();
466  }
467 };

469 /// Conversion pattern for vector.outerproduct.
470 ///
471 /// If the vector.outerproduct is masked (and the mask is from a
472 /// vector.create_mask), then the mask is decomposed into two 1-D masks for the
473 /// operands.
474 ///
475 /// Example:
476 ///
477 /// %mask = vector.create_mask %dimA, %dimB : vector<[4]x[4]xi1>
478 /// %result = vector.mask %mask {
479 /// vector.outerproduct %vecA, %vecB
480 /// : vector<[4]xf32>, vector<[4]xf32>
481 /// } : vector<[4]x[4]xi1> -> vector<[4]x[4]xf32>
482 ///
483 /// is converted to:
484 ///
485 /// %maskA = vector.create_mask %dimA : vector<[4]xi1>
486 /// %maskB = vector.create_mask %dimB : vector<[4]xi1>
487 /// %result = arm_sme.outerproduct %vecA, %vecB masks(%maskA, %maskB)
488 /// : vector<[4]xf32>, vector<[4]xf32>
489 ///
490 /// Unmasked outerproducts can be directly replaced with the arm_sme op.
491 ///
492 /// Example:
493 ///
494 /// %result = vector.outerproduct %vecA, %vecB
495 /// : vector<[4]xf32>, vector<[4]xf32>
496 ///
497 /// is converted to:
498 ///
499 /// %result = arm_sme.outerproduct %vecA, %vecB
500 /// : vector<[4]xf32>, vector<[4]xf32>
501 ///
502 struct VectorOuterProductToArmSMELowering
503  : public OpRewritePattern<vector::OuterProductOp> {
504 
505  using OpRewritePattern<vector::OuterProductOp>::OpRewritePattern;
506 
507  LogicalResult matchAndRewrite(vector::OuterProductOp outerProductOp,
508  PatternRewriter &rewriter) const override {
509 
510  // We don't yet support lowering AXPY operations to SME. These could be
511  // lowered by masking out all but the first element of the LHS.
512  if (!isa<VectorType>(outerProductOp.getOperandTypeRHS()))
513  return outerProductOp.emitError("AXPY operations not supported");
514 
515  if (!arm_sme::isValidSMETileVectorType(
516  outerProductOp.getResultVectorType()))
517  return outerProductOp.emitError(
518  "outer product does not fit into SME tile");
519 
520  auto kind = outerProductOp.getKind();
521  if (kind != vector::CombiningKind::ADD)
522  return outerProductOp.emitError(
523  "unsupported kind (lowering to SME only supports ADD at the moment)");
524 
525  Value lhsMask = {};
526  Value rhsMask = {};
527  Operation *rootOp = outerProductOp;
528  auto loc = outerProductOp.getLoc();
529  if (outerProductOp.isMasked()) {
530  auto maskOp = outerProductOp.getMaskingOp();
531  rewriter.setInsertionPoint(maskOp);
532  rootOp = maskOp;
533  auto operandMasks = decomposeResultMask(loc, maskOp.getMask(), rewriter);
534  if (failed(operandMasks))
535  return failure();
536  std::tie(lhsMask, rhsMask) = *operandMasks;
537  }
538 
539  rewriter.replaceOpWithNewOp<arm_sme::OuterProductOp>(
540  rootOp, outerProductOp.getResultVectorType(), outerProductOp.getLhs(),
541  outerProductOp.getRhs(), lhsMask, rhsMask, outerProductOp.getAcc());
542 
543  return success();
544  }
545 
546  static FailureOr<std::pair<Value, Value>>
547  decomposeResultMask(Location loc, Value mask, PatternRewriter &rewriter) {
548  // Attempt to extract masks from vector.create_mask.
549  // TODO: Add support for other mask sources.
550  auto createMaskOp = mask.getDefiningOp<vector::CreateMaskOp>();
551  if (!createMaskOp)
552  return failure();
553 
554  auto maskType = createMaskOp.getVectorType();
555  Value lhsMaskDim = createMaskOp.getOperand(0);
556  Value rhsMaskDim = createMaskOp.getOperand(1);
557 
558  VectorType operandMaskType = VectorType::Builder(maskType).dropDim(0);
559  Value lhsMask =
560  rewriter.create<vector::CreateMaskOp>(loc, operandMaskType, lhsMaskDim);
561  Value rhsMask =
562  rewriter.create<vector::CreateMaskOp>(loc, operandMaskType, rhsMaskDim);
563 
564  return std::make_pair(lhsMask, rhsMask);
565  }

568 /// Lower `vector.extract` using `arm_sme.move_tile_slice_to_vector`.
569 ///
570 /// Example:
571 /// ```
572 /// %el = vector.extract %tile[%row, %col]: i32 from vector<[4]x[4]xi32>
573 /// ```
574 /// Becomes:
575 /// ```
576 /// %slice = arm_sme.move_tile_slice_to_vector %tile[%row]
577 /// : vector<[4]xi32> from vector<[4]x[4]xi32>
578 /// %el = vector.extract %slice[%col] : i32 from vector<[4]xi32>
579 /// ```
580 struct VectorExtractToArmSMELowering
581  : public OpRewritePattern<vector::ExtractOp> {
582  using OpRewritePattern<vector::ExtractOp>::OpRewritePattern;
583 
584  LogicalResult matchAndRewrite(vector::ExtractOp extractOp,
585  PatternRewriter &rewriter) const override {
586  VectorType sourceType = extractOp.getSourceVectorType();
587  if (!arm_sme::isValidSMETileVectorType(sourceType))
588  return failure();
589 
590  auto loc = extractOp.getLoc();
591  auto position = extractOp.getMixedPosition();
592 
593  Value sourceVector = extractOp.getVector();
594 
595  // Extract entire vector. Should be handled by folder, but just to be safe.
596  if (position.empty()) {
597  rewriter.replaceOp(extractOp, sourceVector);
598  return success();
599  }
600 
601  Value sliceIndex = vector::getAsValues(rewriter, loc, position[0]).front();
602  auto moveTileSliceToVector =
603  rewriter.create<arm_sme::MoveTileSliceToVectorOp>(loc, sourceVector,
604  sliceIndex);
605 
606  if (position.size() == 1) {
607  // Single index case: Extracts a 1D slice.
608  rewriter.replaceOp(extractOp, moveTileSliceToVector);
609  return success();
610  }
611 
612  // Two indices case: Extracts a single element.
613  assert(position.size() == 2);
614  rewriter.replaceOpWithNewOp<vector::ExtractOp>(
615  extractOp, moveTileSliceToVector, position[1]);
616 
617  return success();
618  }
619 };

622 /// `arm_sme.move_tile_slice_to_vector`.
623 ///
624 /// Example:
625 /// ```
626 /// %new_tile = vector.insert %el, %tile[%row, %col]
627 /// : i32 into vector<[4]x[4]xi32>
628 /// ```
629 /// Becomes:
630 /// ```

632 /// : vector<[4]xi32> from vector<[4]x[4]xi32>
633 /// %new_slice = vector.insert %el, %slice[%col] : i32 into vector<[4]xi32>
634 /// %new_tile = arm_sme.move_vector_to_tile_slice %new_slice, %tile, %row
635 /// : vector<[4]xi32> into vector<[4]x[4]xi32>
636 /// ```
637 struct VectorInsertToArmSMELowering
638  : public OpRewritePattern<vector::InsertOp> {
639  using OpRewritePattern<vector::InsertOp>::OpRewritePattern;
640 
641  LogicalResult matchAndRewrite(vector::InsertOp insertOp,
642  PatternRewriter &rewriter) const override {
643  VectorType resultType = insertOp.getResult().getType();
644 
645  if (!arm_sme::isValidSMETileVectorType(resultType))
646  return failure();
647 
648  auto loc = insertOp.getLoc();
649  auto position = insertOp.getMixedPosition();
650 
651  Value source = insertOp.getSource();
652 
653  // Overwrite entire vector with value. Should be handled by folder, but
654  // just to be safe.
655  if (position.empty()) {
656  rewriter.replaceOp(insertOp, source);
657  return success();
658  }
659 
660  Value tileSlice = source;
661  Value sliceIndex = vector::getAsValues(rewriter, loc, position[0]).front();
662  if (position.size() == 2) {
663  // Two indices case: Insert single element into tile.
664  // We need to first extract the existing slice and update the element.
665  tileSlice = rewriter.create<arm_sme::MoveTileSliceToVectorOp>(
666  loc, insertOp.getDest(), sliceIndex);
667  tileSlice = rewriter.create<vector::InsertOp>(loc, source, tileSlice,
668  position[1]);
669  }
670 
671  // Insert the slice into the destination tile.
672  rewriter.replaceOpWithNewOp<arm_sme::MoveVectorToTileSliceOp>(
673  insertOp, tileSlice, insertOp.getDest(), sliceIndex);
674  return success();
675  }

678 /// Lowers `vector.print` of a tile into a loop over the rows of the tile,
679 /// extracting them via `arm_sme.move_tile_slice_to_vector`, then printing with
680 /// a 1D `vector.print`.
681 ///
682 /// BEFORE:
683 /// ```mlir
684 /// vector.print %tile : vector<[4]x[4]xf32>
685 /// ```
686 /// AFTER:
687 /// ```mlir
688 /// %c0 = arith.constant 0 : index
689 /// %c1 = arith.constant 1 : index
690 /// %c4 = arith.constant 4 : index
691 /// %vscale = vector.vscale
692 /// %svl_s = arith.muli %c4, %vscale : index
693 /// scf.for %i = %c0 to %svl_s step %c1 {
694 /// %tile_slice = arm_sme.move_tile_slice_to_vector %tile[%i]
695 /// : vector<[4]xf32> from vector<[4]x[4]xf32>
696 /// vector.print %tile_slice : vector<[4]xf32>
697 /// }
698 /// ```
699 struct VectorPrintToArmSMELowering : public OpRewritePattern<vector::PrintOp> {
700  using OpRewritePattern<vector::PrintOp>::OpRewritePattern;
701 
702  LogicalResult matchAndRewrite(vector::PrintOp printOp,
703  PatternRewriter &rewriter) const override {
704  if (!printOp.getSource())
705  return failure();
706 
707  VectorType vectorType = dyn_cast<VectorType>(printOp.getPrintType());
708  if (!vectorType || !arm_sme::isValidSMETileVectorType(vectorType))
709  return failure();
710 
711  auto loc = printOp.getLoc();
712 
713  // Create a loop over the rows of the tile.
714  auto vscale = rewriter.create<vector::VectorScaleOp>(loc);
715  auto minTileRows =
716  rewriter.create<arith::ConstantIndexOp>(loc, vectorType.getDimSize(0));
717  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
718  auto upperBound = rewriter.create<arith::MulIOp>(loc, minTileRows, vscale);
719  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
720  auto forOp = rewriter.create<scf::ForOp>(loc, lowerBound, upperBound, step);
721  {
722  // Loop body.
723  rewriter.setInsertionPointToStart(forOp.getBody());
724  // Extract the current row from the tile.
725  Value rowIndex = forOp.getInductionVar();
726  auto tileSlice = rewriter.create<arm_sme::MoveTileSliceToVectorOp>(
727  loc, printOp.getSource(), rowIndex);
728  // Print the row with a 1D vector.print.
729  rewriter.create<vector::PrintOp>(loc, tileSlice,
730  printOp.getPunctuation());
731  }
732 
733  rewriter.eraseOp(printOp);
734  return success();
735  }
736 };
