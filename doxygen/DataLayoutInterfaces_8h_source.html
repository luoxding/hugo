106 /// An interface to be implemented by dialects that can have identifiers in the
107 /// data layout specification entries. Provides hooks for verifying the entry
108 /// validity and combining two entries.
109 class DataLayoutDialectInterface
110  : public DialectInterface::Base<DataLayoutDialectInterface> {
111 public:
112  DataLayoutDialectInterface(Dialect *dialect) : Base(dialect) {}
113 
114  /// Checks whether the given data layout entry is valid and reports any errors
115  /// at the provided location. Derived classes should override this.
116  virtual LogicalResult verifyEntry(DataLayoutEntryInterface entry,
117  Location loc) const {
118  return success();
119  }
120 
121  /// Default implementation of entry combination that combines identical
122  /// entries and returns null otherwise.
123  static DataLayoutEntryInterface
124  defaultCombine(DataLayoutEntryInterface outer,
125  DataLayoutEntryInterface inner) {
126  if (!outer || outer == inner)
127  return inner;
128  return {};
129  }
130 
131  /// Combines two entries with identifiers that belong to this dialect. Returns
132  /// the combined entry or null if the entries are not compatible. Derived
133  /// classes likely need to reimplement this.
134  virtual DataLayoutEntryInterface
135  combine(DataLayoutEntryInterface outer,
136  DataLayoutEntryInterface inner) const {
137  return defaultCombine(outer, inner);
138  }
139 };

145 /// The main mechanism for performing data layout queries. Instances of this
146 /// class can be created for an operation implementing DataLayoutOpInterface.
147 /// Upon construction, a layout spec combining that of the given operation with
148 /// all its ancestors will be computed and used to handle further requests. For
149 /// efficiency, results to all requests will be cached in this object.
150 /// Therefore, if the data layout spec for the scoping operation, or any of the
151 /// enclosing operations, changes, the cache is no longer valid. The user is
152 /// responsible creating a new DataLayout object after any spec change. In debug
153 /// mode, the cache validity is being checked in every request.
154 class DataLayout {
155 public:
156  explicit DataLayout();
157  explicit DataLayout(DataLayoutOpInterface op);
158  explicit DataLayout(ModuleOp op);
159 
160  /// Returns the layout of the closest parent operation carrying layout info.
161  static DataLayout closest(Operation *op);
162 
163  /// Returns the size of the given type in the current scope.
164  llvm::TypeSize getTypeSize(Type t) const;
165 
166  /// Returns the size in bits of the given type in the current scope.
167  llvm::TypeSize getTypeSizeInBits(Type t) const;
168 
169  /// Returns the required alignment of the given type in the current scope.
170  uint64_t getTypeABIAlignment(Type t) const;
171 
172  /// Returns the preferred of the given type in the current scope.
173  uint64_t getTypePreferredAlignment(Type t) const;
174 
175  /// Returns the memory space used for AllocaOps.
176  Attribute getAllocaMemorySpace() const;
177 
178  /// Returns the natural alignment of the stack in bits. Alignment promotion of
179  /// stack variables should be limited to the natural stack alignment to
180  /// prevent dynamic stack alignment. Returns zero if the stack alignment is
181  /// unspecified.
182  uint64_t getStackAlignment() const;
183 
184 private:
185  /// Combined layout spec at the given scope.
186  const DataLayoutSpecInterface originalLayout;
187 
188 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
189  /// List of enclosing layout specs.
190  SmallVector<DataLayoutSpecInterface, 2> layoutStack;
191 #endif
192 
193  /// Asserts that the cache is still valid. Expensive in debug mode. No-op in
194  /// release mode.
195  void checkValid() const;
196 
197  /// Operation defining the scope of requests.
198  Operation *scope;
199 
200  /// Caches for individual requests.
201  mutable DenseMap<Type, llvm::TypeSize> sizes;
202  mutable DenseMap<Type, llvm::TypeSize> bitsizes;
203  mutable DenseMap<Type, uint64_t> abiAlignments;
204  mutable DenseMap<Type, uint64_t> preferredAlignments;
205 
206  /// Cache for alloca memory space.
207  mutable std::optional<Attribute> allocaMemorySpace;
208 
209  /// Cache for stack alignment.
210  mutable std::optional<uint64_t> stackAlignment;
211 };
