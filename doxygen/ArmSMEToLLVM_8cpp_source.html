55 /// Lower 'arm_sme.zero' to SME intrinsics.
56 ///
57 /// BEFORE:
58 /// ```mlir
59 /// %v = arm_sme.zero {tile_id = 0 : i32} : vector<[4]x[4]xi32>
60 /// ```
61 ///
62 /// AFTER:
63 /// ```mlir
64 /// "arm_sme.intr.zero"() <{tile_mask = 17 : i32}> : () -> ()
65 /// %v = arm_sme.materialize_ssa_tile : vector<[4]x[4]xi32>
66 /// ```
67 ///
68 /// The 'arm_sme.materialize_ssa_tile' (which models the return) will fold away
69 /// once all ArmSME ops have been converted to LLVM intrinsics.
70 struct ZeroOpConversion : public ConvertOpToLLVMPattern<arm_sme::ZeroOp> {
71  using ConvertOpToLLVMPattern<arm_sme::ZeroOp>::ConvertOpToLLVMPattern;
72 
73  LogicalResult
74  matchAndRewrite(arm_sme::ZeroOp zero, OpAdaptor adaptor,
75  ConversionPatternRewriter &rewriter) const override {
76  auto loc = zero.getLoc();
77 
78  unsigned tileElementWidth =
79  zero.getVectorType().getElementType().getIntOrFloatBitWidth();
80 
81  auto tileId = getTileIdOrError(zero);
82  if (!tileId)
83  return failure();
84 
85  // Get the base mask for tile based on the element size.
86  // The base mask is just the mask to zero the first tile (of a size).
87  // These masks are derived from:
88  // https://developer.arm.com/documentation/ddi0602/2022-06/SME-Instructions/ZERO--Zero-a-list-of-64-bit-element-ZA-tiles-
89  auto baseMaskForSize = [&] {
90  switch (tileElementWidth) {
91  case 8:
92  // Zeroing the 8-bit ZA0.B tile is equivalent to zeroing all eight
93  // 64-bit element tiles named ZA0.D to ZA7.D.
94  return 0b1111'1111;
95  case 16:
96  // Zeroing the 16-bit ZA0.H tile is equivalent to zeroing 64-bit element
97  // tiles named ZA0.D, ZA2.D, ZA4.D, and ZA6.D.
98  // Shift this left once for ZA1.H.
99  return 0b0101'0101;
100  case 32:
101  // Zeroing the 32-bit ZA0.S tile is equivalent to zeroing 64-bit
102  // element tiles named ZA0.D and ZA4.D.
103  // Shift left by 1, 2, or 3 respectively for ZA1.S, ZA2.S, ZA3.S.
104  return 0b0001'0001;
105  case 64:
106  // Zeroing one of the a 64-bit tiles ZA0.D to ZA7.D just requires
107  // setting the bit for that tile.
108  return 0b0000'0001;
109  default:
110  llvm_unreachable("bad element size");
111  }
112  }();
113 
114  // The actual mask is just the base mask shifted by the tile ID.
115  // This will be folded to a constant after tile allocation.
116  //
117  // The shift is just derived from the layout of the tiles, and that the tile
118  // ID is the index of the tile. For example, looking at the 32-bit ZAx.S
119  // tiles:
120  //
121  // ZA0.S = ZA0.D and ZA4.D
122  // * Tile ID -> 0
123  // * Mask -> 00010001 = (00010001 << 0)
124  // ZA1.S = ZA1.D and ZA5.D
125  // * Tile ID -> 1
126  // * Mask -> 00100010 = (00010001 << 1)
127  // ZA2.S = ZA2.D and ZA6.D
128  // * Tile ID -> 2
129  // * Mask -> 01000100 = (00010001 << 2)
130  // ZA3.S = ZA3.D and ZA7.D
131  // * Tile ID -> 3
132  // * Mask -> 10001000 = (00010001 << 3)
133  //
134  // This holds for all tile sizes.
135  int32_t zeroMask = baseMaskForSize << int32_t(tileId.getInt());
136  rewriter.create<arm_sme::aarch64_sme_zero>(
137  loc, rewriter.getI32IntegerAttr(zeroMask));
138 
139  // Create a placeholder op to preserve dataflow.
140  rewriter.replaceOpWithNewOp<arm_sme::MaterializeSSATileOp>(
141  zero, zero.getVectorType());
142 
143  return success();
144  }
145 };

147 /// Lower `arm_sme.load_tile_slice` to SME intrinsics.
148 struct LoadTileSliceConversion
149  : public ConvertOpToLLVMPattern<arm_sme::LoadTileSliceOp> {
150  using ConvertOpToLLVMPattern<
151  arm_sme::LoadTileSliceOp>::ConvertOpToLLVMPattern;
152 
153  LogicalResult
154  matchAndRewrite(arm_sme::LoadTileSliceOp loadTileSliceOp,
155  arm_sme::LoadTileSliceOp::Adaptor adaptor,
156  ConversionPatternRewriter &rewriter) const override {
157  auto loc = loadTileSliceOp.getLoc();
158  auto tileId = getTileIdOrError(loadTileSliceOp);
159  if (!tileId)
160  return failure();
161 
162  Value ptr = this->getStridedElementPtr(loc, loadTileSliceOp.getMemRefType(),
163  adaptor.getBase(),
164  adaptor.getIndices(), rewriter);
165 
166  auto tileSlice = loadTileSliceOp.getTileSliceIndex();
167 
168  // Cast tile slice to i32 for intrinsic.
169  auto tileSliceI32 = rewriter.create<arith::IndexCastUIOp>(
170  loc, rewriter.getI32Type(), tileSlice);
171 
172  // Create all active predicate mask.
173  auto maskOp = loadTileSliceOp.getMask();
174 
175  auto tileType = loadTileSliceOp.getVectorType();
176  auto tileElementType = tileType.getElementType();
177  unsigned tileElementWidth = tileElementType.getIntOrFloatBitWidth();
178  arm_sme::TileSliceLayout layout = loadTileSliceOp.getLayout();
179 
180  // Create 'arm_sme.intr.ld1*.(horiz|vert)' intrinsic to load ZA tile slice.
181  if (layout == arm_sme::TileSliceLayout::Horizontal) {
182  switch (tileElementWidth) {
183  default:
184  llvm_unreachable("unexpected element type!");
185  case 8:
186  rewriter.create<arm_sme::aarch64_sme_ld1b_horiz>(loc, maskOp, ptr,
187  tileId, tileSliceI32);
188  break;
189  case 16:
190  rewriter.create<arm_sme::aarch64_sme_ld1h_horiz>(loc, maskOp, ptr,
191  tileId, tileSliceI32);
192  break;
193  case 32:
194  rewriter.create<arm_sme::aarch64_sme_ld1w_horiz>(loc, maskOp, ptr,
195  tileId, tileSliceI32);
196  break;
197  case 64:
198  rewriter.create<arm_sme::aarch64_sme_ld1d_horiz>(loc, maskOp, ptr,
199  tileId, tileSliceI32);
200  break;
201  case 128:
202  rewriter.create<arm_sme::aarch64_sme_ld1q_horiz>(loc, maskOp, ptr,
203  tileId, tileSliceI32);
204  break;
205  }
206  } else {
207  switch (tileElementWidth) {
208  default:
209  llvm_unreachable("unexpected element type!");
210  case 8:
211  rewriter.create<arm_sme::aarch64_sme_ld1b_vert>(loc, maskOp, ptr,
212  tileId, tileSliceI32);
213  break;
214  case 16:
215  rewriter.create<arm_sme::aarch64_sme_ld1h_vert>(loc, maskOp, ptr,
216  tileId, tileSliceI32);
217  break;
218  case 32:
219  rewriter.create<arm_sme::aarch64_sme_ld1w_vert>(loc, maskOp, ptr,
220  tileId, tileSliceI32);
221  break;
222  case 64:
223  rewriter.create<arm_sme::aarch64_sme_ld1d_vert>(loc, maskOp, ptr,
224  tileId, tileSliceI32);
225  break;
226  case 128:
227  rewriter.create<arm_sme::aarch64_sme_ld1q_vert>(loc, maskOp, ptr,
228  tileId, tileSliceI32);
229  break;
230  }
231  }
232 
233  // The load intrinsics have no result, replace 'arm_sme.tile_load' with
234  // the input tile to preserve dataflow.
235  rewriter.replaceOp(loadTileSliceOp, loadTileSliceOp.getTile());
236 
237  return success();
238  }

331 /// Lower `arm_sme.move_vector_to_tile_slice` to SME intrinsics.
332 struct MoveVectorToTileSliceConversion
333  : public ConvertOpToLLVMPattern<arm_sme::MoveVectorToTileSliceOp> {
334  using ConvertOpToLLVMPattern<
335  arm_sme::MoveVectorToTileSliceOp>::ConvertOpToLLVMPattern;
336 
337  LogicalResult
338  matchAndRewrite(arm_sme::MoveVectorToTileSliceOp moveVectorToTileSliceOp,
339  arm_sme::MoveVectorToTileSliceOp::Adaptor adaptor,
340  ConversionPatternRewriter &rewriter) const override {
341  auto loc = moveVectorToTileSliceOp.getLoc();
342  auto tileType = moveVectorToTileSliceOp.getTileType();
343 
344  auto tileId = getTileIdOrError(moveVectorToTileSliceOp);
345  if (!tileId)
346  return failure();
347 
348  auto tileSlice = moveVectorToTileSliceOp.getTileSliceIndex();
349 
350  // Cast tile slice from index to i32 for intrinsic.
351  auto tileSliceI32 = rewriter.create<arith::IndexCastUIOp>(
352  loc, rewriter.getI32Type(), tileSlice);
353 
354  // Create all active predicate mask.
355  auto one = rewriter.create<arith::ConstantOp>(
356  loc, rewriter.getI1Type(),
357  rewriter.getIntegerAttr(rewriter.getI1Type(), 1));
358  auto predTy = VectorType::get(tileType.getShape()[0], rewriter.getI1Type(),
359  /*scalableDims=*/{true});
360  auto allActiveMask = rewriter.create<vector::SplatOp>(loc, predTy, one);
361 
362  // Create 'arm_sme.intr.write.(horiz|vert)' to write vector to tile slice.
363  switch (moveVectorToTileSliceOp.getLayout()) {
364  case arm_sme::TileSliceLayout::Horizontal:
365  rewriter.create<arm_sme::aarch64_sme_write_horiz>(
366  loc, tileId, tileSliceI32, allActiveMask,
367  moveVectorToTileSliceOp.getVector());
368  break;
369  case arm_sme::TileSliceLayout::Vertical:
370  rewriter.create<arm_sme::aarch64_sme_write_vert>(
371  loc, tileId, tileSliceI32, allActiveMask,
372  moveVectorToTileSliceOp.getVector());
373  break;
374  }
375 
376  // Intrinsic has no result, replace 'arm_sme.move_vector_to_tile_slice' with
377  // the input tile to preserve dataflow.
378  rewriter.replaceOp(moveVectorToTileSliceOp,
379  moveVectorToTileSliceOp.getTile());
380 
381  return success();
382  }

385 /// Lower `arm_sme.move_tile_slice_to_vector` to SME intrinsics.
386 struct MoveTileSliceToVectorConversion
387  : public ConvertOpToLLVMPattern<arm_sme::MoveTileSliceToVectorOp> {
388  using ConvertOpToLLVMPattern<
389  arm_sme::MoveTileSliceToVectorOp>::ConvertOpToLLVMPattern;
390 
391  LogicalResult
392  matchAndRewrite(arm_sme::MoveTileSliceToVectorOp moveTileSliceToVector,
393  OpAdaptor,
394  ConversionPatternRewriter &rewriter) const override {
395  auto loc = moveTileSliceToVector.getLoc();
396  auto sliceType = moveTileSliceToVector.getSliceType();
397  auto sliceIndex = moveTileSliceToVector.getTileSliceIndex();
398 
399  auto tileId = getTileIdOrError(moveTileSliceToVector);
400  if (!tileId)
401  return failure();
402 
403  // Create an 'all true' predicate for the tile slice.
404  auto predicateType = sliceType.cloneWith({}, rewriter.getI1Type());
405  auto allTruePredicate = rewriter.create<arith::ConstantOp>(
406  loc, DenseElementsAttr::get(predicateType, true));
407 
408  // Zero destination/fallback for tile slice extraction.
409  auto zeroVector = rewriter.create<arith::ConstantOp>(
410  loc, sliceType, rewriter.getZeroAttr(sliceType));
411 
412  // Cast tile slice from index to i32 for intrinsic.
413  auto sliceIndexI32 = rewriter.create<arith::IndexCastOp>(
414  loc, rewriter.getI32Type(), sliceIndex);
415 
416  // Create 'arm_sme.intr.read.(horiz|vert)' to extract the tile slice.
417  switch (moveTileSliceToVector.getLayout()) {
418  case arm_sme::TileSliceLayout::Horizontal:
419  rewriter.replaceOpWithNewOp<arm_sme::aarch64_sme_read_horiz>(
420  moveTileSliceToVector, sliceType, zeroVector, allTruePredicate,
421  tileId, sliceIndexI32);
422  break;
423  case arm_sme::TileSliceLayout::Vertical:
424  rewriter.replaceOpWithNewOp<arm_sme::aarch64_sme_read_vert>(
425  moveTileSliceToVector, sliceType, zeroVector, allTruePredicate,
426  tileId, sliceIndexI32);
427  break;
428  }
429 
430  return success();
431  }

434 /// Lower `arm_sme.outerproduct` to SME MOPA intrinsics.
435 ///
436 /// Example:
437 ///
438 /// %0 = arm_sme.outerproduct %lhs, %rhs acc(%acc)
439 /// : vector<[4]xf32>, vector<[4]xf32>
440 ///
441 /// is converted to:
442 ///
443 /// "arm_sme.intr.mopa"(%ptrue_s, %ptrue_s, %lhs, %rhs) <{tile_id = 0 : i32}>
444 /// : (vector<[4]xi1>, vector<[4]xi1>, vector<[4]xf32>,
445 /// vector<[4]xf32>) -> ()
446 ///
447 /// Currently only supports FMOPA and BFMOPA (non-widening).
448 struct OuterProductOpConversion
449  : public ConvertOpToLLVMPattern<arm_sme::OuterProductOp> {
450  using ConvertOpToLLVMPattern<arm_sme::OuterProductOp>::ConvertOpToLLVMPattern;
451 
452  LogicalResult
453  matchAndRewrite(arm_sme::OuterProductOp outerProductOp,
454  arm_sme::OuterProductOp::Adaptor adaptor,
455  ConversionPatternRewriter &rewriter) const override {
456  auto tileId = getTileIdOrError(outerProductOp);
457  if (!tileId)
458  return failure();
459 
460  auto isSupportedType = [](VectorType vectorType) {
461  // TODO: the FP outer product instruction variants are predicated on
462  // different features [1]:
463  //
464  // * FMOPA (non-widening)
465  // * half-precision - +sme2p1,+sme-f16f16
466  // * single-precision - +sme
467  // * double-precision - +sme-f64f64
468  // * BFMOPA
469  // * half-precision - +sme2p1,+b16b16
470  //
471  // It should be possible to control lowering based on target features.
472  // [1]
473  // https://developer.arm.com/downloads/-/exploration-tools/feature-names-for-a-profile
474  if ((vectorType.getRank() != 2) || !vectorType.allDimsScalable())
475  return false;
476 
477  auto elementType = vectorType.getElementType();
478 
479  if (!elementType.isF16() && !elementType.isBF16() &&
480  !elementType.isF32() && !elementType.isF64())
481  return false;
482 
483  unsigned minNumElts = arm_sme::MinStreamingVectorLengthInBits /
484  vectorType.getElementTypeBitWidth();
485  if (vectorType.getShape() != ArrayRef<int64_t>({minNumElts, minNumElts}))
486  return false;
487 
488  return true;
489  };
490 
491  // TODO: Support CombiningKind::Sub for outer products.
492  if (outerProductOp.getKind() != arm_sme::CombiningKind::Add)
493  return outerProductOp.emitError("unsupported kind");
494 
495  auto resultVectorType = outerProductOp.getResultType();
496  if (!isSupportedType(resultVectorType))
497  return outerProductOp.emitError("unsupported type");
498 
499  auto loc = outerProductOp.getLoc();
500 
501  Value acc = outerProductOp.getAcc();
502  if (!acc)
503  // Initalize accumulator with zero.
504  acc = outerProductOp.createOpAndForwardTileId<arm_sme::ZeroOp>(
505  rewriter, loc, resultVectorType);
506 
507  Value lhsMask = outerProductOp.getLhsMask();
508  Value rhsMask = outerProductOp.getRhsMask();
509 
510  if (!lhsMask || !rhsMask) {
511  auto predTy =
512  outerProductOp.getLhsType().cloneWith({}, rewriter.getI1Type());
513  Value allActiveMask = rewriter.create<arith::ConstantOp>(
514  loc, DenseElementsAttr::get(predTy, true));
515  lhsMask = allActiveMask;
516  rhsMask = allActiveMask;
517  }
518 
519  // Create 'arm_sme.intr.mopa' outer product intrinsic.
520  rewriter.create<arm_sme::aarch64_sme_mopa>(loc, tileId, lhsMask, rhsMask,
521  outerProductOp.getLhs(),
522  outerProductOp.getRhs());
523 
524  // The outerproduct intrinsics have no result, replace
525  // 'arm_sme.outerproduct' with the input tile to preserve dataflow.
526  rewriter.replaceOp(outerProductOp, acc);
527 
528  return success();
529  }
