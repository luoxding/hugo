34 /// Pass to lower GPU Dialect to SPIR-V. The pass only converts the gpu.func ops
35 /// inside gpu.module ops. i.e., the function that are referenced in
36 /// gpu.launch_func ops. For each such function
37 ///
38 /// 1) Create a spirv::ModuleOp, and clone the function into spirv::ModuleOp
39 /// (the original function is still needed by the gpu::LaunchKernelOp, so cannot
40 /// replace it).
41 ///
42 /// 2) Lower the body of the spirv::ModuleOp.
43 struct GPUToSPIRVPass final : impl::ConvertGPUToSPIRVBase<GPUToSPIRVPass> {
44  explicit GPUToSPIRVPass(bool mapMemorySpace)
45  : mapMemorySpace(mapMemorySpace) {}
46  void runOnOperation() override;
47 
48 private:
49  bool mapMemorySpace;
50 };
51 
52 void GPUToSPIRVPass::runOnOperation() {
53  MLIRContext *context = &getContext();
54  ModuleOp module = getOperation();
55 
56  SmallVector<Operation *, 1> gpuModules;
57  OpBuilder builder(context);
58 
59  auto targetEnvSupportsKernelCapability = [](gpu::GPUModuleOp moduleOp) {
60  Operation *gpuModule = moduleOp.getOperation();
61  auto targetAttr = spirv::lookupTargetEnvOrDefault(gpuModule);
62  spirv::TargetEnv targetEnv(targetAttr);
63  return targetEnv.allows(spirv::Capability::Kernel);
64  };
65 
66  module.walk([&](gpu::GPUModuleOp moduleOp) {
67  // Clone each GPU kernel module for conversion, given that the GPU
68  // launch op still needs the original GPU kernel module.
69  // For Vulkan Shader capabilities, we insert the newly converted SPIR-V
70  // module right after the original GPU module, as that's the expectation of
71  // the in-tree Vulkan runner.
72  // For OpenCL Kernel capabilities, we insert the newly converted SPIR-V
73  // module inside the original GPU module, as that's the expectaion of the
74  // normal GPU compilation pipeline.
75  if (targetEnvSupportsKernelCapability(moduleOp)) {
76  builder.setInsertionPoint(moduleOp.getBody(),
77  moduleOp.getBody()->begin());
78  } else {
79  builder.setInsertionPoint(moduleOp.getOperation());
80  }
81  gpuModules.push_back(builder.clone(*moduleOp.getOperation()));
82  });
83 
84  // Run conversion for each module independently as they can have different
85  // TargetEnv attributes.
86  for (Operation *gpuModule : gpuModules) {
87  spirv::TargetEnvAttr targetAttr =
88  spirv::lookupTargetEnvOrDefault(gpuModule);
89 
90  // Map MemRef memory space to SPIR-V storage class first if requested.
91  if (mapMemorySpace) {
92  std::unique_ptr<ConversionTarget> target =
93  spirv::getMemorySpaceToStorageClassTarget(*context);
94  spirv::MemorySpaceToStorageClassMap memorySpaceMap =
95  targetEnvSupportsKernelCapability(
96  dyn_cast<gpu::GPUModuleOp>(gpuModule))
97  ? spirv::mapMemorySpaceToOpenCLStorageClass
98  : spirv::mapMemorySpaceToVulkanStorageClass;
99  spirv::MemorySpaceToStorageClassConverter converter(memorySpaceMap);
100 
101  RewritePatternSet patterns(context);
102  spirv::populateMemorySpaceToStorageClassPatterns(converter, patterns);
103 
104  if (failed(applyFullConversion(gpuModule, *target, std::move(patterns))))
105  return signalPassFailure();
106  }
107 
108  std::unique_ptr<ConversionTarget> target =
109  SPIRVConversionTarget::get(targetAttr);
110 
111  SPIRVConversionOptions options;
112  options.use64bitIndex = this->use64bitIndex;
113  SPIRVTypeConverter typeConverter(targetAttr, options);
114  populateMMAToSPIRVCoopMatrixTypeConversion(typeConverter,
115  this->useCoopMatrixNV);
116 
117  RewritePatternSet patterns(context);
118  populateGPUToSPIRVPatterns(typeConverter, patterns);
119  if (this->useCoopMatrixNV) {
120  populateGpuWMMAToSPIRVCoopMatrixNVConversionPatterns(typeConverter,
121  patterns);
122  } else {
123  populateGpuWMMAToSPIRVCoopMatrixKHRConversionPatterns(typeConverter,
124  patterns);
125  }
126 
127  // TODO: Change SPIR-V conversion to be progressive and remove the following
128  // patterns.
129  mlir::arith::populateArithToSPIRVPatterns(typeConverter, patterns);
130  populateMemRefToSPIRVPatterns(typeConverter, patterns);
131  populateFuncToSPIRVPatterns(typeConverter, patterns);
132 
133  if (failed(applyFullConversion(gpuModule, *target, std::move(patterns))))
134  return signalPassFailure();
135  }
136 
137  // For OpenCL, the gpu.func op in the original gpu.module op needs to be
138  // replaced with an empty func.func op with the same arguments as the gpu.func
139  // op. The func.func op needs gpu.kernel attribute set.
140  module.walk([&](gpu::GPUModuleOp moduleOp) {
141  if (targetEnvSupportsKernelCapability(moduleOp)) {
142  moduleOp.walk([&](gpu::GPUFuncOp funcOp) {
143  builder.setInsertionPoint(funcOp);
144  auto newFuncOp = builder.create<func::FuncOp>(
145  funcOp.getLoc(), funcOp.getName(), funcOp.getFunctionType());
146  auto entryBlock = newFuncOp.addEntryBlock();
147  builder.setInsertionPointToEnd(entryBlock);
148  builder.create<func::ReturnOp>(funcOp.getLoc());
149  newFuncOp->setAttr(gpu::GPUDialect::getKernelFuncAttrName(),
150  builder.getUnitAttr());
151  funcOp.erase();
152  });
153  }
154  });
155 }
